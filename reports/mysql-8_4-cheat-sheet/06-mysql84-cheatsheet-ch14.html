🟢 流程控制函式
-- CASE（兩種形式）
CASE status
  WHEN 'A' THEN 'Active'
  WHEN 'I' THEN 'Inactive'
  ELSE 'Unknown'
END

CASE WHEN score >= 90 THEN 'A'
     WHEN score >= 80 THEN 'B'
     ELSE 'C'
END

-- IF / IFNULL / NULLIF
IF(condition, true_val, false_val)
IFNULL(expr, default_val)         -- expr 為 NULL 時回傳 default_val
NULLIF(expr1, expr2)              -- 相等回 NULL，否則回 expr1
COALESCE(a, b, c)                 -- 回傳第一個非 NULL 值
🟢 字串函式
函式說明範例 → 結果
CONCAT(s1,s2,...)串接（任一為 NULL 則結果 NULL）CONCAT('A','B') → AB
CONCAT_WS(sep,s1,s2)以分隔符串接（跳過 NULL）CONCAT_WS('-','2025','04','01') → 2025-04-01
SUBSTRING(s,pos,len)子字串（pos 從 1 起算）SUBSTRING('MySQL',3,3) → SQL
LEFT(s,n) / RIGHT(s,n)左/右取 n 字元LEFT('Hello',3) → Hel
LENGTH(s)位元組長度LENGTH('中') → 3（utf8mb4）
CHAR_LENGTH(s)字元長度CHAR_LENGTH('中') → 1
LOWER(s) / UPPER(s)大小寫轉換
TRIM(s)移除前後空格TRIM(' hi ') → hi
TRIM(LEADING/TRAILING/BOTH x FROM s)移除指定字元TRIM(LEADING '0' FROM '007') → 7
LPAD(s,n,pad) / RPAD左/右填充至 n 字元LPAD('42',5,'0') → 00042
REPLACE(s,from,to)取代所有匹配REPLACE('aaa','a','b') → bbb
REVERSE(s)反轉REVERSE('abc') → cba
REPEAT(s,n)重複 n 次REPEAT('ab',3) → ababab
INSERT(s,pos,len,new)取代指定位置的字串INSERT('Hello',2,3,'i') → Hio
LOCATE(sub,s)找子字串位置（0=未找到）LOCATE('SQL','MySQL') → 3
FIND_IN_SET(s,list)在逗號分隔列表中找位置FIND_IN_SET('b','a,b,c') → 2
FORMAT(n,d)千分位格式化FORMAT(1234567.8,2) → 1,234,567.80
HEX(s) / UNHEX(s)十六進位編碼/解碼HEX('A') → 41
🔶 正規表達式函式（ICU 引擎）
REGEXP_LIKE('abc123', '[0-9]+')                    -- 1（匹配）

REGEXP_INSTR('Hello World', '[[:space:]]')          -- 6（位置）

🟢 日期計算與格式化
-- 日期加減
DATE_ADD('2025-01-15', INTERVAL 30 DAY)        -- 2025-02-14
DATE_SUB(NOW(), INTERVAL 1 HOUR)               -- 一小時前
'2025-01-15' + INTERVAL 3 MONTH                -- 簡寫語法

-- 日期差
DATEDIFF('2025-12-31', '2025-01-01')            -- 364（天數）
TIMESTAMPDIFF(YEAR, '1990-06-15', CURDATE())    -- 年齡
TIMESTAMPDIFF(MINUTE, start_time, end_time)    -- 分鐘差

-- 格式化
DATE_FORMAT(NOW(), '%Y/%m/%d %H:%i:%s')         -- 2025/04/01 14:30:00
DATE_FORMAT(NOW(), '%W, %M %e, %Y')            -- Tuesday, April 1, 2025

-- 字串→日期
STR_TO_DATE('01/04/2025', '%d/%m/%Y')           -- 2025-04-01

-- Unix 時間戳
UNIX_TIMESTAMP()                                  -- 目前 epoch 秒
FROM_UNIXTIME(1711987200)                         -- epoch → datetime

-- 時區轉換
CONVERT_TZ(NOW(), '+00:00', '+08:00')           -- UTC → 台北
INTERVAL 單位範例
MICROSECOND, SECOND, MINUTE, HOURINTERVAL 30 MINUTE
DAY, WEEK, MONTH, QUARTER, YEARINTERVAL 1 YEAR
複合：HOUR_MINUTE, DAY_HOUR 等INTERVAL '1:30' HOUR_MINUTE

🔶 窗口函式（Window Functions）
-- 基本語法
function_name() OVER (
    [PARTITION BY col1, col2]
    [ORDER BY col3 ASC|DESC]
    [frame_clause]
)
函式說明範例
ROW_NUMBER()連續編號（無重複）ROW_NUMBER() OVER (ORDER BY score DESC)
RANK()排名（同分跳號）1,2,2,4
DENSE_RANK()排名（同分不跳號）1,2,2,3
NTILE(n)分為 n 等份NTILE(4) OVER (ORDER BY score)
LAG(expr,n,default)前 n 列的值LAG(price,1,0) OVER (ORDER BY date)
LEAD(expr,n,default)後 n 列的值
FIRST_VALUE(expr)窗口內第一個值
LAST_VALUE(expr)窗口內最後一個值需搭配 RANGE BETWEEN ... AND UNBOUNDED FOLLOWING
NTH_VALUE(expr,n)窗口內第 n 個值
CUME_DIST()累積分布 (0~1)
PERCENT_RANK()百分比排名 (0~1)
-- Frame 語法
ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING      -- 前後各 1 列
ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW  -- 從頭到目前（預設）
RANGE BETWEEN INTERVAL 7 DAY PRECEDING AND CURRENT ROW  -- 時間範圍

-- 具名窗口（避免重複定義）
SELECT SUM(x) OVER w, AVG(x) OVER w
FROM t1
WINDOW w AS (PARTITION BY grp ORDER BY id);
⚠️LAST_VALUE() 預設 frame 為 RANGE BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW，只會到目前列——需改為 AND UNBOUNDED FOLLOWING 才能取到真正的最後值。
🟢 JSON 函式速查
類別函式範例
擷取-> (JSON_EXTRACT)data->'$.name' → "John"（含引號）
->> (JSON_UNQUOTE(EXTRACT))data->>'$.name' → John（無引號）
JSON_VALUE(doc,path RETURNING type)8.0.21+ 推薦，可搭配功能索引

JSON_INSERT(doc,path,val,...)僅新增（不覆蓋已存在的）
JSON_REPLACE(doc,path,val,...)僅覆蓋（不新增不存在的）

搜尋JSON_CONTAINS(doc,val[,path])doc 是否包含 val
JSON_CONTAINS_PATH(doc,one/all,paths)路徑是否存在
val MEMBER OF(json_array)值是否在陣列中
表格化JSON_TABLE(doc,path COLUMNS(...))JSON → 關聯式列
工具JSON_TYPE/VALID/DEPTH/LENGTH/KEYS中繼資料
JSON_PRETTY(doc)格式化輸出
JSON_STORAGE_SIZE/FREE(doc)儲存大小 / 部分更新可用空間

-- JSON_TABLE 範例
SELECT jt.* FROM orders,
  JSON_TABLE(order_data, '$.items[*]' COLUMNS(
    item_id INT PATH '$.id',
    qty     INT PATH '$.quantity',
    name    VARCHAR(100) PATH '$.name'
  )) AS jt;
🟢 轉型函式
CAST(expr AS type)
CONVERT(expr, type)
CONVERT(expr USING charset)

-- 常用型別
CAST('123' AS UNSIGNED)           -- 字串→整數
CAST('2025-04-01' AS DATE)        -- 字串→日期
CAST(price AS DECIMAL(10,2))      -- 浮點→精確
CAST(col AS CHAR CHARACTER SET utf8mb4)
CAST(json_col AS FLOAT)           -- 8.4 支援 FLOAT/DOUBLE

-- 支援的 CAST 目標型別
-- BINARY, CHAR, DATE, DATETIME, DECIMAL, DOUBLE, FLOAT,
-- JSON, NCHAR, REAL, SIGNED, TIME, UNSIGNED, YEAR
⚠️BINARY 作為字串轉型運算子（如 BINARY 'abc'）在 8.4 已棄用。改用 CAST(expr AS BINARY)。

-- AES 加密範例（建議使用 CBC 模式 + 隨機 IV）
SET block_encryption_mode = 'aes-256-cbc';
SET @key = SHA2('my_secret_key', 256);
SET @iv = RANDOM_BYTES(16);
SET @encrypted = AES_ENCRYPT('sensitive data', @key, @iv);
SET @decrypted = AES_DECRYPT(@encrypted, @key, @iv);

-- UUID 做主鍵的最佳實務（swap_flag=1 讓時間部分排在前面，索引友善）
CREATE TABLE t (
    id BINARY(16) DEFAULT (UUID_TO_BIN(UUID(), 1)) PRIMARY KEY,
    name VARCHAR(100)
);
SELECT BIN_TO_UUID(id, 1) AS uuid_str FROM t;
