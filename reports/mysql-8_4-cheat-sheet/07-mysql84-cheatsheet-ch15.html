🟢 DATABASE 操作

  DEFAULT CHARACTER SET utf8mb4

🟢 CREATE TABLE — 完整範例
CREATE TABLE orders (
    id          BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
    user_id     INT UNSIGNED NOT NULL,
    amount      DECIMAL(10,2) NOT NULL DEFAULT 0.00,
    status      ENUM('pending','paid','shipped','done') DEFAULT 'pending',
    memo        VARCHAR(500) COMMENT '訂單備註',
    created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME ON UPDATE CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX idx_user (user_id),
    INDEX idx_status_date (status, created_at),
    CONSTRAINT fk_user FOREIGN KEY (user_id)
        REFERENCES users(id) ON DELETE RESTRICT ON UPDATE CASCADE,
    CONSTRAINT chk_amount CHECK (amount >= 0)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
🔶 CREATE TABLE — GENERATED 欄位 & INVISIBLE 欄位
CREATE TABLE products (
    price      DECIMAL(10,2),
    qty        INT,
    -- 虛擬欄位：不佔磁碟，即時計算
    total_val  DECIMAL(12,2) GENERATED ALWAYS AS (price * qty) VIRTUAL,
    -- 儲存欄位：寫入磁碟，可建索引
    tier       VARCHAR(10) GENERATED ALWAYS AS (
        CASE WHEN price >= 1000 THEN 'high' ELSE 'low' END
    ) STORED,
    -- INVISIBLE 欄位：SELECT * 不顯示
    trace_id   VARCHAR(64) INVISIBLE
);
🟢 ALTER TABLE 常用操作
-- 欄位操作
ALTER TABLE t ADD COLUMN email VARCHAR(255) NOT NULL AFTER name;
ALTER TABLE t MODIFY COLUMN email VARCHAR(500);
ALTER TABLE t CHANGE COLUMN old_name new_name INT;
ALTER TABLE t RENAME COLUMN old_name TO new_name;
ALTER TABLE t DROP COLUMN email;

-- 索引操作
ALTER TABLE t ADD INDEX idx_name (col1, col2);
ALTER TABLE t ADD UNIQUE INDEX uk_email (email);
ALTER TABLE t DROP INDEX idx_name;
ALTER TABLE t RENAME INDEX old_idx TO new_idx;
ALTER TABLE t ALTER INDEX idx INVISIBLE;

-- 外鍵 / CHECK
ALTER TABLE t ADD CONSTRAINT fk FOREIGN KEY (dept_id) REFERENCES depts(id);
ALTER TABLE t DROP FOREIGN KEY fk;
ALTER TABLE t ADD CONSTRAINT chk CHECK (age >= 0);

-- 字元集轉換（會轉換欄位資料）
ALTER TABLE t CONVERT TO CHARACTER SET utf8mb4;

-- 碎片整理（重建表）
ALTER TABLE t FORCE;
🔶 ALTER TABLE — ALGORITHM & LOCK
ALGORITHM說明速度
INSTANT8.4 預設。僅修改中繼資料⚡ 極快

🔶 CREATE INDEX — 函式索引・多值索引
-- 函式索引（表達式須用括號）
CREATE INDEX idx_upper ON t ((UPPER(name)));
CREATE INDEX idx_year ON orders ((YEAR(created_at)));

-- 多值索引（JSON 陣列）
CREATE INDEX idx_tags ON articles ((CAST(tags->'$' AS CHAR(50) ARRAY)));

-- 不可見索引
CREATE INDEX idx_test ON t (col) INVISIBLE;

🟢 JOIN 類型速查
類型說明
[INNER] JOIN內連接（僅匹配行）
LEFT [OUTER] JOIN左表全部 + 右表匹配
RIGHT [OUTER] JOIN右外連接
CROSS JOIN笛卡爾積
NATURAL JOIN自動以同名欄位連接
STRAIGHT_JOIN強制依書寫順序
SELECT o.id, u.name, o.amount
FROM orders o
INNER JOIN users u ON o.user_id = u.id
LEFT JOIN payments p ON o.id = p.order_id
WHERE o.status = 'paid'
ORDER BY o.created_at DESC LIMIT 20;
🔶 LATERAL 衍生表 & 集合操作
-- LATERAL：子查詢可參照前面的表
SELECT u.name, recent.*
FROM users u,
LATERAL (
    SELECT * FROM orders WHERE user_id = u.id
    ORDER BY created_at DESC LIMIT 3
) AS recent;

-- UNION / INTERSECT / EXCEPT（8.4 完整支援）
SELECT id FROM a UNION ALL SELECT id FROM b;
SELECT id FROM a INTERSECT SELECT id FROM b;
SELECT id FROM a EXCEPT SELECT id FROM b;
🟢 WITH（CTE）— 非遞迴 & 遞迴
-- 非遞迴 CTE
WITH top_users AS (
    SELECT user_id, SUM(amount) AS total
    FROM orders GROUP BY user_id HAVING total > 10000
)
SELECT u.name, t.total
FROM top_users t JOIN users u ON t.user_id = u.id;

-- 遞迴 CTE（組織架構樹）
WITH RECURSIVE org_tree AS (
    SELECT id, name, manager_id, 0 AS depth
    FROM employees WHERE manager_id IS NULL
    UNION ALL
    SELECT e.id, e.name, e.manager_id, t.depth + 1
    FROM employees e JOIN org_tree t ON e.manager_id = t.id
)
SELECT * FROM org_tree;
⚠️遞迴深度上限 cte_max_recursion_depth = 1000。遞迴部分不可用聚合、窗口函式、GROUP BY、ORDER BY、DISTINCT。
🟢 INSERT
-- 基本插入
INSERT INTO t (col1, col2) VALUES (1, 'a'), (2, 'b');
INSERT INTO t SET col1 = 1, col2 = 'a';

-- INSERT ... SELECT
INSERT INTO archive SELECT * FROM orders WHERE created_at < '2024-01-01';
🔶 INSERT ... ON DUPLICATE KEY UPDATE（8.4 新語法）
-- ⚠️ VALUES() 函式已棄用，8.4 推薦 row alias 語法
INSERT INTO daily_stats (date, views, clicks)
VALUES ('2025-03-15', 100, 20) AS new_row
ON DUPLICATE KEY UPDATE
    views  = new_row.views,
    clicks = new_row.clicks;

-- 含 column alias
INSERT INTO daily_stats (date, views, clicks)
VALUES ('2025-03-15', 100, 20) AS new_row(d, v, c)
ON DUPLICATE KEY UPDATE
    views = v, clicks = c;
🟢 UPDATE & DELETE
-- 單表 UPDATE
UPDATE orders SET status = 'cancelled'
WHERE status = 'pending' AND created_at < '2025-01-01'
ORDER BY created_at LIMIT 1000;

-- 多表 UPDATE（JOIN）
UPDATE orders o JOIN users u ON o.user_id = u.id
SET o.discount = 0.1 WHERE u.vip_level >= 3;

-- 單表 DELETE
DELETE FROM logs
WHERE created_at < '2024-01-01'
ORDER BY created_at LIMIT 10000;

-- 多表 DELETE（刪除沒有訂單的使用者）
DELETE u FROM users u
LEFT JOIN orders o ON u.id = o.user_id
WHERE o.id IS NULL;
⚠️UPDATE SET 賦值由左到右求值——SET a=a+1, b=a 中 b 得到 a 的已更新值。多表 UPDATE/DELETE 不支援 ORDER BY + LIMIT。
🔶 REPLACE & LOAD DATA INFILE
-- REPLACE（若 PK/UNIQUE 衝突：DELETE + INSERT）
REPLACE INTO t (id, name) VALUES (1, 'Alice');
-- ⚠️ 建議大多數情況用 INSERT ... ON DUPLICATE KEY UPDATE

-- LOAD DATA INFILE（匯入 CSV）
LOAD DATA INFILE '/var/lib/mysql-files/data.csv'
INTO TABLE t
CHARACTER SET utf8mb4
FIELDS TERMINATED BY ',' ENCLOSED BY '"'
LINES TERMINATED BY '\n'
IGNORE 1 LINES
(name, email, @birth)
SET birthday = STR_TO_DATE(@birth, '%Y/%m/%d');

交易與鎖定

🟢 交易控制
START TRANSACTION;    -- 或 BEGIN
-- ... 操作 ...
COMMIT;               -- 提交
ROLLBACK;             -- 回滾

-- 修飾子
START TRANSACTION WITH CONSISTENT SNAPSHOT;  -- InnoDB 一致快照
START TRANSACTION READ ONLY;                 -- 唯讀（啟用優化）

-- autocommit 控制
SET autocommit = 0;   -- 關閉（需明確 COMMIT）
SET autocommit = 1;   -- 開啟（預設）
🚨交易不可巢狀——開始新交易會隱式提交當前交易。DDL（CREATE/ALTER/DROP TABLE）會觸發隱式 COMMIT。
🟢 SAVEPOINT
START TRANSACTION;
INSERT INTO orders ...;
SAVEPOINT before_payment;
UPDATE accounts SET balance = balance - 100 WHERE id = 1;
-- 若扣款失敗：
ROLLBACK TO SAVEPOINT before_payment;
-- 訂單仍在，可選其他付款方式
UPDATE accounts SET balance = balance - 100 WHERE id = 2;
COMMIT;
🟢 隔離等級
SET [GLOBAL|SESSION] TRANSACTION ISOLATION LEVEL
    {READ UNCOMMITTED | READ COMMITTED | REPEATABLE READ | SERIALIZABLE};

SELECT @@transaction_isolation;  -- 查看目前等級
隔離等級髒讀不可重複讀幻讀
READ UNCOMMITTED✅✅✅
READ COMMITTED❌✅✅
REPEATABLE READ（預設）❌❌InnoDB gap lock 避免
SERIALIZABLE❌❌❌
🔶 鎖定讀取：FOR UPDATE / FOR SHARE
-- 排他鎖（其他 session 不能修改或鎖定讀取）
SELECT * FROM orders WHERE id = 100 FOR UPDATE;

-- 共享鎖（允許其他 session 讀取）
SELECT * FROM orders WHERE id = 100 FOR SHARE;

-- NOWAIT：被鎖則立即報錯
SELECT * FROM tasks WHERE status = 'pending' FOR UPDATE NOWAIT;

-- SKIP LOCKED：跳過已鎖定行（工作佇列首選）
SELECT * FROM job_queue WHERE status = 'pending'
LIMIT 1 FOR UPDATE SKIP LOCKED;

-- OF：僅鎖指定表
SELECT * FROM orders o JOIN items i ON o.id = i.order_id
FOR UPDATE OF o;
🔶 XA 分散式交易
XA START 'order-12345';
INSERT INTO orders ...;
XA END 'order-12345';
XA PREPARE 'order-12345';
XA COMMIT 'order-12345';
-- 或 XA ROLLBACK 'order-12345';

XA RECOVER;  -- 查看已 PREPARE 的 XA 交易
💡8.4 新特性：xa_detach_on_prepare=ON（預設），XA PREPARE 後交易與 session 脫鉤，其他連線可 COMMIT/ROLLBACK。

🟢 SHOW 語句速查
SHOW DATABASES;
SHOW TABLES [FROM db];

SHOW GRANTS [FOR user];

⚠️8.4 更名：SHOW BINARY LOG STATUS（取代 MASTER）、SHOW REPLICA STATUS（取代 SLAVE）。SHOW PROCESSLIST 已棄用，改用 performance_schema.processlist。
🔶 FLUSH & 表維護
FLUSH PRIVILEGES;             -- 重載授權表
FLUSH BINARY LOGS;            -- 輪轉 binlog
FLUSH TABLES WITH READ LOCK;  -- 全域讀鎖

ANALYZE TABLE t;              -- 更新索引統計
ANALYZE TABLE t UPDATE HISTOGRAM ON amount WITH 100 BUCKETS AUTO UPDATE;
OPTIMIZE TABLE t;             -- 碎片整理
CHECK TABLE t;
REPAIR TABLE t;               -- 不適用 InnoDB
🟢 Prepared Statements & KILL
-- 預處理語句
PREPARE stmt FROM 'SELECT * FROM users WHERE id = ?';
SET @id = 42;
EXECUTE stmt USING @id;
DEALLOCATE PREPARE stmt;

-- 終止連線/查詢
KILL 12345;          -- 終止連線 + 查詢
KILL QUERY 12345;    -- 僅終止查詢
