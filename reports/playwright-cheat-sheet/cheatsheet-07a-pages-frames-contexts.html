📄 頁面導航（Navigation）

Node.js
await page.goto('https://example.com');
await page.goto('/dashboard');  // 使用 baseURL

// waitUntil 選項
await page.goto(url, { waitUntil: 'load' });         // 預設
await page.goto(url, { waitUntil: 'domcontentloaded' });
await page.goto(url, { waitUntil: 'networkidle' });   // ⚠️ 不建議
await page.goto(url, { waitUntil: 'commit' });        // 最快

// 等待載入狀態
await page.waitForLoadState('load');
await page.waitForLoadState('domcontentloaded');
await page.waitForLoadState('networkidle');

// 導航歷史
await page.goBack();
await page.goForward();
await page.reload();

Python
page.goto("https://example.com")
page.goto(url, wait_until="domcontentloaded")
page.wait_for_load_state("load")
page.go_back()
page.go_forward()
page.reload()
⚠️ 避免使用 networkidle，它對 WebSocket 或長輪詢不可靠。使用 web-first 斷言等待特定元素出現更佳。

📑 多頁面/新分頁處理

Node.js
// 等待新分頁開啟
const pagePromise = context.waitForEvent('page');
await page.getByText('Open new tab').click();
const newPage = await pagePromise;
await newPage.waitForLoadState();
console.log(await newPage.title());

// 等待彈出視窗（popup）
const popupPromise = page.waitForEvent('popup');
await page.getByText('Open popup').click();
const popup = await popupPromise;

// 取得所有頁面
const allPages = context.pages();

Python
with context.expect_page() as new_page_info:
    page.get_by_text("Open new tab").click()
new_page = new_page_info.value
new_page.wait_for_load_state()

with page.expect_popup() as popup_info:
    page.get_by_text("Open popup").click()
popup = popup_info.value

all_pages = context.pages

🖼️ Frames（iframe）

Node.js
// 推薦方式：FrameLocator（可串接 Locator 方法）
const frame = page.frameLocator('#my-iframe');
await frame.getByRole('button', { name: '送出' }).click();

// 也可用 locator → contentFrame()
const frameElement = page.locator('#my-iframe');
const content = frameElement.contentFrame();
await content.getByText('Hello').click();

// 巢狀 iframe
const inner = page
  .frameLocator('#outer-frame')
  .frameLocator('#inner-frame');
await inner.getByRole('button').click();

// 反向：從 FrameLocator 取得 Locator
const iframeLocator = frame.owner();

Python
frame = page.frame_locator("#my-iframe")
frame.get_by_role("button", name="送出").click()

# locator → contentFrame
content = page.locator("#my-iframe").content_frame
content.get_by_text("Hello").click()

# 巢狀
inner = (page.frame_locator("#outer")
             .frame_locator("#inner"))
inner.get_by_role("button").click()
💡 frameLocator 不回傳 frame 物件，而是允許在 iframe 內使用定位器方法。是處理 iframe 的推薦方式。

🔒 瀏覽器上下文（Browser Context）

Node.js
// 建立隔離上下文
const context = await browser.newContext({
  viewport: { width: 1920, height: 1080 },
  locale: 'zh-TW',
  timezoneId: 'Asia/Taipei',
  geolocation: { longitude: 121.5, latitude: 25.0 },
  permissions: ['geolocation'],
  colorScheme: 'dark',
  userAgent: 'Custom UA',
  storageState: 'auth.json',
  httpCredentials: { username: 'user', password: 'pass' },
  ignoreHTTPSErrors: true,
  offline: false,
  proxy: { server: 'http://proxy:8080' },
  recordVideo: { dir: 'videos/', size: {width:1280,height:720} },
  recordHar: { path: 'network.har' },
});
const page = await context.newPage();

// 多使用者模擬
const adminCtx = await browser.newContext();
const userCtx = await browser.newContext();
const adminPage = await adminCtx.newPage();
const userPage = await userCtx.newPage();

Python
context = browser.new_context(
    viewport={"width": 1920, "height": 1080},
    locale="zh-TW",
    timezone_id="Asia/Taipei",
    geolocation={"longitude": 121.5, "latitude": 25.0},
    permissions=["geolocation"],
    color_scheme="dark",
)
page = context.new_page()
💡 每個上下文完全隔離（cookies、localStorage）。建立成本極低，用來模擬多使用者場景很理想。

⚡ JavaScript 執行

Node.js
// 在瀏覽器環境執行 JS
const title = await page.evaluate(() => document.title);

// 傳遞參數
const text = await page.evaluate(
  ([selector, prop]) => {
    const el = document.querySelector(selector);
    return el[prop];
  },
  ['.my-class', 'textContent']
);

// 取得 DOM handle
const handle = await page.evaluateHandle(
  () => document.body
);

// 在 Locator 元素上執行
const data = await page.locator('.card').evaluate(
  el => el.dataset.id
);

// 多個元素
const texts = await page.locator('.item').evaluateAll(
  elements => elements.map(e => e.textContent)
);

Python
title = page.evaluate("document.title")
data = page.locator(".card").evaluate("el => el.dataset.id")
texts = page.locator(".item").evaluate_all(
    "elements => elements.map(e => e.textContent)"
)

📡 頁面事件

事件說明
page.on('load')頁面載入完成
page.on('domcontentloaded')DOM 解析完成
page.on('request')發出請求
page.on('response')收到回應
page.on('requestfailed')請求失敗
page.on('dialog')出現對話框
page.on('download')觸發下載
page.on('popup')彈出視窗
page.on('console')控制台訊息
page.on('pageerror')頁面 JS 錯誤
page.on('close')頁面關閉
context.on('page')新頁面建立
Node.js
page.on('console', msg => {
  if (msg.type() === 'error') console.log(msg.text());
});
page.on('pageerror', err => console.log(err.message));

// 一次性監聽
page.once('dialog', d => d.accept());

// 移除監聽
const handler = msg => console.log(msg.text());
page.on('console', handler);
page.off('console', handler);

Python
page.on("console", lambda msg: print(msg.text))
page.on("pageerror", lambda err: print(err.message))
page.remove_listener("console", handler)

⏰ 時鐘（Clock）

Node.js
// 固定時間（推薦 — 所有日期/時間凍結）
await page.clock.setFixedTime(new Date('2024-02-02T10:00:00'));

// 安裝模擬時鐘 + 快進
await page.clock.install({ time: new Date('2024-02-02T08:00:00') });
await page.clock.fastForward('30:00');      // 快進 30 分鐘
await page.clock.fastForward(2000);         // 快進 2 秒

// 精確推進（觸發每個定時器）
await page.clock.runFor(2000);              // 推進 2 秒

// 暫停/恢復
await page.clock.pauseAt(new Date('2024-02-02T10:00:00'));
await page.clock.resume();

// 設定系統時間（不影響定時器）
await page.clock.setSystemTime(new Date('2024-12-25'));

Python
page.clock.set_fixed_time(datetime.datetime(2024, 2, 2, 10, 0, 0))
page.clock.install(time=datetime.datetime(2024, 2, 2, 8, 0, 0))
page.clock.fast_forward("30:00")
page.clock.run_for(2000)
💡 setFixedTime 最簡單，適合大多數場景。install + fastForward 適合需要測試 setTimeout/setInterval 的場景。

📸 視覺比較

Node.js（僅 Node.js 支援）
// 基本截圖比對
await expect(page).toHaveScreenshot('landing.png');
await expect(page.locator('.hero')).toHaveScreenshot();

// 自訂容差
await expect(page).toHaveScreenshot({
  maxDiffPixels: 100,       // 允許差異像素數
  maxDiffPixelRatio: 0.05,  // 允許差異比例
  threshold: 0.2,           // 像素門檻 (0=嚴格 1=寬鬆)
  animations: 'disabled',   // 停用動畫
  mask: [page.locator('.dynamic-ad')],
  maskColor: '#FF00FF',
  stylePath: './screenshot.css',  // 自訂遮罩 CSS
});

// 更新快照基準
// npx playwright test --update-snapshots

// 非圖像快照
expect(await page.textContent('.stats')).toMatchSnapshot('stats.txt');

🏷️ ARIA 快照測試

Node.js
await expect(page.getByRole('navigation')).toMatchAriaSnapshot(`
  - navigation:
    - link "首頁"
    - link "產品"
    - link "關於我們"
`);

// 使用正則
await expect(page.locator('main')).toMatchAriaSnapshot(`
  - heading /歡迎.*/ [level=1]
  - paragraph: /.*最新消息.*/
`);

Python
expect(page.get_by_role("navigation")).to_match_aria_snapshot("""
  - navigation:
    - link "首頁"
    - link "產品"
""")

YAML 格式規則
- role "name" [attribute=value]    # 基本格式
- heading "title" [level=1]        # 帶屬性
- link /pattern/                    # 正則名稱
- list:                             # 子元素（冒號 + 縮排）
  - listitem "Item 1"
  - listitem "Item 2"
/children: equal                    # 強制嚴格匹配（預設部分匹配）
產生快照
// 程式碼
const snapshot = await locator.ariaSnapshot();  // Node.js
snapshot = locator.aria_snapshot()                # Python
