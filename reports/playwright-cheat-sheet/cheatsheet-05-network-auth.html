🌐 網路攔截（Route）

Node.js
// 模擬回應（不發送實際請求）
await page.route('**/api/users', route => {
  route.fulfill({
    status: 200,
    contentType: 'application/json',
    json: [{ id: 1, name: 'Alice' }],
  });
});

// 阻止請求（圖片/CSS 等）
await page.route('**/*.{png,jpg,css}', route => route.abort());

// 修改請求（添加/移除 header）
await page.route('**/*', async route => {
  const headers = route.request().headers();
  headers['X-Custom'] = 'test';
  await route.continue_({ headers });
});

// 修改回應
await page.route('**/api/data', async route => {
  const response = await route.fetch();
  const json = await response.json();
  json.modified = true;
  await route.fulfill({ response, json });
});

// 移除路由
await page.unroute('**/api/users');
Python
# 模擬回應
page.route("**/api/users", lambda route: route.fulfill(
    status=200,
    content_type="application/json",
    json=[{"id": 1, "name": "Alice"}],
))

# 阻止請求
page.route("**/*.{png,jpg,css}", lambda route: route.abort())

# 修改請求 ⚠️ continue_ 避免關鍵字衝突
def handle_route(route):
    headers = route.request.headers.copy()
    headers["X-Custom"] = "test"
    route.continue_(headers=headers)
page.route("**/*", handle_route)

# 修改回應
def modify_response(route):
    response = route.fetch()
    json_data = response.json()
    json_data["modified"] = True
    route.fulfill(response=response, json=json_data)
page.route("**/api/data", modify_response)

# 移除路由
page.unroute("**/api/users")

🔗 URL 匹配模式

模式說明範例
*任意字元（不含 /）'**/api/*'
**任意字元（含 /）'**/api/**'
{a,b}選項列表'**/*.{png,jpg}'
RegExp正則表達式/api\/v\d+\/users/
函式自訂謂詞url => url.includes('api')
範例
'**/api/v1/users'       // 精確路徑
'**/api/**'             // api 下所有路徑
'**/*.{png,jpg,jpeg}'   // 所有圖片
/\/api\/v\d+\/.*/       // 正則匹配

📡 事件監聽

Node.js
// 請求事件
page.on('request', req => console.log(req.url()));
page.on('response', res => console.log(res.status()));
page.on('requestfailed', req => console.log(req.failure()));

// 等待特定回應
const responsePromise = page.waitForResponse('**/api/data');
await page.getByText('Load').click();
const response = await responsePromise;
console.log(await response.json());

// 等待請求
const requestPromise = page.waitForRequest('**/api/save');
await page.getByText('Save').click();
const request = await requestPromise;

Python
page.on("request", lambda req: print(req.url))
page.on("response", lambda res: print(res.status))

with page.expect_response("**/api/data") as resp_info:
    page.get_by_text("Load").click()
response = resp_info.value
print(response.json())

📼 HAR 錄製與回放

Node.js
// 錄製 HAR（update: true 首次錄製，之後回放）
await page.routeFromHAR('./data.har', {
  url: '**/api/**',
  update: true,         // true=錄製, false=回放
});

// 上下文層級
await context.routeFromHAR('./data.har', {
  url: '**/api/**',
  updateMode: 'full',   // 'full' | 'minimal'
  updateContent: 'embed',  // 'embed' | 'attach'
});

Python
page.route_from_har("./data.har",
    url="**/api/**", update=True)
context.route_from_har("./data.har",
    url="**/api/**", update=True)
💡 工作流：① update:true 首次執行錄製 HAR → ② 提交 HAR 檔案 → ③ update:false 後續測試用 HAR 回放，無需真實網路。

import { test as setup, expect } from '@playwright/test';

const authFile = 'playwright/.auth/user.json';

setup('authenticate', async ({ page }) => {
  await page.goto('https://example.com/login');

  await page.waitForURL('https://example.com/dashboard');

🔌 WebSocket 攔截

Node.js
await page.routeWebSocket('wss://example.com/ws', ws => {
  const server = ws.connectToServer();

  // 攔截客戶端訊息
  ws.onMessage(message => {
    if (message === 'ping') {
      ws.send('pong');  // 直接回覆客戶端
      return;
    }
    server.send(message);  // 轉發到伺服器
  });

  // 攔截伺服器訊息
  server.onMessage(message => {
    ws.send(message);  // 轉發到客戶端
  });
});

Python
def handle_ws(ws):
    server = ws.connect_to_server()
    ws.on_message(lambda msg: server.send(msg))
    server.on_message(lambda msg: ws.send(msg))
page.route_web_socket("wss://example.com/ws", handle_ws)

🎭 Mock API 完整模式

模式方法說明
完全模擬route.fulfill({ json })不發送網路請求
修改回應route.fetch() → 修改 → route.fulfill()先取得真實回應再修改
HAR 回放routeFromHAR()預錄資料回放
阻止route.abort()完全阻止請求
放行修改route.continue_()修改 headers/url 後繼續
fulfill 選項
route.fulfill({
  status: 200,             // HTTP 狀態碼
  contentType: 'text/html',// Content-Type
  headers: {},             // 回應 headers
  body: 'string',          // 字串回應
  json: {},                // JSON 回應（自動設定 Content-Type）
  path: './mock/data.json',// 從檔案回應
  response: fetchedResp,   // 基於 fetch 結果
});
