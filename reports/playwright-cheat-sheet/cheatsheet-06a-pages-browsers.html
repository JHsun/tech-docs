頁面（Pages）
📄 多頁面 & 彈出視窗
Node.js
// 建立新頁面
const newPage = await context.newPage();

// 等待新分頁（target="_blank" 觸發）
const pagePromise = context.waitForEvent('page');
await page.getByText('開啟新分頁').click();
const newPage = await pagePromise;
await newPage.waitForLoadState();

// 等待彈出視窗（window.open 觸發）
const popupPromise = page.waitForEvent('popup');
await page.getByText('開啟').click();
const popup = await popupPromise;

// 取得所有頁面
const allPages = context.pages();
Python
# 等待新分頁
with context.expect_page() as new_page_info:
    page.get_by_text("開啟新分頁").click()
new_page = new_page_info.value
new_page.wait_for_load_state()

# 等待彈出視窗
with page.expect_popup() as popup_info:
    page.get_by_text("開啟").click()
popup = popup_info.value
🧭 導航方法
Node.js
// 導航至 URL
await page.goto('https://example.com', {
  waitUntil: 'load',  // 預設
  timeout: 30_000,
});

// waitUntil 選項：
// 'load'              — load 事件
// 'domcontentloaded'  — DOM 載入
// 'networkidle'       — ⚠ 不推薦
// 'commit'            — 收到回應

// 等待 URL 變化
await page.waitForURL('**/dashboard');

// 前進/後退
await page.goBack();
await page.goForward();

// 重新載入
await page.reload();

// 取得資訊
page.url();    // 目前 URL
page.title();  // 頁面標題
⚠避免用 'networkidle'，它等待 500ms 無網路活動，不可靠且慢。改用 web-first 斷言等待特定 UI 狀態。
框架（Frames）& iframes
🖼 Frame 定位器
Node.js
// 方式 1：frameLocator（推薦）
const frame = page.frameLocator('#my-iframe');
await frame.getByRole('button', { name: '送出' })
  .click();

// 方式 2：locator → contentFrame
const frame = page.locator('#my-iframe')
  .contentFrame();
await frame.getByText('Hello').click();

// 巢狀 iframe
const inner = page.frameLocator('#outer')
  .frameLocator('#inner');
await inner.getByText('Deep').click();

// 反向：frameLocator → locator
const loc = page.frameLocator('#frame')
  .owner();  // 取得 iframe 元素本身
Python
# frameLocator（推薦）
frame = page.frame_locator("#my-iframe")
frame.get_by_role("button", name="送出").click()

# locator → content_frame
frame = page.locator("#my-iframe").content_frame
frame.get_by_text("Hello").click()

# 巢狀
inner = page.frame_locator("#outer") \
    .frame_locator("#inner")
💡frameLocator 只是定位器，本身不執行操作。必須接續 getByRole() 等方法才能操作 iframe 內元素。
🔒 瀏覽器上下文（隔離）
Node.js
// 建立隔離上下文（模擬多使用者）
const adminCtx = await browser.newContext({
  storageState: 'admin.json',
});
const userCtx = await browser.newContext({
  storageState: 'user.json',
});
const adminPage = await adminCtx.newPage();
const userPage = await userCtx.newPage();

// 上下文選項
const context = await browser.newContext({
  viewport: { width: 1920, height: 1080 },
  locale: 'zh-TW',
  timezoneId: 'Asia/Taipei',
  geolocation: { longitude: 121.5, latitude: 25 },
  permissions: ['geolocation'],
  colorScheme: 'dark',
  userAgent: 'Custom UA',
  httpCredentials: { username: 'u', password: 'p' },
  proxy: { server: 'http://proxy:8080' },
  recordVideo: { dir: './videos' },
  recordHar: { path: './trace.har' },
});
🔒每個上下文完全隔離（cookies、localStorage、sessionStorage）。建立開銷極低（毫秒級）。

🎨 執行時動態變更
Node.js
// 變更視窗大小
await page.setViewportSize({
  width: 1600, height: 1200
});

// 變更色彩方案
await page.emulateMedia({
  colorScheme: 'dark',
  media: 'print',
  reducedMotion: 'reduce',
  forcedColors: 'active',
});

// 變更地理位置
await context.setGeolocation({
  longitude: 121.5, latitude: 25.0
});

// 設定權限
await context.grantPermissions(
  ['geolocation'],
  { origin: 'https://example.com' }
);
await context.clearPermissions();
下載 · 對話框 · JavaScript 執行 · 事件
📥 下載 & 對話框
Node.js — 下載
const downloadPromise =
  page.waitForEvent('download');
await page.getByText('下載').click();
const download = await downloadPromise;

console.log(download.suggestedFilename());
await download.saveAs('./file.pdf');
// download.url(), download.failure()
Python — 下載
with page.expect_download() as dl_info:
    page.get_by_text("下載").click()
download = dl_info.value
download.save_as("./file.pdf")
Node.js — 對話框
// ⚠ 必須 accept/dismiss，否則頁面凍結
page.on('dialog', dialog => {
  console.log(dialog.type()); // alert|confirm|prompt
  console.log(dialog.message());
  dialog.accept('輸入值');  // 或 dialog.dismiss()
});
⚠未註冊 dialog 監聽器時，Playwright 自動 dismiss。如需處理 confirm/prompt，必須在觸發前註冊。
⚡ JavaScript 執行 & 瀏覽器事件
Node.js — evaluate
// 執行 JS 並返回結果
const title = await page.evaluate(() =>
  document.title);

// 帶參數
const text = await page.evaluate(
  ([sel]) => document.querySelector(sel).textContent,
  ['.header']
);

// 返回 JSHandle（保持引用）
const handle = await page.evaluateHandle(() =>
  document.body);
Python — evaluate
title = page.evaluate("document.title")
text = page.evaluate(
    "([sel]) => document.querySelector(sel).textContent",
    [".header"]
)
常用頁面事件：
事件說明
'load'頁面載入完成
'domcontentloaded'DOM 載入
'request'發送請求
'response'收到回應
'dialog'對話框出現
'download'下載開始
'popup'彈出視窗
'console'console.log 等
'pageerror'未捕獲的例外
'close'頁面關閉
