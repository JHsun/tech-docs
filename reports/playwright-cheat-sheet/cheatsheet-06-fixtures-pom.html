🔧 自訂 Fixture（Node.js）

fixtures.ts — 定義
import { test as base } from '@playwright/test';
import { TodoPage } from './todo-page';

export const test = base.extend<{
  todoPage: TodoPage;           // test 範圍
  account: { name: string };    // test 範圍
}>({
  todoPage: async ({ page }, use) => {
    // 設定（Setup）
    const todoPage = new TodoPage(page);
    await todoPage.goto();
    await todoPage.addDefaultItems();

    await use(todoPage);  // ← 提供給測試使用

    // 拆卸（Teardown）
    await todoPage.removeAll();
  },

  account: async ({}, use) => {
    const account = await createAccount();
    await use(account);
    await deleteAccount(account);
  },
});

export { expect } from '@playwright/test';

使用
import { test, expect } from './fixtures';

test('add todo', async ({ todoPage }) => {
  await todoPage.addItem('Buy milk');
  await expect(todoPage.items).toHaveCount(4);
});

⚡ Worker Fixture & Auto Fixture

Worker 範圍（跨測試共用）
export const test = base.extend<{}, {
  dbConnection: DatabaseConnection;
}>({
  dbConnection: [async ({}, use) => {
    const db = await connectToDatabase();
    await use(db);
    await db.close();
  }, { scope: 'worker' }],
});

自動 Fixture（每個測試自動執行）
export const test = base.extend<{
  saveLogs: void;
}>({
  saveLogs: [async ({}, use, testInfo) => {
    const logs: string[] = [];
    await use();
    if (testInfo.status !== testInfo.expectedStatus) {
      const logFile = testInfo.outputPath('logs.txt');
      await fs.writeFile(logFile, logs.join('\n'));
      testInfo.attach('logs', { path: logFile });
    }
  }, { auto: true }],
});

選項 Fixture（可在 config 覆寫）
export const test = base.extend<{
  defaultItem: string;
}>({
  defaultItem: ['Buy milk', { option: true }],
});
// config 中：use: { defaultItem: 'Buy bread' }

🏗️ 頁面物件模型（POM）

Node.js — 頁面物件
// pages/login-page.ts
import { type Locator, type Page, expect } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly usernameInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorMessage: Locator;

  constructor(page: Page) {
    this.page = page;
    this.usernameInput = page.getByLabel('帳號');
    this.passwordInput = page.getByLabel('密碼');
    this.submitButton = page.getByRole('button',
      { name: '登入' });
    this.errorMessage = page.getByRole('alert');
  }

  async goto() {
    await this.page.goto('/login');
  }

  async login(user: string, pass: string) {
    await this.usernameInput.fill(user);
    await this.passwordInput.fill(pass);
    await this.submitButton.click();
  }

  async expectError(message: string) {
    await expect(this.errorMessage)
      .toContainText(message);
  }
}

// 測試中使用
test('登入失敗', async ({ page }) => {
  const loginPage = new LoginPage(page);
  await loginPage.goto();
  await loginPage.login('wrong', 'wrong');
  await loginPage.expectError('帳號或密碼錯誤');
});
Python — 頁面物件
# pages/login_page.py
from playwright.sync_api import Page, expect

class LoginPage:
    def __init__(self, page: Page):
        self.page = page
        self.username_input = page.get_by_label("帳號")
        self.password_input = page.get_by_label("密碼")
        self.submit_button = page.get_by_role(
            "button", name="登入")
        self.error_message = page.get_by_role("alert")

    def goto(self):
        self.page.goto("/login")

    def login(self, user: str, password: str):
        self.username_input.fill(user)
        self.password_input.fill(password)
        self.submit_button.click()

    def expect_error(self, message: str):
        expect(self.error_message).to_contain_text(
            message)

# test_login.py
from pages.login_page import LoginPage

def test_login_failure(page):
    login_page = LoginPage(page)
    login_page.goto()
    login_page.login("wrong", "wrong")
    login_page.expect_error("帳號或密碼錯誤")

💡 POM 原則：① 在 constructor 中定義所有 Locator ② 方法封裝使用者操作 ③ 斷言可放在 POM 或測試中 ④ 避免在 POM 中暴露 page 物件

⭐ 最佳實踐（Best Practices）

類別✅ 推薦❌ 避免
定位器使用 getByRole()、getByText()、getByTestId()CSS class、XPath、DOM 結構依賴
斷言Web-first：await expect(loc).toBeVisible()expect(await loc.isVisible()).toBe(true)
等待信任自動等待和 web-first 斷言手動 waitForTimeout()、page.pause()
隔離每個測試完全獨立測試間共用狀態或依賴執行順序
第三方使用 Network API mock 外部服務直接呼叫真實第三方 API
測試數據每個測試建立所需數據依賴預設或共用數據
瀏覽器跨所有瀏覽器測試僅測試 Chromium
CI每次 commit/PR 執行僅本機執行或手動觸發
重試CI 上設定重試（2-3 次）無重試導致 flaky 失敗
版本保持 Playwright 最新版鎖定舊版不更新
Lint@typescript-eslint/no-floating-promises忘記 await 導致誤判

🔧 全域設定/拆卸

推薦方式：Project Dependencies
// tests/global.setup.ts
import { test as setup } from '@playwright/test';
setup('seed database', async ({}) => {
  await seedDatabase();
});

// playwright.config.ts
projects: [
  { name: 'setup', testMatch: /.*\.setup\.ts/,
    teardown: 'cleanup' },
  { name: 'cleanup', testMatch: /.*\.teardown\.ts/ },
  { name: 'tests', dependencies: ['setup'] },
]

替代方式：globalSetup（不推薦）
// playwright.config.ts
globalSetup: require.resolve('./global-setup'),
// 缺點：不支援 fixtures、不顯示在報告中
Python
@pytest.fixture(scope="session", autouse=True)
def seed_database():
    # setup
    db.seed()
    yield
    # teardown
    db.cleanup()

  fullPage: true,        // 完整頁面
  clip: { x:0, y:0, width:800, height:600 },

  animations: 'disabled',  // 停用動畫
  scale: 'css',          // 'css' | 'device'
