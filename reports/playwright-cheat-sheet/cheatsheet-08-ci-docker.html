🐙 GitHub Actions（Node.js）

.github/workflows/playwright.yml
name: Playwright Tests
on: [push, pull_request]
jobs:
  test:
    timeout-minutes: 60
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: lts/*
      - name: Install dependencies
        run: npm ci
      - name: Install Playwright Browsers
        run: npx playwright install --with-deps
      - name: Run tests
        run: npx playwright test
      - uses: actions/upload-artifact@v4
        if: ${{ !cancelled() }}
        with:
          name: playwright-report
          path: playwright-report/
          retention-days: 30

🐙 GitHub Actions（Python）

.github/workflows/playwright.yml
name: Playwright Tests
on: [push, pull_request]
jobs:
  test:
    timeout-minutes: 60
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: '3.12'
      - name: Install dependencies
        run: pip install -r requirements.txt
      - name: Install browsers
        run: python -m playwright install --with-deps
      - name: Run tests
        run: pytest --tracing=retain-on-failure
      - uses: actions/upload-artifact@v4
        if: ${{ !cancelled() }}
        with:
          name: test-results
          path: test-results/

🔀 分片 CI（GitHub Actions）

矩陣分片策略
jobs:
  test:
    strategy:
      fail-fast: false
      matrix:
        shardIndex: [1, 2, 3, 4]
        shardTotal: [4]
    steps:
      # ... setup steps ...
      - name: Run tests
        run: npx playwright test --shard=${{ matrix.shardIndex }}/${{ matrix.shardTotal }}
      - name: Upload blob report
        uses: actions/upload-artifact@v4
        with:
          name: blob-report-${{ matrix.shardIndex }}
          path: blob-report/

  merge-reports:
    needs: [test]
    runs-on: ubuntu-latest
    steps:
      - uses: actions/download-artifact@v4
        with:
          path: all-blob-reports
          pattern: blob-report-*
          merge-multiple: true
      - run: npx playwright merge-reports --reporter html ./all-blob-reports
      - uses: actions/upload-artifact@v4
        with:
          name: html-report
          path: playwright-report/

🐳 Docker

Node.js Docker
# 官方映像
mcr.microsoft.com/playwright:v1.58.2-noble

# 執行（E2E 測試用）
docker run -it --rm --ipc=host --init \
  mcr.microsoft.com/playwright:v1.58.2-noble \
  /bin/bash

# 推薦旗標
--ipc=host    # 防止 Chromium OOM
--init        # 避免殭屍程序
--cap-add=SYS_ADMIN  # 解決奇怪錯誤

# 自建映像
FROM node:20-bookworm
RUN npx -y playwright@1.58.2 install --with-deps
COPY . /app
WORKDIR /app
RUN npm ci
CMD ["npx", "playwright", "test"]
Python Docker
# 官方映像
mcr.microsoft.com/playwright/python:v1.58.0-noble

# 執行
docker run -it --rm --ipc=host --init \
  mcr.microsoft.com/playwright/python:v1.58.0-noble \
  /bin/bash

# 自建映像
FROM python:3.12-bookworm
RUN pip install pytest-playwright
RUN playwright install --with-deps
COPY . /app
WORKDIR /app
CMD ["pytest"]

遠端連線

啟動伺服器
docker run -p 3000:3000 --rm --init -it \
  --workdir /home/pwuser --user pwuser \
  mcr.microsoft.com/playwright:v1.58.2-noble \
  /bin/sh -c "npx -y playwright@1.58.2 run-server --port 3000 --host 0.0.0.0"

連接方式
# 方式 1：環境變數（@playwright/test）
PW_TEST_CONNECT_WS_ENDPOINT=ws://127.0.0.1:3000/ npx playwright test

# 方式 2：API 連接
const browser = await playwright.chromium.connect('ws://127.0.0.1:3000/');
⚠️ 安全：Docker 映像僅供測試/開發用途，不建議用來瀏覽不信任的網站。不信任內容請用獨立使用者 + seccomp profile。

🤖 Agents（實驗性）

Node.js 專屬
# 初始化代理系統
npx playwright init-agents --loop=vscode   # VS Code
npx playwright init-agents --loop=claude    # Claude
npx playwright init-agents --loop=opencode  # OpenCode

# 三種代理角色：
# 🎭 planner  — 探索應用、產生 Markdown 測試計畫
# 🎭 generator — 將計畫轉換為測試檔案
# 🎭 healer   — 執行測試並自動修復失敗
💡 Agents 是 LLM 驅動的測試自動化工具，可自動探索應用程式並產生/維護測試。僅 Node.js 版本支援。
