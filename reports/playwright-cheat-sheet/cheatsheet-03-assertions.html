📄 Page 斷言

Node.js
await expect(page).toHaveTitle(/Playwright/);
await expect(page).toHaveTitle('Playwright', { timeout: 10000 });
await expect(page).toHaveURL('https://example.com/dashboard');
await expect(page).toHaveURL(/.*dashboard/);

Python
expect(page).to_have_title(re.compile("Playwright"))
expect(page).to_have_url("https://example.com/dashboard")
expect(page).to_have_url(re.compile(".*dashboard"))

🚫 否定斷言

Node.js — .not 前綴
await expect(locator).not.toBeVisible();

await expect(page).not.toHaveURL(/login/);

🔄 軟斷言（Soft Assertions）

Node.js
// 軟斷言：失敗不中斷測試，最後統一報告
await expect.soft(locator).toHaveText('Success');
await expect.soft(locator).toBeVisible();
// 測試繼續執行...

// 條件中斷：收集到錯誤時停止
expect(test.info().errors).toHaveLength(0);

Python
expect(locator).to_have_text("Success")
# Python 無原生軟斷言，使用 pytest-check 外掛
💡 軟斷言適合同一頁面上需要驗證多項屬性的場景，避免第一個失敗就中斷後續檢查。

⏱️ 自訂逾時

Node.js
// 全域設定
export default defineConfig({
  expect: { timeout: 10_000 },   // 預設 5000ms
});

// 單一斷言
await expect(locator).toBeVisible({ timeout: 15000 });

// 測試層級
test('slow test', async ({ page }) => {
  test.slow();  // 逾時 ×3
});

Python
# 全域設定
expect.set_options(timeout=10_000)

# 單一斷言
expect(locator).to_be_visible(timeout=15000)

# 頁面預設
page.set_default_timeout(10000)

🔁 輪詢斷言（Polling）

Node.js
// 持續輪詢直到條件滿足
await expect.poll(async () => {
  const response = await page.request.get('/api/status');
  return response.status();
}, {
  message: '等待 API 就緒',
  intervals: [1_000, 2_000, 5_000],  // 自訂間隔
  timeout: 60_000,
}).toBe(200);

Python
# 使用自訂迴圈 + expect
import time
for _ in range(30):
    response = page.request.get("/api/status")
    if response.status == 200:
        break
    time.sleep(1)

🔁 重試區塊（toPass）

Node.js — 重試多步驟操作
await expect(async () => {
  const response = await page.request.get('/api/data');
  expect(response.status()).toBe(200);
  const data = await response.json();
  expect(data).toHaveProperty('items');
  expect(data.items.length).toBeGreaterThan(0);
}).toPass({
  timeout: 60_000,
  intervals: [1_000, 2_000, 5_000],
});
💡 toPass() 將整個區塊作為原子操作重試，任何步驟失敗都會重試整個區塊。

📝 斷言使用範例

Node.js 完整範例
test('購物車功能', async ({ page }) => {
  await page.goto('/shop');

  // 文字斷言（字串 = 完整匹配）
  await expect(page.getByRole('heading'))
    .toHaveText('線上商店');

  // 文字斷言（正則 = 部分匹配）
  await expect(page.locator('.price'))
    .toHaveText(/\$\d+\.\d{2}/);

  // 包含文字
  await expect(page.locator('.desc'))
    .toContainText('免運費');

  // 屬性斷言
  await expect(page.locator('img.hero'))
    .toHaveAttribute('src', /hero/);

  // CSS 斷言
  await expect(page.locator('.btn'))
    .toHaveCSS('background-color', 'rgb(45, 125, 79)');

  // 計數斷言
  await expect(page.getByRole('listitem'))
    .toHaveCount(5);

  // 陣列文字斷言（驗證清單順序）
  await expect(page.getByRole('listitem'))
    .toHaveText(['蘋果', '香蕉', '橘子']);

  // 截圖比對（僅 Node.js）
  await expect(page).toHaveScreenshot('cart.png');
});
Python 完整範例
def test_shopping_cart(page: Page):
    page.goto("/shop")

    # 文字斷言（字串 = 完整匹配）
    expect(page.get_by_role("heading")).to_have_text(
        "線上商店")

    # 文字斷言（正則 = 部分匹配）
    expect(page.locator(".price")).to_have_text(
        re.compile(r"\$\d+\.\d{2}"))

    # 包含文字
    expect(page.locator(".desc")).to_contain_text(
        "免運費")

    # 屬性斷言
    expect(page.locator("img.hero")).to_have_attribute(
        "src", re.compile("hero"))

    # CSS 斷言
    expect(page.locator(".btn")).to_have_css(
        "background-color", "rgb(45, 125, 79)")

    # 計數斷言
    expect(page.get_by_role("listitem")).to_have_count(5)

    # 陣列文字斷言
    expect(page.get_by_role("listitem")).to_have_text(
        ["蘋果", "香蕉", "橘子"])

    # Python 無內建截圖比對

❌ 常見錯誤 vs ✅ 正確寫法

❌ 錯誤（非 Web-first）✅ 正確（Web-first，自動重試）
expect(await locator.isVisible()).toBe(true)await expect(locator).toBeVisible()
expect(await locator.textContent()).toBe('hi')await expect(locator).toHaveText('hi')
expect(await locator.count()).toBe(3)await expect(locator).toHaveCount(3)
expect(await page.title()).toContain('X')await expect(page).toHaveTitle(/X/)
expect(page.url()).toContain('/dash')await expect(page).toHaveURL(/dash/)
⚠️ 關鍵差異：左側在呼叫時僅檢查一次，可能因頁面尚未更新而失敗。右側（Web-first）會自動重試直到條件滿足或逾時，是 Playwright 推薦的斷言方式。
