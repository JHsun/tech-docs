除錯（Debugging）
🔍 Playwright Inspector
Node.js
# 啟動 Inspector（headed + 1 worker + timeout=0）
npx playwright test --debug

# 從特定行號開始除錯
npx playwright test example.spec.ts:21 --debug
Python
PWDEBUG=1 pytest -s
Inspector 功能：
▶️ Resume繼續執行到下一個 pause/breakpoint
⏭ Step Over執行下一步操作
🎯 Pick Locator在頁面上選取元素，產生定位器
📝 Locator 欄即時編輯並測試定位器匹配
📋 操作日誌顯示每步操作的執行紀錄
⏸ 斷點 & 日誌
測試中插入斷點
test('my test', async ({ page }) => {
  await page.goto('https://example.com');

  // ⏸ 暫停並開啟 Inspector
  await page.pause();

  await page.getByRole('button').click();
});
Python
page.pause()  # 暫停

📋 Trace CLI 分析 — npx playwright trace
Node.js
# 開啟 trace.zip
npx playwright trace open test-results/.../trace.zip

# 列出動作（可 grep 篩選）
npx playwright trace actions --grep="expect"

# 檢視特定動作詳情
npx playwright trace action 9

# 取得快照（動作前/後）
npx playwright trace snapshot 9 --name after

# 關閉當前 session
npx playwright trace close
💡取代傳統 GUI Trace Viewer，適合 CI 日誌與 Agent 自動化分析失敗測試。

截圖（Screenshots）
📸 截圖 API
Node.js
// 整頁截圖
await page.screenshot({ path: 'page.png' });

// 全頁截圖（含捲軸外內容）
await page.screenshot({
  path: 'full.png',
  fullPage: true,
});

// 元素截圖
await page.locator('.hero').screenshot({
  path: 'hero.png',
});

// 完整選項
await page.screenshot({
  path: 'shot.png',
  type: 'jpeg',          // 'png'|'jpeg'
  quality: 80,           // jpeg 品質 0-100
  fullPage: false,
  clip: { x: 0, y: 0, width: 300, height: 200 },
  omitBackground: true,  // 透明背景
  mask: [page.locator('.ad')],  // 遮罩元素
  maskColor: '#FF00FF',
  scale: 'css',          // 'css'|'device'
  animations: 'disabled',
  caret: 'hide',
  timeout: 30_000,
});

時鐘（Clock）
⏰ 時間操控
Node.js
// 固定時間（凍結 Date.now）
await page.clock.setFixedTime(
  new Date('2024-02-02T10:00:00')
);

// 安裝假時鐘 + 手動推進
await page.clock.install({
  time: new Date('2024-02-02T08:00:00')
});
await page.clock.fastForward('30:00'); // 快轉 30 分
await page.clock.runFor(2000);         // 推進 2 秒

// 暫停 / 恢復
await page.clock.pauseAt(
  new Date('2024-02-02T10:00:00'));
await page.clock.resume();

// 設定系統時間（不影響計時器）
await page.clock.setSystemTime(
  new Date('2024-12-25'));
Python
import datetime

頁面物件模型（POM）
Node.js — Page Object
// models/LoginPage.ts
import { type Page, type Locator } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly usernameInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;

  constructor(page: Page) {
    this.page = page;
    this.usernameInput = page.getByLabel('帳號');
    this.passwordInput = page.getByLabel('密碼');
    this.submitButton = page.getByRole('button',
      { name: '登入' });
  }

  async goto() {
    await this.page.goto('/login');
  }

  async login(user: string, pass: string) {
    await this.usernameInput.fill(user);
    await this.passwordInput.fill(pass);
    await this.submitButton.click();
    await this.page.waitForURL('/dashboard');
  }
}

// 測試中使用
test('登入', async ({ page }) => {
  const login = new LoginPage(page);
  await login.goto();
  await login.login('admin', 'password');
});
Python — Page Object
# models/login_page.py
class LoginPage:
    def __init__(self, page):
        self.page = page
        self.username = page.get_by_label("帳號")
        self.password = page.get_by_label("密碼")
        self.submit = page.get_by_role(
            "button", name="登入")

    def goto(self):
        self.page.goto("/login")

    def login(self, user, password):
        self.username.fill(user)
        self.password.fill(password)
        self.submit.click()
        self.page.wait_for_url("/dashboard")

# 測試中使用
def test_login(page):
    login = LoginPage(page)
    login.goto()
    login.login("admin", "password")
💡POM 原則：在 constructor 中定義 Locator，在方法中封裝操作流程。避免在 POM 中寫斷言（讓測試決定驗什麼）。
