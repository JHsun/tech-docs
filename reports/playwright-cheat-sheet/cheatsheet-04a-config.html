📝 playwright.config.ts 完整範本

TypeScript
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests',               // 測試目錄
  fullyParallel: true,              // 所有測試平行
  forbidOnly: !!process.env.CI,     // CI 禁止 test.only
  retries: process.env.CI ? 2 : 0,  // CI 重試 2 次
  workers: process.env.CI ? 1 : undefined,  // CI 用 1 worker
  reporter: 'html',                 // 報告器
  timeout: 30_000,                  // 測試逾時 30 秒
  globalTimeout: 0,                 // 全域逾時（0=無限）
  expect: { timeout: 5_000 },       // 斷言逾時 5 秒

  use: {
    baseURL: 'http://localhost:3000',
    trace: 'on-first-retry',
    screenshot: 'only-on-failure',
    video: 'retain-on-failure',
  },

  projects: [
    { name: 'chromium', use: { ...devices['Desktop Chrome'] } },
    { name: 'firefox',  use: { ...devices['Desktop Firefox'] } },
    { name: 'webkit',   use: { ...devices['Desktop Safari'] } },
    { name: 'mobile',   use: { ...devices['iPhone 13'] } },
  ],

  webServer: {
    command: 'npm run start',
    url: 'http://localhost:3000',
    reuseExistingServer: !process.env.CI,
    timeout: 120_000,
  },
});

🔄 重試與分片

重試（Retries）

Config  retries: process.env.CI ? 2 : 0
CLI    npx playwright test --retries=3

結果分類：passed（首次通過）· flaky（重試後通過）· failed（全失敗）。每次重試用全新 Worker。testInfo.retry 取得當前重試次數。

分片（Sharding）

執行    npx playwright test --shard=1/4

合併    # 各分片用 blob reporter
         reporter: [['blob', { outputDir: 'blob-report' }]]
         # 合併所有分片報告
         npx playwright merge-reports --reporter html ./all-blob-reports

📐 專案與依賴（Projects）

TypeScript — 設定專案依賴
projects: [
  { name: 'setup',
    testMatch: /.*\.setup\.ts/,
    teardown: 'cleanup' },

  { name: 'cleanup',
    testMatch: /.*\.teardown\.ts/ },

  { name: 'chromium',
    dependencies: ['setup'],
    use: {
      ...devices['Desktop Chrome'],
      storageState: 'playwright/.auth/user.json',
    }
  },
  { name: 'firefox',
    dependencies: ['setup'],
    use: { ...devices['Desktop Firefox'] }
  },
]
💡 dependencies 確保 setup 在測試前完成。teardown 在所有依賴專案完成後執行。

🏷️ 標註（Annotations）

Node.js
test.skip('跳過此測試', async () => {});
test.skip(browserName === 'firefox', '尚未支援');

test.fixme('待修復', async () => {});     // 不執行
test.fail('預期失敗', async () => {});    // 執行並確認失敗
test.slow('效能較慢', async () => {});    // 逾時 ×3
test.only('僅執行此測試', async () => {});

// 標籤
test('login @fast @smoke', { tag: ['@fast', '@smoke'] },
  async ({ page }) => {});
// 篩選：npx playwright test --grep @smoke
// 排除：npx playwright test --grep-invert @slow
Python — pytest markers
@pytest.mark.skip(reason="尚未實作")
@pytest.mark.skip_browser("firefox")
@pytest.mark.only_browser("chromium")

# 標籤篩選
pytest -m "smoke"
pytest -m "not slow"

🆕 v1.59 Config 新選項

新 trace 模式 — 成功與失敗的 trace 都保留
use: {
  trace: 'retain-on-failure-and-retries',
  // 每次嘗試都錄製，失敗時全部保留
  // 可比對成功 trace vs. 失敗 trace 找 flaky 原因
}
Video 顯示動作標註（原 screencast 能力）
use: {
  video: {
    mode: 'on',
    show: {
      actions: { position: 'top-left' },
      test:    { position: 'top-right' },
    },
  },
}
browserType.launch() 新增 artifactsDir
const browser = await chromium.launch({
  artifactsDir: './my-artifacts',  // 自訂產出目錄
});
tracing.start({ live: true }) 即時更新
await context.tracing.start({
  screenshots: true,
  snapshots: true,
  live: true,      // v1.59：即時同步到檔案
});
💡 retain-on-failure-and-retries 最適合除錯 flaky 測試 — 可同時拿到通過和失敗的 trace 對照。

🧩 平行處理設定

Node.js
// 全域：所有測試平行
fullyParallel: true,

// 檔案內設定
test.describe.configure({ mode: 'parallel' });  // 檔案內平行
test.describe.configure({ mode: 'serial' });    // 強制循序
test.describe.configure({ mode: 'default' });   // 預設

// Worker 資訊
test('example', async ({ page }, testInfo) => {
  console.log(testInfo.workerIndex);    // 唯一遞增
  console.log(testInfo.parallelIndex);  // 0 至 workers-1
});
