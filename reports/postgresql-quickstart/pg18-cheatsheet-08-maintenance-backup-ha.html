🧹 VACUUM 維護 (Ch24.1)

VACUUM 種類

-- 標準 VACUUM（回收空間，不鎖表）
VACUUM table_name;
VACUUM (VERBOSE) table_name;

-- VACUUM FULL（重寫表，取得 AccessExclusive 鎖）
VACUUM FULL table_name;

-- VACUUM + ANALYZE
VACUUM ANALYZE table_name;

-- 凍結（防止 transaction ID wraparound）
VACUUM FREEZE table_name;

-- 平行 VACUUM (PG13+)
VACUUM (PARALLEL 4) table_name;

-- 跳過鎖定的表 (PG12+)
VACUUM (SKIP_LOCKED) table_name;
比較VACUUMVACUUM FULL
鎖定不阻塞讀寫AccessExclusive（全鎖）
空間標記可重用（不歸還 OS）重寫表（歸還 OS）
速度快慢（需雙倍空間）
使用場景例行維護嚴重膨脹時

Autovacuum 參數

autovacuumon啟用自動清理
autovacuum_max_workers3最大 worker 數
autovacuum_naptime1min檢查間隔
autovacuum_vacuum_threshold50最小死元組數
autovacuum_vacuum_scale_factor0.2死元組比例門檻
autovacuum_analyze_threshold50ANALYZE 最小變動數
autovacuum_analyze_scale_factor0.1ANALYZE 變動比例
autovacuum_vacuum_cost_delay2msI/O 節流延遲
autovacuum_vacuum_cost_limit-1I/O 成本上限
autovacuum_freeze_max_age2×10⁸強制凍結年齡

大表調整（表級設定）

ALTER TABLE big_table SET (
  autovacuum_vacuum_scale_factor = 0.01,
  autovacuum_analyze_scale_factor = 0.005,
  autovacuum_vacuum_cost_delay = 0
);

關鍵系統視圖

實用查詢

-- 目前活躍查詢
SELECT pid, age(clock_timestamp(), query_start),
  state, query
FROM pg_stat_activity
WHERE state != 'idle'
ORDER BY query_start;

-- 未使用的索引
SELECT relname, indexrelname, idx_scan
FROM pg_stat_user_indexes
WHERE idx_scan = 0;
💾 備份與還原 (Ch25)

pg_dump 邏輯備份

# 純 SQL 文字
pg_dump -d mydb -f backup.sql

# 自訂格式（推薦，可壓縮、可選擇性還原）
pg_dump -d mydb -Fc -f backup.dump

# 目錄格式（可平行）
pg_dump -d mydb -Fd -j 4 -f backup_dir/

# 只備份綱要
pg_dump -d mydb -s -f schema.sql

# 只備份資料
pg_dump -d mydb -a -f data.sql

# 備份特定表
pg_dump -d mydb -t public.users -f users.sql

# 排除表
pg_dump -d mydb -T temp_* -f backup.sql

# 全叢集備份
pg_dumpall -f all.sql
pg_dumpall --roles-only -f roles.sql

pg_restore 還原

# 從自訂/目錄格式還原
pg_restore -d mydb backup.dump

# 平行還原
pg_restore -d mydb -j 4 backup_dir/

# 只還原特定表
pg_restore -d mydb -t users backup.dump

# 清除再還原
pg_restore -d mydb --clean --if-exists backup.dump

# 列出備份內容
pg_restore -l backup.dump

# 純 SQL 檔直接用 psql
psql -d mydb -f backup.sql

COPY 高速匯入匯出

-- 伺服器端
COPY t TO '/tmp/t.csv' CSV HEADER;
COPY t FROM '/tmp/t.csv' CSV HEADER;

-- 客戶端（psql）
\copy t TO 't.csv' CSV HEADER
\copy t FROM 't.csv' CSV HEADER
🕐 PITR 時間點還原 (Ch25.3)

基礎備份

# pg_basebackup
pg_basebackup -D /backup/base \
  -Ft -z \              # tar 壓縮格式
  -Xs                   # 串流 WAL
  -P                    # 顯示進度
  --checkpoint=fast

WAL 歸檔設定

# postgresql.conf
wal_level = replica
archive_mode = on
archive_command = 'cp %p /archive/%f'
# 或使用 archive_library (PG15+)

還原 (recovery.signal)

# 1. 還原基礎備份到資料目錄
# 2. 建立 recovery.signal 檔案
touch $PGDATA/recovery.signal

# 3. 設定 postgresql.conf
restore_command = 'cp /archive/%f %p'
recovery_target_time = '2024-03-15 14:30:00'
# 或 recovery_target_lsn = '0/1A2B3C4D'
# 或 recovery_target_xid = '12345'
# 或 recovery_target = 'immediate'
recovery_target_action = 'promote'

# 4. 啟動 PostgreSQL
🔄 串流複寫 (Ch26)

主庫設定

# postgresql.conf
wal_level = replica
max_wal_senders = 10
max_replication_slots = 10
hot_standby = on

# 同步複寫（選用）
synchronous_standby_names = 's1'
# 或 'FIRST 1 (s1, s2)'
# 或 'ANY 1 (s1, s2)'

# pg_hba.conf
host replication replicator 10.0.0.0/8 scram-sha-256

建立備庫

# 從主庫建立基礎備份
pg_basebackup -h primary_host \
  -U replicator -D /data/standby \
  -Xs -P -R
# -R 自動建立 standby.signal 和設定連線

# 或手動設定 postgresql.conf
primary_conninfo = 'host=primary port=5432 user=replicator password=xxx application_name=s1'

複寫槽 (Replication Slots)

-- 物理複寫槽（保留 WAL 直到備庫消費）
SELECT pg_create_physical_replication_slot('slot1');

-- 邏輯複寫槽
SELECT pg_create_logical_replication_slot('slot1', 'pgoutput');

-- 查看複寫槽
SELECT * FROM pg_replication_slots;

-- 刪除
SELECT pg_drop_replication_slot('slot1');

監控複寫延遲

-- 主庫查看
SELECT client_addr, state,
  sent_lsn, write_lsn, flush_lsn,
  replay_lsn,
  pg_wal_lsn_diff(sent_lsn, replay_lsn) lag_bytes
FROM pg_stat_replication;

-- 備庫查看
SELECT
  pg_is_in_recovery(),
  pg_last_wal_receive_lsn(),
  pg_last_wal_replay_lsn(),
  pg_last_xact_replay_timestamp();

手動故障切換

# 備庫升為主庫
pg_ctl promote -D $PGDATA
-- 或
SELECT pg_promote();
📡 邏輯複寫 (Ch29)

發佈端（主庫）

# wal_level = logical

CREATE PUBLICATION my_pub
  FOR TABLE users, orders;
-- 或 FOR ALL TABLES
-- 或 FOR TABLES IN SCHEMA public

-- 只複寫特定操作
CREATE PUBLICATION insert_only
  FOR TABLE logs
  WITH (publish = 'insert');

-- 列篩選 (PG15+)
CREATE PUBLICATION filtered
  FOR TABLE orders WHERE (region = 'TW');

-- 欄位篩選 (PG15+)
CREATE PUBLICATION partial
  FOR TABLE users (id, name, email);

訂閱端

CREATE SUBSCRIPTION my_sub
  CONNECTION 'host=pub_host dbname=mydb user=replicator password=xxx'
  PUBLICATION my_pub
  WITH (
    copy_data = true,  -- 初始同步
    create_slot = true
  );

ALTER SUBSCRIPTION my_sub REFRESH PUBLICATION;
ALTER SUBSCRIPTION my_sub DISABLE;
DROP SUBSCRIPTION my_sub;
邏輯複寫 vs 物理複寫：邏輯可選表/列/列篩選、跨版本、多對一合併。物理是完整叢集複製
📜 WAL 與 Checkpoint (Ch28)

WAL (Write-Ahead Logging)

原理所有修改先寫入 WAL，再寫入資料檔案
目的crash recovery、複寫、PITR
位置$PGDATA/pg_wal/
段大小預設 16MB（initdb --wal-segsize 可調）

WAL 等級

minimal只做 crash recovery
replica支援物理複寫和歸檔（預設）
logical支援邏輯複寫

Checkpoint 參數

checkpoint_timeout5min最大 checkpoint 間隔
max_wal_size1GB觸發 checkpoint 的 WAL 量
checkpoint_completion_target0.9分散寫入的目標比例

synchronous_commit 選項

on本地 WAL flush（預設）
off不等 WAL flush（快但可能丟少量資料）
local本地 flush，不等遠端
remote_write等遠端寫入 OS 緩衝
remote_apply等遠端回放完成
📈 pg_stat_statements & JIT (Ch30-31)

pg_stat_statements（必裝擴充）

# postgresql.conf
shared_preload_libraries = 'pg_stat_statements'
pg_stat_statements.track = all

-- 啟用
CREATE EXTENSION pg_stat_statements;

-- 查詢最慢的 SQL
SELECT query,
  calls, mean_exec_time, total_exec_time,
  rows, shared_blks_hit, shared_blks_read
FROM pg_stat_statements
ORDER BY mean_exec_time DESC
LIMIT 20;

-- 重置統計
SELECT pg_stat_statements_reset();

JIT 編譯 (Ch30)

jiton啟用 JIT
jit_above_cost100000啟用 JIT 的成本門檻
jit_inline_above_cost500000內聯函式的門檻
jit_optimize_above_cost500000最佳化的門檻
JIT 對大量資料處理有幫助，但對簡單 OLTP 查詢可能增加延遲
