🧩 PL/pgSQL 函式 (Ch41)

函式結構

CREATE [OR REPLACE] FUNCTION func_name(
  p1 integer,
  p2 text DEFAULT 'hello'
)
RETURNS return_type
LANGUAGE plpgsql
[ IMMUTABLE | STABLE | VOLATILE ]
[ PARALLEL { SAFE | RESTRICTED | UNSAFE } ]
[ SECURITY DEFINER | SECURITY INVOKER ]
[ STRICT ]   -- NULL 參數直接回 NULL
[ COST 100 ] [ ROWS 1000 ]
AS $$
DECLARE
  var1 integer := 0;
  var2 text;
  rec  record;
BEGIN
  -- 函式主體
  RETURN result;
END;
$$;

回傳類型

RETURNS type單一值
RETURNS SETOF type多列
RETURNS TABLE (col type, ...)表格
RETURNS void無回傳
RETURNS trigger觸發器函式
OUT param type輸出參數
INOUT param type輸入輸出參數

📋 PROCEDURE 程序 (PG11+)
CREATE PROCEDURE transfer(
  from_acct int,
  to_acct int,
  amount numeric
)
LANGUAGE plpgsql
AS $$
BEGIN
  UPDATE accounts SET balance = balance - amount
    WHERE id = from_acct;
  UPDATE accounts SET balance = balance + amount
    WHERE id = to_acct;

  -- 程序可以中途提交/回滾！
  COMMIT;
END;
$$;

-- 呼叫
CALL transfer(1, 2, 100.00);
FUNCTION vs PROCEDURE：PROCEDURE 可以執行交易控制（COMMIT/ROLLBACK），FUNCTION 不行。PROCEDURE 沒有回傳值
⚡ 觸發器 (Ch37)

建立觸發器

CREATE TRIGGER trg_name
{ BEFORE | AFTER | INSTEAD OF }
{ INSERT | UPDATE [ OF col ] | DELETE | TRUNCATE }
[ OR ... ]
ON table_name
[ REFERENCING
    OLD TABLE AS old_tbl
    NEW TABLE AS new_tbl ]
[ FOR EACH { ROW | STATEMENT } ]
[ WHEN (condition) ]
EXECUTE FUNCTION func_name();

觸發器函式

CREATE FUNCTION trg_func()
RETURNS trigger AS $$
BEGIN
  -- 特殊變數
  -- NEW: 新列值 (INSERT/UPDATE)
  -- OLD: 舊列值 (UPDATE/DELETE)
  -- TG_OP: INSERT/UPDATE/DELETE
  -- TG_TABLE_NAME: 表名
  -- TG_WHEN: BEFORE/AFTER

  NEW.updated_at := now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

常見觸發器模式

-- 自動更新時間戳
CREATE TRIGGER set_updated
  BEFORE UPDATE ON t
  FOR EACH ROW
  EXECUTE FUNCTION set_updated_at();

-- 審計日誌
CREATE TRIGGER audit_trg
  AFTER INSERT OR UPDATE OR DELETE ON t
  FOR EACH ROW
  EXECUTE FUNCTION audit_log();

-- 條件觸發
CREATE TRIGGER check_salary
  BEFORE UPDATE OF salary ON employees
  FOR EACH ROW
  WHEN (NEW.salary > OLD.salary * 1.5)
  EXECUTE FUNCTION salary_check();

-- 管理觸發器
ALTER TABLE t DISABLE TRIGGER trg;
ALTER TABLE t ENABLE TRIGGER trg;
ALTER TABLE t DISABLE TRIGGER ALL;
DROP TRIGGER trg ON t;

🌐 Foreign Data Wrapper (Ch36, Ch57)
-- 安裝 postgres_fdw
CREATE EXTENSION postgres_fdw;

-- 建立遠端伺服器
CREATE SERVER remote_db
  FOREIGN DATA WRAPPER postgres_fdw
  OPTIONS (host '10.0.0.2',
    port '5432', dbname 'prod');

-- 使用者映射
CREATE USER MAPPING FOR local_user
  SERVER remote_db
  OPTIONS (user 'remote_user',
    password 'xxx');

-- 匯入外部表
IMPORT FOREIGN SCHEMA public
  FROM SERVER remote_db
  INTO local_schema;

-- 或手動建立
CREATE FOREIGN TABLE remote_users (
  id int, name text
) SERVER remote_db
  OPTIONS (table_name 'users');

內部結構要點 (Ch51-70)

查詢處理流程解析 → 重寫 → 規劃 → 執行
規劃器基於成本的最佳化，使用 pg_stats 統計
執行器火山模型（需求驅動、逐元組拉取）
B-tree 結構Lehman-Yao 高並發 B-tree
MVCC多版本並行控制（不刪除舊版本，VACUUM 回收）
WAL預寫式日誌，確保 crash recovery
