🏗️ CREATE TABLE 完整語法 (Ch5.1–5.3)
CREATE [ TEMPORARY | UNLOGGED ] TABLE [ IF NOT EXISTS ] table_name (
    column_name  data_type  [ COLLATE collation ]
      [ DEFAULT expr ]
      [ column_constraint [...] ]
      [ GENERATED ALWAYS AS (expr) STORED ]              -- 產生欄位
      [ GENERATED { ALWAYS | BY DEFAULT } AS IDENTITY ] -- 識別欄位
    [, ... ]
    [, table_constraint [...] ]
)
[ INHERITS (parent [, ...]) ]
[ PARTITION BY { RANGE | LIST | HASH } (col | expr [, ...]) ]
[ USING method ]
[ TABLESPACE name ]
[ WITH ( storage_param = val [, ...] ) ];

欄位約束 (Column Constraints)

NOT NULL不可為空
NULL可為空（預設）
UNIQUE唯一約束（允許多個 NULL）
UNIQUE NULLS NOT DISTINCT唯一（NULL 視為相同）PG15+
PRIMARY KEY主鍵 = UNIQUE + NOT NULL
CHECK (expr)自訂檢查條件
REFERENCES tbl(col)外鍵
DEFAULT expr預設值

表級約束 (Table Constraints)

UNIQUE (c1, c2)複合唯一
PRIMARY KEY (c1, c2)複合主鍵
CHECK (expr)可引用多欄位
FOREIGN KEY (c) REFERENCES t(c)外鍵
EXCLUDE USING ...排除約束
🔗 外鍵參照動作 (Ch5.4)
FOREIGN KEY (col) REFERENCES parent(col)
  ON DELETE action
  ON UPDATE action
  [ MATCH { FULL | PARTIAL | SIMPLE } ]
  [ DEFERRABLE [ INITIALLY DEFERRED ] ]
動作說明
NO ACTION預設；檢查延遲時在交易結束時才檢查
RESTRICT立即阻止
CASCADE連鎖刪除/更新子列
SET NULL將子欄位設為 NULL
SET DEFAULT將子欄位設為預設值
可加 (col, col) 指定只更新部分欄位（PG15+）
📊 資料表分割 (Partitioning) (Ch5.11)

分割策略

策略用途範例
RANGE日期、數值範圍按月/年分割
LIST離散值按地區/狀態分割
HASH均勻分佈按 ID 雜湊分散

建立分割表

-- 主表
CREATE TABLE orders (
  id serial,
  created date,
  amount numeric
) PARTITION BY RANGE (created);

-- 分割區
CREATE TABLE orders_2024
  PARTITION OF orders
  FOR VALUES FROM ('2024-01-01')
  TO ('2025-01-01');

-- 預設分割區
CREATE TABLE orders_default
  PARTITION OF orders DEFAULT;

LIST 分割

CREATE TABLE sales (
  region text, amount numeric
) PARTITION BY LIST (region);

CREATE TABLE sales_asia
  PARTITION OF sales
  FOR VALUES IN ('TW','JP','KR');

HASH 分割

CREATE TABLE logs (
  id int, msg text
) PARTITION BY HASH (id);

CREATE TABLE logs_0
  PARTITION OF logs
  FOR VALUES WITH (
    MODULUS 4, REMAINDER 0
  );

多層分割

-- 子分割區也可以再分割
CREATE TABLE orders_2024
  PARTITION OF orders
  FOR VALUES FROM ('2024-01-01')
  TO ('2025-01-01')
  PARTITION BY LIST (region);
分割表可建立索引，會自動繼承到所有分割區
🔧 ALTER TABLE 常用操作 (Ch5.6)
-- 新增/刪除欄位
ALTER TABLE t ADD COLUMN c type;
ALTER TABLE t DROP COLUMN c [CASCADE];

-- 修改欄位
ALTER TABLE t ALTER COLUMN c TYPE new_type;
ALTER TABLE t ALTER COLUMN c SET DEFAULT expr;
ALTER TABLE t ALTER COLUMN c DROP DEFAULT;
ALTER TABLE t ALTER COLUMN c SET NOT NULL;
ALTER TABLE t ALTER COLUMN c DROP NOT NULL;

-- 約束操作
ALTER TABLE t ADD CONSTRAINT name ...;
ALTER TABLE t DROP CONSTRAINT name;
ALTER TABLE t VALIDATE CONSTRAINT name;

-- 重新命名
ALTER TABLE t RENAME TO new_name;
ALTER TABLE t RENAME COLUMN c TO new_c;

-- 擁有者/綱要
ALTER TABLE t OWNER TO role;
ALTER TABLE t SET SCHEMA schema;

-- 分割操作
ALTER TABLE t ATTACH PARTITION p
  FOR VALUES ...;
ALTER TABLE t DETACH PARTITION p
  [CONCURRENTLY];
🆔 產生欄位與序列 (Ch5.3)

IDENTITY 識別欄位（推薦）

CREATE TABLE t (
  id int GENERATED ALWAYS AS IDENTITY,
  -- 或 GENERATED BY DEFAULT AS IDENTITY
  name text
);
ALWAYS：INSERT 時不可手動指定（除非 OVERRIDING SYSTEM VALUE）
BY DEFAULT：可手動指定值，未指定則自動產生

GENERATED 產生欄位

CREATE TABLE t (
  height_cm numeric,
  height_in numeric
    GENERATED ALWAYS AS (height_cm / 2.54) STORED
);
只支援 STORED（實際儲存），不支援 VIRTUAL

Serial（舊式，仍可用）

CREATE TABLE t (
  id serial PRIMARY KEY  -- = int + sequence
);
-- bigserial = bigint + sequence
-- smallserial = smallint + sequence

🧬 資料表繼承 (Ch5.10)
CREATE TABLE cities (
  name text,
  population float,
  elevation int
);
CREATE TABLE capitals (
  state char(2)
) INHERITS (cities);
SELECT * FROM cities查詢含所有子表資料
SELECT * FROM ONLY cities只查詢 cities 本身
ALTER TABLE child NO INHERIT parent解除繼承
繼承 vs 分割：新專案建議使用分割表（Partitioning），繼承是較早的特性，有些限制（如唯一約束不跨子表）
🔐 權限與安全 (Ch5.7)

GRANT / REVOKE

GRANT { SELECT | INSERT | UPDATE | DELETE
       | TRUNCATE | REFERENCES | TRIGGER
       | ALL }
  ON table_name
  TO role_name
  [WITH GRANT OPTION];

REVOKE privilege ON obj FROM role;

列級安全 (RLS)

ALTER TABLE t ENABLE ROW LEVEL SECURITY;

CREATE POLICY p ON t
  FOR { ALL | SELECT | INSERT | UPDATE | DELETE }
  TO role
  USING (tenant_id = current_setting('app.tenant'))
  WITH CHECK (tenant_id = current_setting('app.tenant'));

預設權限

ALTER DEFAULT PRIVILEGES IN SCHEMA public
  GRANT SELECT ON TABLES TO readonly;
💾 儲存與 TOAST (Ch5.12–5.13)

Tablespace 表空間

CREATE TABLESPACE fast_ssd
  LOCATION '/ssd/pg_data';
CREATE TABLE t (...) TABLESPACE fast_ssd;
SET default_tablespace = 'fast_ssd';

TOAST (The Oversized-Attribute Storage Technique)

策略說明
PLAIN不壓縮、不外存（固定長度型別）
EXTENDED先壓縮再外存（預設 for text/bytea）
EXTERNAL不壓縮但外存
MAIN先壓縮，盡量不外存
ALTER TABLE t ALTER COLUMN c
  SET STORAGE EXTERNAL;
超過 ~2KB 的值會觸發 TOAST 機制。壓縮演算法可選 pglz 或 lz4

UNLOGGED TABLE

CREATE UNLOGGED TABLE temp_data (...);
不寫 WAL、不可複寫、crash 後資料遺失，適用於暫時性大量寫入場景
🏷️ Domain、Enum 與自訂型別 (Ch5.14, 8.7)

Domain

CREATE DOMAIN email AS text
  CHECK (VALUE ~ '^[^@]+@[^@]+$');

CREATE DOMAIN positive_int AS int
  CHECK (VALUE > 0);

Enum

CREATE TYPE mood AS ENUM
  ('sad', 'ok', 'happy');

-- 新增值
ALTER TYPE mood ADD VALUE 'excited'
  AFTER 'happy';

Composite Type

CREATE TYPE address AS (
  street text,
  city   text,
  zip    text
);

Range Type

CREATE TYPE float_range AS RANGE (
  SUBTYPE = float8
);
