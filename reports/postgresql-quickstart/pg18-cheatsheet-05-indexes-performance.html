🔨 CREATE INDEX 語法
CREATE [ UNIQUE ] INDEX [ CONCURRENTLY ]
  [ IF NOT EXISTS ] idx_name
  ON tbl [ USING method ]
  ( col | (expr)
    [ opclass ] [ ASC|DESC ]
    [ NULLS FIRST|LAST ] [, ...] )
  [ INCLUDE (col [, ...]) ]
  [ WHERE predicate ]
  [ WITH (params) ]
  [ TABLESPACE name ];
REINDEX INDEX idx;
REINDEX TABLE t;
REINDEX (VERBOSE) DATABASE db;
DROP INDEX [CONCURRENTLY] idx;
🎯 索引進階策略

多欄索引

CREATE INDEX ON orders (customer_id, created_at DESC);
-- 最左前綴：可服務 WHERE customer_id = ?
-- 可服務 WHERE customer_id = ? AND created_at > ?
-- 不可服務 WHERE created_at > ?（無前綴）

表達式索引

CREATE INDEX ON users (lower(email));
CREATE INDEX ON logs ((data->>'type'));

部分索引

CREATE INDEX ON orders (created_at)
  WHERE status = 'pending';

覆蓋索引 (INCLUDE)

CREATE INDEX ON orders (customer_id)
  INCLUDE (total, status);
-- Index-Only Scan 不用回表

CONCURRENTLY（不阻塞寫入）

CREATE INDEX CONCURRENTLY idx ON big_table (col);
-- 失敗→INVALID，需 DROP + 重建

GIN 索引 opclass

CREATE INDEX ON t USING GIN (data);
-- 支援 @> ? ?| ?&
CREATE INDEX ON t USING GIN (data jsonb_path_ops);
-- 只支援 @>，更小更快

BRIN 索引

CREATE INDEX ON logs USING BRIN (created_at)
  WITH (pages_per_range = 128);

🔍 全文搜尋 (Ch12)

核心型別與運算子

tsvector正規化詞彙集（含位置/權重）
tsquery搜尋查詢（& | ! <->）
@@匹配運算子

建立 tsvector / tsquery

to_tsvector('english', 'The quick brown foxes')
-- 'brown':3 'fox':4 'quick':2

-- 帶權重
setweight(to_tsvector('english', title), 'A') ||
setweight(to_tsvector('english', body), 'D')

to_tsquery('english', 'fox & (quick | fast)')
plainto_tsquery('english', 'quick fox')    -- &
phraseto_tsquery('english', 'quick fox')   -- <->
websearch_to_tsquery('english',
  '"quick fox" or brown -lazy')             -- Google 風格

排名與高亮

SELECT title,
  ts_rank(tsv, q) rank,
  ts_rank_cd(tsv, q) rank_cd
FROM articles, to_tsquery('english','database') q
WHERE tsv @@ q ORDER BY rank DESC;

SELECT ts_headline('english', body, q,
  'StartSel=<b>, StopSel=</b>, MaxFragments=3')
FROM articles, to_tsquery(...) q;

GIN 索引加速

ALTER TABLE articles ADD COLUMN tsv tsvector
  GENERATED ALWAYS AS (
    to_tsvector('english', title||' '||body)
  ) STORED;
CREATE INDEX ON articles USING GIN(tsv);
