🔍 SELECT 完整語法 (Ch7)
SELECT [ ALL | DISTINCT [ ON (expr [, ...]) ] ]
    select_list                                      -- ① 要取回的欄位/運算式
[ FROM from_item [, ...] ]                          -- ② 資料來源
[ WHERE condition ]                                  -- ③ 列篩選（聚合前）
[ GROUP BY [ ALL | DISTINCT ] grouping_element [, ...] ]  -- ④ 分組
[ HAVING condition ]                                 -- ⑤ 群組篩選（聚合後）
[ WINDOW window_name AS (window_def) [, ...] ]      -- ⑥ 視窗定義
[ { UNION | INTERSECT | EXCEPT } [ ALL | DISTINCT ] select ] -- ⑦ 集合運算
[ ORDER BY expr [ ASC | DESC ] [ NULLS { FIRST | LAST } ] [, ...] ] -- ⑧ 排序
[ LIMIT { count | ALL } ]                            -- ⑨ 限制列數
[ OFFSET start ]                                     -- ⑩ 跳過列數
[ FETCH { FIRST | NEXT } count { ROW | ROWS } { ONLY | WITH TIES } ] -- SQL 標準分頁
[ FOR { UPDATE | NO KEY UPDATE | SHARE | KEY SHARE } [ OF table [, ...] ] [ NOWAIT | SKIP LOCKED ] ] -- ⑪ 鎖定
執行順序：FROM → WHERE → GROUP BY → HAVING → SELECT → DISTINCT → ORDER BY → LIMIT/OFFSET
📋 FROM 子句與 JOIN (Ch7.2)

JOIN 類型

語法說明
CROSS JOIN笛卡爾積（所有組合）
[INNER] JOIN ... ON內聯結（預設）
LEFT [OUTER] JOIN左外聯結（保留左表所有列）
RIGHT [OUTER] JOIN右外聯結（保留右表所有列）
FULL [OUTER] JOIN完全外聯結（保留雙方所有列）
NATURAL JOIN自動依同名欄位聯結
LATERAL JOIN子查詢可引用外部表欄位

JOIN 條件方式

-- ON 子句
a JOIN b ON a.id = b.a_id

-- USING 子句（同名欄位）
a JOIN b USING (id)

-- NATURAL（所有同名欄位）
a NATURAL JOIN b

LATERAL 子查詢

SELECT u.*, t.total
FROM users u,
  LATERAL (
    SELECT sum(amount) total
    FROM orders o
    WHERE o.user_id = u.id
  ) t;

📊 GROUP BY 與聚合 (Ch7.4)

常用聚合函式

count(*)計數所有列
count(col)計數非 NULL
count(DISTINCT col)計數不重複值
sum(col)總和
avg(col)平均值
min(col) / max(col)最小/最大值
bool_and(col)所有為 true?
bool_or(col)任一為 true?
string_agg(col, ',')字串串接
array_agg(col)聚合為陣列
jsonb_agg(col)聚合為 JSON 陣列
jsonb_object_agg(k,v)聚合為 JSON 物件
percentile_cont(0.5)連續百分位數
percentile_disc(0.5)離散百分位數
mode()眾數

FILTER 子句

SELECT
  count(*) FILTER(WHERE status='active') active_cnt,
  count(*) FILTER(WHERE status='inactive') inactive_cnt
FROM users;

GROUP BY 進階

-- GROUPING SETS
GROUP BY GROUPING SETS (
  (region, product),
  (region),
  (product),
  ()                   -- 全體小計
);

-- ROLLUP = 階層式小計
GROUP BY ROLLUP (region, product);
-- 等同 (region,product),(region),()

-- CUBE = 所有維度組合
GROUP BY CUBE (region, product);
-- 等同 (r,p),(r),(p),()

有序集合聚合

SELECT
  percentile_cont(0.5)
    WITHIN GROUP (ORDER BY salary) AS median
FROM employees;
🔄 DISTINCT 與集合運算 (Ch7.5)

DISTINCT

SELECT DISTINCT region FROM orders;

-- DISTINCT ON (PostgreSQL 特有)
SELECT DISTINCT ON (region) region, city, pop
FROM cities
ORDER BY region, pop DESC;
-- 每個 region 取 pop 最大的一列

集合運算

UNION聯集（去重複）
UNION ALL聯集（保留重複）
INTERSECT交集
EXCEPT差集
集合運算要求欄位數量和型別匹配。UNION ALL 比 UNION 快
🔗 子查詢運算式 (Ch7.6)

純量子查詢

SELECT name,
  (SELECT count(*) FROM orders o
   WHERE o.user_id = u.id) cnt
FROM users u;

EXISTS

SELECT * FROM users u
WHERE EXISTS (
  SELECT 1 FROM orders o
  WHERE o.user_id = u.id
);

IN / ANY / ALL

WHERE id IN (SELECT user_id FROM orders);
WHERE salary > ALL (SELECT salary FROM ...);
WHERE salary > ANY (SELECT salary FROM ...);

列值比較

WHERE (a, b) = (SELECT x, y FROM ...);
WHERE (a, b) IN (SELECT x, y FROM ...);
📦 WITH 子句 (CTE) (Ch7.8)

基本 CTE

WITH regional_sales AS (
  SELECT region, sum(amount) total
  FROM orders
  GROUP BY region
),
top_regions AS (
  SELECT region
  FROM regional_sales
  WHERE total > (
    SELECT avg(total) FROM regional_sales
  )
)
SELECT * FROM orders
WHERE region IN (SELECT * FROM top_regions);

遞迴 CTE

WITH RECURSIVE tree AS (
  -- 基底 (非遞迴項)
  SELECT id, name, parent_id, 0 AS depth
  FROM categories
  WHERE parent_id IS NULL

  UNION ALL

  -- 遞迴項
  SELECT c.id, c.name, c.parent_id, t.depth+1
  FROM categories c
  JOIN tree t ON c.parent_id = t.id
)
SELECT * FROM tree;

CTE 實體化控制 (PG12+)

WITH cte AS MATERIALIZED (...)       -- 強制實體化（結果會暫存）
WITH cte AS NOT MATERIALIZED (...)   -- 允許內聯展開（可能更快）

可寫入 CTE

WITH deleted AS (
  DELETE FROM orders WHERE status = 'cancelled' RETURNING *
)
INSERT INTO order_archive SELECT * FROM deleted;

➕ INSERT (Ch6.1)
-- 基本插入
INSERT INTO t (c1, c2)
VALUES (1, 'a'), (2, 'b');

-- 從查詢插入
INSERT INTO t2 SELECT * FROM t1;

-- 取回插入值
INSERT INTO t (name) VALUES ('x')
RETURNING id, name;

-- UPSERT (ON CONFLICT)
INSERT INTO t (id, name, cnt)
VALUES (1, 'a', 1)
ON CONFLICT (id) DO UPDATE SET
  cnt = t.cnt + EXCLUDED.cnt;

-- ON CONFLICT DO NOTHING
INSERT INTO t (id, name)
VALUES (1, 'a')
ON CONFLICT DO NOTHING;

-- OVERRIDING（IDENTITY 欄位）
INSERT INTO t (OVERRIDING SYSTEM VALUE)
VALUES (999, 'force');
✏️ UPDATE (Ch6.2)
-- 基本更新
UPDATE t SET c1 = 10 WHERE id = 1;

-- 多欄位
UPDATE t SET (c1, c2) = (10, 20) WHERE ...;

-- 從其他表取值
UPDATE t1 SET c1 = t2.val
FROM t2
WHERE t1.id = t2.t1_id;

-- 用子查詢
UPDATE t SET c1 = (SELECT max(v) FROM t2);

-- RETURNING
UPDATE t SET c1 = c1 + 1
RETURNING id, c1;
🗑️ DELETE / TRUNCATE (Ch6.3)
-- 條件刪除
DELETE FROM t WHERE status = 'expired';

-- 用其他表條件
DELETE FROM t1
USING t2
WHERE t1.ref_id = t2.id
  AND t2.obsolete = true;

-- RETURNING
DELETE FROM t WHERE ... RETURNING *;

-- 刪除所有列（快速，但不可回滾）
TRUNCATE t [RESTART IDENTITY] [CASCADE];

-- 多表同時 TRUNCATE
TRUNCATE t1, t2, t3 CASCADE;
TRUNCATE 比 DELETE 快很多（不逐列刪除），但不觸發 DELETE 觸發器。不可 RETURNING
🔀 MERGE (PG15+) (Ch6.4)
MERGE INTO target t
USING source s ON t.id = s.id
WHEN MATCHED AND s.qty = 0 THEN
  DELETE
WHEN MATCHED THEN
  UPDATE SET
    name = s.name,
    qty = s.qty
WHEN NOT MATCHED THEN
  INSERT (id, name, qty)
  VALUES (s.id, s.name, s.qty)
RETURNING merge_action(), t.*;
PG17+ 支援 RETURNING 子句和 merge_action() 函式
MERGE 是 SQL 標準的 UPSERT/同步操作，比 INSERT ON CONFLICT 更靈活

📑 排序與分頁
-- 基本排序
ORDER BY col ASC      -- 預設升冪
ORDER BY col DESC
ORDER BY col NULLS FIRST
ORDER BY col NULLS LAST
ORDER BY col USING <   -- 自訂運算子

-- LIMIT / OFFSET
LIMIT 10 OFFSET 20

-- SQL 標準語法
FETCH FIRST 10 ROWS ONLY
FETCH FIRST 10 ROWS WITH TIES
-- WITH TIES 會包含與第10名相同排序值的額外列
效能注意：大 OFFSET 會掃描並丟棄列。大型分頁建議使用鍵集分頁 (keyset pagination)：
WHERE id > last_seen_id ORDER BY id LIMIT 10
🔒 SELECT FOR UPDATE / SHARE (Ch7.9)
鎖定模式用途
FOR UPDATE最強鎖；準備更新/刪除時
FOR NO KEY UPDATE較弱的更新鎖（不影響外鍵檢查）
FOR SHARE共享鎖；防止他人更新
FOR KEY SHARE最弱鎖；只防止刪除和主鍵更新

選項

SELECT * FROM t
WHERE id = 1
FOR UPDATE
  OF t             -- 指定鎖定哪個表
  NOWAIT;          -- 無法鎖定時立即報錯

FOR UPDATE SKIP LOCKED; -- 跳過已鎖定的列
-- 適用於工作佇列場景
