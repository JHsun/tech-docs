🗑️ DDL — DROP 刪除
DROP { TABLE | INDEX | VIEW | SEQUENCE
     | FUNCTION | PROCEDURE | TYPE
     | SCHEMA | DATABASE | ROLE
     | TRIGGER | EXTENSION | DOMAIN
     | POLICY | PUBLICATION | SUBSCRIPTION
     | SERVER | FOREIGN TABLE | ... }
  [ IF EXISTS ] name [, ...]
  [ CASCADE | RESTRICT ];
IF EXISTS不存在時不報錯
CASCADE連帶刪除依賴物件
RESTRICT有依賴時拒絕（預設）
-- 特殊 DROP
DROP INDEX CONCURRENTLY idx; -- 不鎖表
DROP DATABASE db WITH (FORCE); -- 強制斷線
DROP OWNED BY role;  -- 刪除角色擁有的所有物件
REASSIGN OWNED BY old TO new; -- 移轉擁有權

🔐 權限管理
GRANT { SELECT | INSERT | UPDATE | DELETE
       | TRUNCATE | REFERENCES | TRIGGER
       | ALL [PRIVILEGES] }
ON { TABLE t | ALL TABLES IN SCHEMA s }
TO role [WITH GRANT OPTION];

-- 其他物件權限
GRANT USAGE ON SCHEMA s TO role;
GRANT CONNECT ON DATABASE db TO role;
GRANT EXECUTE ON FUNCTION f TO role;
GRANT USAGE ON SEQUENCE seq TO role;
GRANT CREATE ON DATABASE db TO role;

-- 角色成員
GRANT group_role TO user_role;

-- 欄位級權限
GRANT SELECT (col1, col2) ON t TO role;
GRANT UPDATE (col1) ON t TO role;

-- 撤銷
REVOKE priv ON obj FROM role [CASCADE];
📌 預備語句與游標

PREPARE / EXECUTE

PREPARE my_stmt (int, text) AS
  SELECT * FROM users
  WHERE id = $1 AND name = $2;
EXECUTE my_stmt(1, 'Alice');
DEALLOCATE my_stmt;

CURSOR 游標

BEGIN;
DECLARE cur CURSOR FOR
  SELECT * FROM big_table;
FETCH 100 FROM cur;  -- 取 100 列
FETCH NEXT FROM cur;  -- 取下一列
MOVE FORWARD 10 FROM cur;  -- 跳過
CLOSE cur;
COMMIT;

LISTEN / NOTIFY

-- 監聽（客戶端 A）
LISTEN my_channel;

-- 發送通知（客戶端 B）
NOTIFY my_channel, 'payload_data';
-- 或
SELECT pg_notify('my_channel', 'data');

UNLISTEN my_channel;
UNLISTEN *;

  --old-datadir=/data/pg17 \
  --new-datadir=/data/pg18 \

  --new-bindir=/usr/pgsql-18/bin \
