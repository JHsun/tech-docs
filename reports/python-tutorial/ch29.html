§29.1檔案批次處理

python 
from pathlib import Path
import shutil

# 1. 找特定檔案
root = Path("/path/to/docs")
list(root.glob("*.pdf"))                     # 目前層 PDF
list(root.rglob("*.jpg"))                    # 遞迴找 JPG
list(root.glob("**/2025*.xlsx"))              # 所有層

# 2. 批次改名（加日期前綴）
from datetime import datetime
prefix = datetime.now().strftime("%Y%m%d_")
for f in root.glob("*.jpg"):
    f.rename(f.parent / (prefix + f.name))

# 3. 依類型歸類到不同資料夾
extensions = {
    "images": {".jpg", ".png", ".gif"},
    "docs": {".pdf", ".docx", ".txt"},
    "videos": {".mp4", ".mov"},
}
for f in root.iterdir():
    if not f.is_file(): continue
    for folder, exts in extensions.items():
        if f.suffix.lower() in exts:
            target = root / folder
            target.mkdir(exist_ok=True)
            shutil.move(f, target / f.name)
            break

# 4. 找重複檔案（用雜湊）
import hashlib
from collections import defaultdict

def file_hash(path, chunk=8192):
    h = hashlib.sha256()
    with open(path, "rb") as f:
        for b in iter(lambda: f.read(chunk), b""):
            h.update(b)
    return h.hexdigest()

groups = defaultdict(list)
for f in root.rglob("*"):
    if f.is_file():
        groups[file_hash(f)].append(f)

duplicates = {h: paths for h, paths in groups.items() if len(paths) > 1}

# 5. 清理：刪除 N 天前的檔案
import time
now = time.time()
cutoff = now - 30 * 86400                    # 30 天
for f in root.rglob("*.log"):
    if f.stat().st_mtime < cutoff:
        f.unlink()
        print(f"已刪除: {f}")

ZIP 壓縮/解壓

python 
import zipfile

# 打包整個目錄
with zipfile.ZipFile("backup.zip", "w", zipfile.ZIP_DEFLATED) as zf:
    for f in Path("docs").rglob("*"):
        zf.write(f, arcname=f.relative_to("docs"))

# 解壓
with zipfile.ZipFile("archive.zip") as zf:
    zf.extractall("output")

# 也可用 shutil（更簡單）
shutil.make_archive("backup", "zip", "docs")
shutil.unpack_archive("archive.zip", "output")

§29.2Excel 自動化

💡 Excel 套件選擇
  • pandas / polars：讀寫資料 → 最快最順
  • openpyxl：需要寫公式、儲存格格式、圖表、合併格時用
  • xlsxwriter：寫入效能最佳，但只能寫不能讀
  • xlwings：需要自動化已開啟的 Excel（macro 風格）

讀寫資料（pandas / polars）

with pd.ExcelWriter("out.xlsx", engine="openpyxl") as w:

# uv add openpyxl
from openpyxl import Workbook, load_workbook
from openpyxl.styles import Font, PatternFill, Alignment
from openpyxl.utils import get_column_letter

wb = Workbook()
ws = wb.active
ws.title = "月報"

# 寫入
ws["A1"] = "項目"
ws["B1"] = "金額"
ws.append(["電費", 3500])
ws.append(["水費", 800])

# 公式
ws["B4"] = "=SUM(B2:B3)"

# 樣式
header_font = Font(bold=True, color="FFFFFF")
header_fill = PatternFill("solid", fgColor="3b82f6")

for cell in ws[1]:
    cell.font = header_font
    cell.fill = header_fill
    cell.alignment = Alignment(horizontal="center")

# 調欄寬
ws.column_dimensions["A"].width = 20
ws.column_dimensions["B"].width = 15

# 合併儲存格
ws.merge_cells("A6:B6")
ws["A6"] = "2026 年 4 月"

# 加圖表
from openpyxl.chart import BarChart, Reference
chart = BarChart()
data_ref = Reference(ws, min_col=2, min_row=1, max_row=3)
cats = Reference(ws, min_col=1, min_row=2, max_row=3)
chart.add_data(data_ref, titles_from_data=True)
chart.set_categories(cats)
ws.add_chart(chart, "D2")

wb.save("report.xlsx")

# 讀取現有檔案並修改
wb = load_workbook("existing.xlsx")

§29.3PDF 操作

bash 
uv add pypdf reportlab

讀取 + 合併 + 拆分

python 
from pypdf import PdfReader, PdfWriter

# 讀文字
reader = PdfReader("document.pdf")
print(len(reader.pages), "頁")
print(reader.pages[0].extract_text())
meta = reader.metadata
print(meta.title, meta.author)

# 合併多個 PDF
writer = PdfWriter()
for pdf_file in ["a.pdf", "b.pdf", "c.pdf"]:
    reader = PdfReader(pdf_file)
    for page in reader.pages:
        writer.add_page(page)

with open("merged.pdf", "wb") as f:
    writer.write(f)

# 拆分：每頁一個檔
reader = PdfReader("big.pdf")
for i, page in enumerate(reader.pages, 1):
    writer = PdfWriter()
    writer.add_page(page)
    with open(f"page_{i:03d}.pdf", "wb") as f:
        writer.write(f)

# 加密
writer = PdfWriter(clone_from="input.pdf")
writer.encrypt("mypassword")
with open("locked.pdf", "wb") as f:
    writer.write(f)

生成 PDF（reportlab）

python 
from reportlab.lib.pagesizes import A4
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle
from reportlab.lib.styles import getSampleStyleSheet
from reportlab.lib import colors

doc = SimpleDocTemplate("report.pdf", pagesize=A4)
styles = getSampleStyleSheet()

elements = [
    Paragraph("2026 Q1 月報", styles["Title"]),
    Spacer(1, 12),
    Paragraph("本季業績成長 15%...", styles["BodyText"]),
    Spacer(1, 20),
    Table(
        [["月份", "金額"], ["一月", "$50k"], ["二月", "$60k"]],
        style=TableStyle([
            ("BACKGROUND", (0, 0), (-1, 0), colors.lightblue),
            ("GRID", (0, 0), (-1, -1), 0.5, colors.grey),
        ]),
    ),
]

doc.build(elements)

§29.4Email 自動化

python 
import smtplib
from email.message import EmailMessage
from pathlib import Path

msg = EmailMessage()
msg["From"] = "me@example.com"
msg["To"] = "you@example.com"
msg["Cc"] = "boss@example.com"
msg["Subject"] = "月報"
msg.set_content("您好，附件為本月報表")

# HTML 版本
msg.add_alternative("""
<html><body>
<h2>月報</h2>
<p>請見附件, subtype="html")

# 附件
for path in [Path("report.pdf"), Path("data.xlsx")]:
    msg.add_attachment(
        path.read_bytes(),
        maintype="application",
        subtype="octet-stream",
        filename=path.name,
    )

# 寄出（Gmail 示範，需 app password）
import os
with smtplib.SMTP_SSL("smtp.gmail.com", 465) as s:
    s.login(os.environ["EMAIL"], os.environ["EMAIL_APP_PWD"])
    s.send_message(msg)
⚠️ 注意事項
  • Gmail 需要開 2FA 並生成「應用程式密碼」，不能直接用登入密碼
  • 密碼請用環境變數或 secret manager，不要硬編碼
  • 大量寄信會被標記為 spam，用 SendGrid / Mailgun / AWS SES 等專業服務
  • 需要讀郵件（不只寄）用 imaplib / imap-tools

§29.5GUI 自動化 pyautogui

自動控制滑鼠、鍵盤、截圖 —— 適合自動化無 API 的桌面應用（老系統、會計軟體等）。

bash 
uv add pyautogui pillow
python 
import pyautogui
import time

# 安全設定（重要！）
pyautogui.FAILSAFE = True          # 滑鼠移到角落 → 中止腳本
pyautogui.PAUSE = 0.5              # 每個操作後暫停 0.5s

# 螢幕資訊
pyautogui.size()                   # (1920, 1080)
pyautogui.position()               # 目前滑鼠位置

# 滑鼠
pyautogui.moveTo(500, 300, duration=0.5)    # 移到座標
pyautogui.moveRel(100, 0)                    # 相對移動
pyautogui.click()                            # 左鍵
pyautogui.rightClick()
pyautogui.doubleClick()
pyautogui.dragTo(700, 500, duration=1)
pyautogui.scroll(-500)                        # 負數向下

# 鍵盤
pyautogui.typewrite("Hello World", interval=0.05)
pyautogui.press("enter")
pyautogui.hotkey("ctrl", "s")                 # 複合鍵

# ⚠️ pyautogui.typewrite 不支援中文
# 中文用剪貼簿 + Ctrl+V
import pyperclip
pyperclip.copy("你好世界")
pyautogui.hotkey("ctrl", "v")

# 截圖
pyautogui.screenshot("screen.png")
region_img = pyautogui.screenshot(region=(100, 100, 300, 200))

# 影像辨識：在畫面上找按鈕圖片
pos = pyautogui.locateOnScreen("submit_button.png", confidence=0.9)
if pos:
    pyautogui.click(pyautogui.center(pos))

# 彈出對話框
name = pyautogui.prompt("輸入你的名字：")
pyautogui.alert("完成！")
if pyautogui.confirm("繼續嗎？") == "OK":
    ...
⚠️ GUI 自動化的脆弱性
  • 解析度、字型、主題不同 → 座標偏移
  • 應用程式更新後 → 按鈕位置變動
  • 建議：用 圖像辨識（locateOnScreen）比固定座標穩定
  • 先嘗試 API、再嘗試網頁自動化（Playwright）、最後才 pyautogui
  • Windows 桌面應用有更專業的 pywinauto

§29.6檔案監控 watchdog

檔案/目錄一有變動就觸發動作 —— 例如新檔上傳 → 自動壓縮、程式碼有變 → 自動重跑測試。

python 
# uv add watchdog
import time
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler, PatternMatchingEventHandler

class MyHandler(PatternMatchingEventHandler):
    def __init__(self):
        super().__init__(patterns=["*.csv", "*.xlsx"], ignore_directories=True)

    def on_created(self, event):
        print(f"新檔: {event.src_path}")
        # 處理新檔…

    def on_modified(self, event):
        print(f"修改: {event.src_path}")

    def on_deleted(self, event):
        print(f"刪除: {event.src_path}")

    def on_moved(self, event):
        print(f"移動: {event.src_path} -> {event.dest_path}")

observer = Observer()
observer.schedule(MyHandler(), path="./data", recursive=True)
observer.start()

try:
    while True:
        time.sleep(1)
except KeyboardInterrupt:
    observer.stop()
observer.join()

§29.7排程執行

方式 1：schedule（簡單）

python 
# uv add schedule
import schedule
import time

def job():
    print("執行任務！")

# 各種排程
schedule.every(10).seconds.do(job)
schedule.every(1).hour.do(job)
schedule.every().day.at("09:00").do(job)
schedule.every().monday.at("10:30").do(job)
schedule.every(5).to(10).minutes.do(job)      # 隨機 5~10 分鐘

while True:
    schedule.run_pending()
    time.sleep(1)

方式 2：APScheduler（進階）

python 
# uv add apscheduler
from apscheduler.schedulers.blocking import BlockingScheduler
from apscheduler.triggers.cron import CronTrigger

sched = BlockingScheduler()

# Cron 風格
@sched.scheduled_job(CronTrigger.from_crontab("0 9 * * MON"))
def weekly_report():
    print("週一早上 9 點執行")

# 固定間隔
@sched.scheduled_job("interval", minutes=30)
def heartbeat():
    print("每 30 分鐘")

# 特定時間
@sched.scheduled_job("date", run_date="2026-12-31 23:59:00")
def new_year():
    print("跨年")

sched.start()

方式 3：系統 cron（生產首選）

bash 
# Unix crontab 格式
# 分 時 日 月 週 命令
# 0 9 * * MON  每週一 9 點
crontab -e

# 加入這一行：每天 2 點跑備份
0 2 * * * /usr/bin/uv run /path/to/backup.py >> /var/log/backup.log 2>&1

# Windows 用工作排程器（Task Scheduler）
# 或用 WSL + cron
💡 排程選擇
  • simple Python：個人腳本、桌面用 → schedule
  • 可靠服務：長期運行、多任務、持久化 → APScheduler（SQLAlchemy JobStore）
  • 生產部署：Linux → cron、Windows → Task Scheduler
  • 分散式任務：大型應用 → Celery + Redis / RabbitMQ
  • 雲端：AWS EventBridge、GCP Cloud Scheduler、K8s CronJob
