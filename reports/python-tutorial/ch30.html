§30.1sqlite3 —— 標準庫內建

python 
import sqlite3

with sqlite3.connect("app.db") as conn:
    conn.row_factory = sqlite3.Row     # 結果可用欄位名存取
    cur = conn.cursor()

    # 建表
    cur.execute("""
        CREATE TABLE IF NOT EXISTS users (
            id INTEGER PRIMARY KEY,
            name TEXT NOT NULL,
            email TEXT UNIQUE,
            age INTEGER
        )
    """)

    # ⚠️ 永遠用參數化 query（防 SQL injection）
    cur.execute(
        "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
        ("Alice", "alice@example.com", 30)
    )

    # 批次插入
    cur.executemany(
        "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
        [("Bob", "bob@x.com", 25), ("Charlie", "c@x.com", 35)]
    )

    conn.commit()

    # 查詢
    cur.execute("SELECT * FROM users WHERE age > ?", (20,))
    for row in cur.fetchall():
        print(row["name"], row["email"])

    # 一筆
    cur.execute("SELECT * FROM users WHERE id=?", (1,))
    user = cur.fetchone()
⚠️ SQL Injection 永遠要防
# ❌ 危險！名字含 "; DROP TABLE ..." 就毀了
cur.execute(f"SELECT * FROM users WHERE name = '{name}'")

# ✅ 參數化
cur.execute("SELECT * FROM users WHERE name = ?", (name,))

🆕 SQLAlchemy 2.0 核心理念
  • 型別友善：用 Mapped[...] + mapped_column()
  • 統一 API：ORM 與 Core 介面統一（都用 select()）
  • Session 一律顯式：全部用 session.execute(stmt)，不再有舊版的隱式查詢
  • 原生 asyncio：AsyncSession 完整支援

§30.3定義模型（現代風格）

python 
from datetime import datetime
from typing import Optional
from sqlalchemy import String, ForeignKey, func
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship

class Base(DeclarativeBase):
    pass

class User(Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(120), unique=True, index=True)
    age: Mapped[int | None] = mapped_column(default=None)   # nullable
    created_at: Mapped[datetime] = mapped_column(
        server_default=func.now()                    # DB 自己填
    )

    # 關聯（另一邊是 Post）
    posts: Mapped[list["Post"]] = relationship(
        back_populates="author", cascade="all, delete-orphan"
    )

    def __repr__(self) -> str:
        return f"User(id={self.id!r}, name={self.name!r})"

class Post(Base):
    __tablename__ = "posts"

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    content: Mapped[str]
    user_id: Mapped[int] = mapped_column(ForeignKey("users.id"))

    author: Mapped["User"] = relationship(back_populates="posts")

建立 engine 與 session

python 
from sqlalchemy import create_engine
from sqlalchemy.orm import Session, sessionmaker

# 連線字串（driver 由 + 後指定）
#   postgresql+psycopg://user:pwd@host/db
#   postgresql+asyncpg://user:pwd@host/db     ← async
#   mysql+pymysql://user:pwd@host/db
#   sqlite:///./app.db
#   sqlite+aiosqlite:///./app.db              ← async

engine = create_engine(
    "postgresql+psycopg://user:pwd@localhost/mydb",
    echo=False,              # True 印出 SQL
    pool_size=10,             # 連線池大小
    max_overflow=20,          # 臨時可超出數量
    pool_pre_ping=True,       # 用前確認連線健康
    pool_recycle=3600,        # 一小時回收連線
)

# 建立所有表（開發用，生產用 Alembic）
Base.metadata.create_all(engine)

# Session 工廠
SessionLocal = sessionmaker(engine, expire_on_commit=False)

§30.4CRUD 操作

python 
from sqlalchemy import select, update, delete

# Create
with SessionLocal() as session:
    alice = User(name="Alice", email="alice@x.com", age=30)
    session.add(alice)
    session.commit()
    print(alice.id)                # 寫入後取 PK

    # 批次新增
    session.add_all([
        User(name="Bob", email="bob@x.com"),
        User(name="Charlie", email="c@x.com"),
    ])
    session.commit()

# Read — SQLAlchemy 2.0 風格
with SessionLocal() as session:
    # 取一筆（by PK）
    user = session.get(User, 1)

    # select + scalar_one
    stmt = select(User).where(User.email == "alice@x.com")
    user = session.scalar(stmt)          # 一個或 None
    # user = session.scalars(stmt).one()    # 一個，沒就例外

    # 全部
    stmt = select(User).order_by(User.name)
    users = session.scalars(stmt).all()

    # 迭代（大量資料友善）
    for user in session.scalars(stmt):
        print(user.name)

# Update — 改屬性
with SessionLocal() as session:
    user = session.get(User, 1)
    user.age = 31
    session.commit()

# 批次 update（不載入物件）
with SessionLocal() as session:
    stmt = (update(User)
            .where(User.age < 20)
            .values(age=20))
    session.execute(stmt)
    session.commit()

# Delete
with SessionLocal() as session:
    user = session.get(User, 1)
    session.delete(user)
    session.commit()

# 批次 delete
stmt = delete(User).where(User.age < 18)
session.execute(stmt)
session.commit()

交易（transaction）

python 
# 方式 1：with session.begin() 自動提交/回滾
with SessionLocal() as session, session.begin():
    session.add(user)
    session.add(post)
    # 離開時若無例外 → commit
    # 有例外 → rollback

# 方式 2：手動
with SessionLocal() as session:
    try:
        session.add(user)
        session.commit()
    except:
        session.rollback()
        raise

§30.5查詢進階

python 
from sqlalchemy import select, and_, or_, func, desc

# 多條件
stmt = select(User).where(
    User.age >= 18,
    User.age < 65,
)                               # AND 是預設
# 或顯式
stmt = select(User).where(or_(User.age < 18, User.age >= 65))

# IN / NOT IN
stmt = select(User).where(User.id.in_([1, 2, 3]))
stmt = select(User).where(User.id.not_in([1, 2]))

# 模糊
stmt = select(User).where(User.name.like("A%"))
stmt = select(User).where(User.name.ilike("%alice%"))   # 不分大小寫 (PostgreSQL)

# 排序 / 分頁
stmt = (select(User)
        .order_by(desc(User.created_at))
        .limit(10)
        .offset(20))

# 聚合
stmt = select(func.count(User.id))
count = session.scalar(stmt)

# group by
stmt = (select(
            User.age,
            func.count(User.id).label("n"),
         )
         .group_by(User.age)
         .having(func.count(User.id) > 1))
for row in session.execute(stmt):
    print(row.age, row.n)

# JOIN
stmt = (select(Post.title, User.name)
        .join(User, Post.user_id == User.id)
        .where(User.age > 30))

# 自動透過 relationship join
stmt = (select(Post)
        .join(Post.author)
        .where(User.age > 30))

# 原生 SQL（進階才用）
from sqlalchemy import text
result = session.execute(
    text("SELECT * FROM users WHERE age > :min_age"),
    {"min_age": 18}
)

§30.6關聯載入策略

python 
from sqlalchemy.orm import selectinload, joinedload, lazyload

# ❌ N+1 問題：每個 user 發一次 query 取 posts
for user in session.scalars(select(User)):
    print(user.name, len(user.posts))    # 🚫 每次都查 DB

# ✅ selectinload：另一次 IN 查詢載入所有關聯
stmt = select(User).options(selectinload(User.posts))
for user in session.scalars(stmt):
    print(user.name, len(user.posts))    # 只發 2 個 query

# ✅ joinedload：JOIN 一次搞定（小量資料較快）
stmt = select(User).options(joinedload(User.posts))

# 深層巢狀
stmt = select(User).options(
    selectinload(User.posts).selectinload(Post.comments)
)

§30.7異步 SQLAlchemy

python 
import asyncio
from sqlalchemy.ext.asyncio import (
    AsyncSession, create_async_engine, async_sessionmaker
)
from sqlalchemy import select

# 1. 建立異步 engine
engine = create_async_engine(
    "postgresql+asyncpg://user:pwd@localhost/mydb",
    echo=False,
    pool_size=10,
)

# 2. Session 工廠
AsyncSessionLocal = async_sessionmaker(
    engine, expire_on_commit=False
)

# 3. 建表（異步）
async def init_db():
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)

# 4. 查詢
async def get_users():
    async with AsyncSessionLocal() as session:
        stmt = select(User).where(User.age > 18)
        result = await session.execute(stmt)
        return result.scalars().all()

# 5. 新增
async def create_user(name: str, email: str):
    async with AsyncSessionLocal() as session:
        async with session.begin():
            user = User(name=name, email=email)
            session.add(user)
        await session.refresh(user)
        return user

# 6. FastAPI 整合（依賴注入）
async def get_db():
    async with AsyncSessionLocal() as session:
        yield session

@app.get("/users")
async def list_users(db: Annotated[AsyncSession, Depends(get_db)]):
    result = await db.execute(select(User))
    return result.scalars().all()

asyncio.run(init_db())
⚠️ async 下的 lazy load 陷阱

異步 session 不能用 lazy load（預設）—— 取完資料後呼叫 user.posts 會拋錯。一律用 selectinload 或在交易內一次取完需要的資料。

# 產生 SQL（不執行，給 DBA 審核）
alembic upgrade head --sql

典型 migration 檔案

python 
"""add email_verified column

Revision ID: abc123
Revises: def456
"""
from alembic import op
import sqlalchemy as sa

revision = "abc123"
down_revision = "def456"

def upgrade() -> None:
    op.add_column("users",
        sa.Column("email_verified", sa.Boolean(), server_default="false")
    )
    op.create_index("ix_users_email_verified", "users", ["email_verified"])

def downgrade() -> None:
    op.drop_index("ix_users_email_verified", "users")
    op.drop_column("users", "email_verified")
💡 autogenerate 常見問題
  • 空的 migration：env.py 沒 import 所有 model（Base.metadata 是空的）
  • 無法偵測改動：autogenerate 對某些變更（如複雜約束、資料）不敏感，需手動補
  • 一定要人工審核：生成的 migration 只是草稿，commit 前務必檢查
  • 複雜資料搬移可以在 upgrade() 內用 op.get_bind() 取得連線執行 SQL

§30.9SQLModel —— FastAPI 完美搭檔

FastAPI 作者 tiangolo 設計的 ORM：一個 class 同時是 SQLAlchemy 模型 + Pydantic schema，省掉重複定義。

python 
# uv add sqlmodel
from sqlmodel import Field, Session, SQLModel, create_engine, select

class Hero(SQLModel, table=True):
    id: int | None = Field(default=None, primary_key=True)
    name: str = Field(index=True)
    age: int | None = None

engine = create_engine("sqlite:///heroes.db")
SQLModel.metadata.create_all(engine)

# 用起來就像 Pydantic + SQLAlchemy 合一
with Session(engine) as session:
    hero = Hero(name="Spider-Man", age=18)
    session.add(hero)
    session.commit()
    session.refresh(hero)

    heroes = session.exec(
        select(Hero).where(Hero.age < 30)
    ).all()

# 在 FastAPI 中可同時作為 request body & response model！
@app.post("/heroes", response_model=Hero)
def create(hero: Hero):
    with Session(engine) as s:
        s.add(hero)
        s.commit()
        s.refresh(hero)
    return hero
💡 SQLModel vs 純 SQLAlchemy
  • 簡單 FastAPI 專案 → SQLModel（少寫一半程式碼）
  • 複雜查詢、多對多、進階 ORM 特性 → 純 SQLAlchemy（彈性更大）
  • 需要 Django 風 admin → Django ORM
