CHAPTER 25 · PART 7

爬蟲技術

網路是最大的公開資料庫。爬蟲能把散落在頁面中的資訊結構化 —— 從商品價格、新聞、論文到社群資料。本章從合法性、靜態頁面的 BeautifulSoup，到 JavaScript 渲染的 Playwright、反爬蟲對策、並行爬取，最後示範完整 pipeline。

Python 3.14 bs4 · Playwright · 反爬 預估閱讀 28 分鐘

§0章節導論

爬蟲技術棧：

  • HTTP 請求：httpx（上章已介紹）
  • HTML 解析：BeautifulSoup、lxml、selectolax（最快）
  • JavaScript 渲染：Playwright、Selenium
  • 並行爬取：asyncio + httpx
  • 企業級框架：Scrapy

§25.1倫理與合法性

⚠️ 爬蟲的底線
  • 遵守 robots.txt：網站明確禁止的路徑不要爬
  • 尊重 Terms of Service：許多網站 ToS 禁止自動抓取
  • 限速：加 time.sleep()、限制並發數，不要打垮對方
  • 標示 User-Agent：讓網站知道誰在抓、聯絡方式
  • 個資法規：GDPR、CCPA、台灣個資法 — 不要亂抓個人資料
  • 版權：抓到的內容不等於擁有版權
  • 有 API 就用 API：官方 API 優於爬蟲

rp.set_url("https://example.com/robots.txt")

python 
import httpx
from bs4 import BeautifulSoup

r = httpx.get("https://news.example.com", timeout=10)
soup = BeautifulSoup(r.text, "lxml")    # 也可用 "html.parser"

# 基本導航
soup.title              # <title>...</title>
soup.title.text         # 純文字
soup.title.name         # 'title'
soup.title.string

# find / find_all
soup.find("h1")            # 第一個 h1
soup.find_all("a")         # 所有 a 標籤
soup.find_all("div", class_="card")
soup.find_all("div", id="main")
soup.find_all("a", href=True)   # 有 href 屬性的
# 正則
import re
soup.find_all("a", href=re.compile(r"^https://"))

# 屬性存取
a = soup.find("a")
a["href"]               # 'https://...'
a.get("href", "")        # 不存在回預設值
a.text                  # 文字（含子元素）
a.get_text(strip=True)    # 去前後空白

# select — CSS selector（更直覺）
soup.select("div.card h2 a")
soup.select("a.external[href^='https']")
soup.select_one("#main-title")

# 遍歷
for article in soup.select("article.post"):
    title = article.select_one("h2").get_text(strip=True)
    url = article.select_one("a")["href"]
    print(title, url)
💡 解析器速度
  • html.parser：純 Python，無相依，慢
  • lxml：C 實作，快 10 倍以上（推薦）
  • selectolax：Rust 實作，更快，API 稍不同

CSS Selector 速查

語法意義

.clsclass="cls"
#idid="id"

div p後代（任意深度）
div + p相鄰兄弟

tree = html.fromstring(r.text)

# XPath 有時比 CSS 選擇器更強大（特別是「包含某文字」）

§25.4Playwright —— JavaScript 動態網站

現代單頁應用（SPA）初始 HTML 幾乎沒內容 —— 資料是 JS 執行後才填入。httpx + bs4 抓不到。此時需要 瀏覽器自動化。

bash 
uv add playwright
uv run playwright install chromium    # 下載瀏覽器（首次）
python · 同步版 
from playwright.sync_api import sync_playwright

with sync_playwright() as p:
    browser = p.chromium.launch(headless=True)
    page = browser.new_page()

    page.goto("https://example.com/spa")

    # 等待某元素出現
    page.wait_for_selector("div.product", timeout=10000)

    # 取得渲染後的 HTML
    html = page.content()

    # 直接用 Playwright 的選擇器
    titles = page.query_selector_all("div.product h2")
    for t in titles:
        print(t.inner_text())

    # 或 Locator API（更穩健）
    for item in page.locator("div.product").all():
        name = item.locator("h2").text_content()
        price = item.locator(".price").text_content()

    # 互動：點擊、輸入
    page.fill("input#search", "laptop")
    page.click("button.submit")
    page.wait_for_load_state("networkidle")

    # 截圖
    page.screenshot(path="result.png", full_page=True)

    browser.close()

異步版（推薦，效能更好）

python 
import asyncio
from playwright.async_api import async_playwright

async def scrape(url):
    async with async_playwright() as p:
        browser = await p.chromium.launch()
        page = await browser.new_page()
        await page.goto(url)
        await page.wait_for_selector("article")
        data = await page.evaluate("""() => {
            return Array.from(document.querySelectorAll("article"))
                .map(a => ({
                    title: a.querySelector("h2").innerText,
                    link: a.querySelector("a").href,
                }));
        }""")
        await browser.close()
        return data

data = asyncio.run(scrape("https://spa.example.com"))

§25.6並行爬蟲

python 
import asyncio
import httpx
from bs4 import BeautifulSoup

async def fetch_page(client, url, sem):
    async with sem:           # 限制同時並發數
        try:
            async with asyncio.timeout(10):
                r = await client.get(url)
                r.raise_for_status()
                return url, r.text
        except Exception as e:
            print(f"[失敗] {url}: {e}")
            return url, None

def parse(html):
    soup = BeautifulSoup(html, "lxml")
    return {
        "title": soup.title.text.strip() if soup.title else "",
        "links": [a["href"] for a in soup.select("a[href]")],
    }

async def crawl(urls, concurrency=10):
    sem = asyncio.Semaphore(concurrency)
    headers = {"User-Agent": "MyBot 1.0 (contact@example.com)"}

    async with httpx.AsyncClient(headers=headers, follow_redirects=True) as client:
        tasks = [fetch_page(client, u, sem) for u in urls]
        results = await asyncio.gather(*tasks)

    parsed = []
    for url, html in results:
        if html:
            parsed.append({"url": url, **parse(html)})
    return parsed

urls = ["https://example.com/p1", "https://example.com/p2", "..."]
data = asyncio.run(crawl(urls))

python · spiders/news.py 
import scrapy

class NewsSpider(scrapy.Spider):
    name = "news"
    start_urls = ["https://news.example.com"]

    custom_settings = {
        "DOWNLOAD_DELAY": 1.0,        # 每個請求間隔
        "CONCURRENT_REQUESTS": 8,
        "ROBOTSTXT_OBEY": True,
    }

    def parse(self, response):
        for article in response.css("article.post"):
            yield {
                "title": article.css("h2::text").get(),
                "url": article.css("a::attr(href)").get(),
                "date": article.css("time::attr(datetime)").get(),
            }

        # 自動跟蹤下一頁
        next_page = response.css("a.next::attr(href)").get()
        if next_page:
            yield response.follow(next_page, self.parse)

§25.9完整爬蟲範例

python · 完整可執行的爬蟲骨架 
# /// script
# requires-python = ">=3.14"
# dependencies = ["httpx", "beautifulsoup4", "lxml"]
# ///
import asyncio
import json
import logging
import sqlite3
import random
from pathlib import Path
import httpx
from bs4 import BeautifulSoup

logging.basicConfig(level=logging.INFO, format="%(asctime)s %(message)s")
log = logging.getLogger(__name__)

class Scraper:
    def __init__(self, db_path="out.db", concurrency=5):
        self.sem = asyncio.Semaphore(concurrency)
        self.conn = sqlite3.connect(db_path)
        self.conn.execute("""
            CREATE TABLE IF NOT EXISTS items (
                url TEXT PRIMARY KEY, title TEXT, price REAL,
                scraped_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP)
        """)

    async def fetch(self, client, url):
        async with self.sem:
            try:
                await asyncio.sleep(random.uniform(0.5, 1.5))
                async with asyncio.timeout(10):
                    r = await client.get(url)
                    r.raise_for_status()
                    return url, r.text
            except Exception as e:
                log.warning(f"失敗 {url}: {e}")
                return url, None

    def parse(self, url, html):
        soup = BeautifulSoup(html, "lxml")
        title = soup.select_one("h1.product")
        price = soup.select_one("span.price")
        return {
            "url": url,
            "title": title.get_text(strip=True) if title else None,
            "price": float(price.get_text().replace("$", "")) if price else None,
        }

    def save(self, item):
        self.conn.execute(
            "INSERT OR REPLACE INTO items (url, title, price) VALUES (?, ?, ?)",
            (item["url"], item["title"], item["price"])
        )
        self.conn.commit()

    async def run(self, urls):
        headers = {"User-Agent": "MyBot (contact@example.com)"}
        async with httpx.AsyncClient(headers=headers, follow_redirects=True) as c:
            tasks = [self.fetch(c, u) for u in urls]
            for coro in asyncio.as_completed(tasks):
                url, html = await coro
                if html:
                    item = self.parse(url, html)
                    self.save(item)
                    log.info(f"[OK] {item['title']}")

if __name__ == "__main__":
    urls = ["https://shop.example.com/item/{}".format(i) for i in range(1, 101)]
    asyncio.run(Scraper().run(urls))
