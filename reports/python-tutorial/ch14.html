檔案與路徑操作

寫 Python 就是與檔案打交道：設定檔、日誌、資料匯入匯出、使用者資料... 本章專注於現代 Python 的正確做法 —— pathlib 取代 os.path、context manager 保證資源釋放、JSON/TOML/CSV 標準庫的靈活運用。

§0章節導論

本章重點：

§14.1open 與檔案讀寫

python 
# ⭐ 永遠使用 with（context manager），保證自動關閉
with open("data.txt", encoding="utf-8") as f:
    content = f.read()              # 一次讀全部（小檔案）

# 逐行讀（大檔案首選）
with open("big.log", encoding="utf-8") as f:
    for line in f:
        process(line.rstrip())

# 讀所有行成 list
with open("data.txt", encoding="utf-8") as f:
    lines = f.readlines()          # 含每行結尾 \n

# 寫入（會覆蓋）
with open("out.txt", "w", encoding="utf-8") as f:
    f.write("hello\n")
    f.writelines(["line 1\n", "line 2\n"])

# 附加（不覆蓋）
with open("log.txt", "a", encoding="utf-8") as f:
    f.write(f"{now}: event\n")
⚠️ 永遠指定 encoding

不指定 encoding，Python 會用「locale 預設」：Windows 常是 cp950/big5，Linux/macOS 通常是 utf-8。不同平台結果不同 → 亂碼 bug！處理任何文字檔都一律寫 encoding="utf-8"。Python 3.15 起預設編碼會改為 utf-8（PEP 686），但現在要明確寫出。

§14.2開啟模式

模式意義檔案不存在既存內容
r讀（預設）FileNotFoundError保留
w寫建立清空！
a附加建立保留，從尾端寫
x建立（獨占）建立FileExistsError
r+讀寫FileNotFoundError保留
b二進位（搭配其他）——
t文字模式（預設）——
python 
open("f.txt")                 # = "rt" 文字讀取
open("f.txt", "rb")            # 二進位讀（bytes）
open("f.txt", "w")             # 文字寫入（覆蓋）
open("f.txt", "a+")            # 讀寫+附加
open("f.txt", "x", encoding="utf-8")   # 只建立新檔

§14.3pathlib — 現代路徑物件

不要再用字串拼路徑！pathlib.Path 讓路徑操作變成物件導向、跨平台、語法優雅。

python 
from pathlib import Path

# 建立路徑
p = Path("/home/alice/data/file.txt")
p = Path.home()                  # ~
p = Path.cwd()                   # 當前工作目錄

# 用 / 運算子拼接（像寫 URL 一樣直覺）
base = Path.home()
config = base / ".config" / "myapp" / "settings.toml"

# 基本屬性
p = Path("/home/alice/docs/report.pdf")
p.name          # 'report.pdf'
p.stem          # 'report'（不含副檔名）
p.suffix        # '.pdf'
p.suffixes      # ['.pdf']（例如 'a.tar.gz' → ['.tar', '.gz']）
p.parent        # Path('/home/alice/docs')
p.parents[0]     # 同上
p.parents[1]     # Path('/home/alice')
p.parts         # ('/', 'home', 'alice', 'docs', 'report.pdf')
p.root          # '/'
p.drive         # ''（Windows 上可能是 'C:'）
p.anchor        # root + drive

§14.4路徑操作大全

python 
p = Path("data/file.txt")

# 檢查
p.exists()              # 是否存在
p.is_file()             # 是檔案？
p.is_dir()              # 是目錄？
p.is_symlink()          # 是符號連結？
p.is_absolute()         # 是絕對路徑？

# 轉換
p.resolve()             # 轉絕對路徑（含解析 symlink）
p.absolute()            # 轉絕對路徑（不解析）
str(p)                  # 字串形式
p.as_posix()            # POSIX 風格（永遠用 /）
p.as_uri()              # file:// URL

# 修改路徑（不改實際檔案）
p.with_name("other.txt")        # 換檔名
p.with_suffix(".bak")          # 換副檔名
p.with_stem("new_name")         # 3.9+ 換主檔名

# 實際檔案操作
p.read_text(encoding="utf-8")   # 一次讀文字
p.write_text("content", encoding="utf-8")
p.read_bytes()
p.write_bytes(b"...")

# 目錄操作
p.mkdir()                     # 建立目錄
p.mkdir(parents=True)         # 遞迴建立（像 mkdir -p）
p.mkdir(exist_ok=True)        # 已存在不報錯
p.mkdir(parents=True, exist_ok=True)

# 刪除
p.unlink()                    # 刪檔案
p.unlink(missing_ok=True)       # 不存在也不報錯
p.rmdir()                     # 刪空目錄

# 重新命名 / 移動
p.rename(Path("new.txt"))      # 重新命名
p.replace(Path("other.txt"))    # 移動並取代（原子操作）

# 檔案資訊
stat = p.stat()
stat.st_size            # 檔案大小（bytes）
stat.st_mtime           # 修改時間（unix timestamp）

# 遞迴刪除整個資料夾要用 shutil
import shutil
shutil.rmtree("old_data")

§14.5走訪目錄

python 
p = Path("project")

# 列出目錄內容（僅一層）
for child in p.iterdir():
    print(child)

# Glob 搜尋（僅一層）
for py_file in p.glob("*.py"):
    print(py_file)

# 遞迴 glob（重要！用 ** 搭配 glob）
for py_file in p.rglob("*.py"):
    print(py_file)

# 等同於
list(p.glob("**/*.py"))

# Python 3.12+ Path.walk（類似 os.walk 但回傳 Path）
for dirpath, dirnames, filenames in p.walk():
    for name in filenames:
        print(dirpath / name)

# 組合篩選
py_files = [f for f in p.rglob("*")
            if f.is_file() and f.suffix == ".py"]

§14.6JSON 讀寫

python 
import json
from pathlib import Path

# str ↔ dict
data = json.loads('{"name": "Alice"}')
s = json.dumps(data)

# 檔案 ↔ dict
with open("config.json", encoding="utf-8") as f:
    config = json.load(f)

with open("out.json", "w", encoding="utf-8") as f:
    json.dump(data, f,
              indent=2,              # 縮排（方便閱讀）
              ensure_ascii=False,   # 讓中文直接輸出，不要跳脫
              sort_keys=True)       # key 排序

# 更簡潔：pathlib + json
config = json.loads(Path("config.json").read_text(encoding="utf-8"))
Path("out.json").write_text(
    json.dumps(data, indent=2, ensure_ascii=False),
    encoding="utf-8"
)

處理不標準 JSON

python 
# JSON 不支援的型別 → 需自訂 encoder
from datetime import datetime
from decimal import Decimal

class CustomEncoder(json.JSONEncoder):
    def default(self, obj):
        if isinstance(obj, datetime):
            return obj.isoformat()
        if isinstance(obj, Decimal):
            return float(obj)
        if isinstance(obj, set):
            return list(obj)
        return super().default(obj)

json.dumps(data, cls=CustomEncoder)

§14.7TOML / YAML / CSV

TOML（3.11+ 標準庫）

python 
# 讀取 TOML（例如 pyproject.toml）— 3.11+ 標準庫
import tomllib        # ⚠️ 只能讀，不能寫

with open("pyproject.toml", "rb") as f:     # 必須二進位模式
    config = tomllib.load(f)

print(config["project"]["name"])

# 寫入 TOML 要用第三方套件
# uv add tomli-w
import tomli_w
with open("out.toml", "wb") as f:
    tomli_w.dump(data, f)

YAML（需 pyyaml）

python 
# uv add pyyaml
import yaml

with open("config.yaml", encoding="utf-8") as f:
    config = yaml.safe_load(f)      # ⚠️ 一律用 safe_load，避免程式碼注入

with open("out.yaml", "w", encoding="utf-8") as f:
    yaml.safe_dump(data, f, allow_unicode=True)

CSV

python 
import csv

# 讀取（每列為 list）
with open("data.csv", encoding="utf-8", newline="") as f:
    reader = csv.reader(f)
    header = next(reader)
    for row in reader:
        print(row)

# 讀取為 dict（推薦）
with open("data.csv", encoding="utf-8", newline="") as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row["name"], row["age"])

# 寫入
rows = [
    {"name": "Alice", "age": 30},
    {"name": "Bob", "age": 25},
]
with open("out.csv", "w", encoding="utf-8", newline="") as f:
    writer = csv.DictWriter(f, fieldnames=["name", "age"])
    writer.writeheader()
    writer.writerows(rows)
📘 CSV 的 newline=""

永遠在開檔時加 newline="" —— 讓 csv 模組自行處理換行，避免 Windows 上出現空行。

§14.8二進位檔案與 pickle

python 
# 讀寫二進位（如圖片、音訊）
with open("image.png", "rb") as f:
    data = f.read()       # bytes

with open("copy.png", "wb") as f:
    f.write(data)

# pickle — 序列化 Python 物件（僅供信任環境使用！）
import pickle

data = {"users": [User("Alice"), User("Bob")]}
with open("state.pkl", "wb") as f:
    pickle.dump(data, f)

with open("state.pkl", "rb") as f:
    data = pickle.load(f)
⚠️ pickle 的安全性警告

絕對不要 unpickle 來路不明的資料！ pickle 檔在載入時會執行程式碼，是完整的遠端程式碼執行（RCE）漏洞。序列化資料要給別人用，選 JSON、msgpack 或 Protocol Buffers。

§14.9暫存檔案與原子寫入

python 
import tempfile

# 暫存檔（with 結束自動刪除）
with tempfile.NamedTemporaryFile(mode="w", encoding="utf-8",
                                  suffix=".json", delete=True) as tmp:
    tmp.write('{"temp": true}')
    print(tmp.name)   # 檔案路徑

# 暫存目錄（with 結束整個刪除）
with tempfile.TemporaryDirectory() as tmp_dir:
    p = Path(tmp_dir) / "work.txt"
    p.write_text("...", encoding="utf-8")
    # 處理...
# 離開 with → 整個目錄連檔案自動刪除

# 原子寫入（避免中斷時留下半寫的檔案）
import os
def atomic_write_text(path, text, encoding="utf-8"):
    p = Path(path)
    tmp = p.with_suffix(p.suffix + ".tmp")
    tmp.write_text(text, encoding=encoding)
    os.replace(tmp, p)          # 原子性取代（OS 保證）
