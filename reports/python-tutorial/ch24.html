Web 框架

2026 年，FastAPI 已是 Python Web API 的預設選擇：基於型別註解自動驗證輸入、自動生成 OpenAPI 文件、原生支援 async、效能媲美 Go。本章深入 FastAPI 的核心，順便速覽 Flask、Django，讓你能為專案選對工具。

§24.1FastAPI 入門

§24.2路徑參數與查詢參數

python 
from fastapi import FastAPI, Query, Path
from typing import Annotated

app = FastAPI()

# 路徑參數：型別會自動轉換與驗證
@app.get("/users/{user_id}")
def get_user(user_id: int):
    return {"user_id": user_id}

# /users/abc 會自動回 422，因為 abc 不是 int

# 查詢參數（function 中非路徑的參數）
@app.get("/items")
def list_items(
    skip: int = 0,
    limit: int = 10,
    q: str | None = None,          # 可選
):
    return {"skip": skip, "limit": limit, "q": q}

# 加驗證（使用 Annotated + Query）
@app.get("/search")
def search(
    q: Annotated[str, Query(min_length=3, max_length=50)],
    page: Annotated[int, Query(ge=1)] = 1,
):
    return {"q": q, "page": page}

# 路徑參數也可加驗證
@app.get("/products/{id}")
def product(id: Annotated[int, Path(gt=0)]):
    return {"id": id}

§24.3Pydantic 資料驗證（核心）

FastAPI 搭配 Pydantic v2 —— 後者底層用 Rust 實作，驗證速度極快（比 v1 快 50 倍）。

python 
from pydantic import BaseModel, Field, EmailStr
from datetime import datetime

class UserCreate(BaseModel):
    name: str = Field(min_length=2, max_length=50)
    email: EmailStr                          # 自動驗證 email 格式
    age: int = Field(ge=0, le=150)
    tags: list[str] = []
    bio: str | None = None

class User(UserCreate):
    id: int
    created_at: datetime

@app.post("/users", response_model=User, status_code=201)
async def create_user(payload: UserCreate):
    # payload 已經是完整驗證過的 UserCreate 物件
    user = User(
        id=1,
        created_at=datetime.now(),
        **payload.model_dump()
    )
    return user

進階驗證

python 
from pydantic import field_validator, model_validator
from typing import Self

class Registration(BaseModel):
    username: str
    password: str
    password_confirm: str

    @field_validator("username")
    @classmethod
    def username_alphanum(cls, v: str) -> str:
        if not v.isalnum():
            raise ValueError("只允許字母數字")
        return v

    @model_validator(mode="after")
    def passwords_match(self) -> Self:
        if self.password != self.password_confirm:
            raise ValueError("密碼不一致")
        return self

§24.4回應模型

python 
# response_model 控制「輸出」—— 可與輸入分開，避免洩漏敏感欄位
class UserInDB(BaseModel):
    id: int
    name: str
    email: str
    hashed_password: str     # 絕對不該回給客戶端！

class UserOut(BaseModel):
    id: int
    name: str
    email: str

@app.get("/users/{id}", response_model=UserOut)
async def get_user(id: int) -> UserInDB:
    return UserInDB(id=id, name="...", email="...",
                    hashed_password="secret")
    # FastAPI 會過濾 UserInDB 只保留 UserOut 有的欄位
    # hashed_password 不會回給客戶端！

# 狀態碼
@app.post("/items", status_code=201)
@app.delete("/items/{id}", status_code=204)

# 不同狀態碼不同 response（OpenAPI 文件會更完整）
@app.get("/users/{id}",
          responses={
              200: {"model": UserOut},
              404: {"description": "找不到使用者"},
          })

§24.5依賴注入（Dependency Injection）

FastAPI 最強大的特色 —— 用 Depends 把共用邏輯（驗證、DB 連線、權限檢查）抽取，自動注入。

python 
from fastapi import Depends, HTTPException

# 簡單依賴 — 共用查詢參數
def common_params(skip: int = 0, limit: int = 100):
    return {"skip": skip, "limit": limit}

@app.get("/items")
async def list_items(params: Annotated[dict, Depends(common_params)]):
    return params

# 類別作為依賴
class Pagination:
    def __init__(self, page: int = 1, per_page: int = 20):
        self.skip = (page - 1) * per_page
        self.limit = per_page

@app.get("/items")
async def items(p: Annotated[Pagination, Depends()]):
    return {"skip": p.skip, "limit": p.limit}

# 有狀態的依賴 — DB session（推薦模式）
async def get_db():
    db = SessionLocal()
    try:
        yield db             # 依賴值是 db
    finally:
        await db.close()     # request 結束自動清理

@app.get("/users")
async def list_users(db: Annotated[AsyncSession, Depends(get_db)]):
    return await db.query(User).all()

# 巢狀依賴（依賴可以依賴其他依賴）
async def get_current_user(
    token: Annotated[str, Depends(oauth2_scheme)],
    db: Annotated[AsyncSession, Depends(get_db)],
):
    user = await decode_token(token, db)
    if not user:
        raise HTTPException(401, "未授權")
    return user

@app.get("/me")
async def me(user: Annotated[User, Depends(get_current_user)]):
    return user

§24.6中介層與 CORS

python 
from fastapi import FastAPI, Request
from fastapi.middleware.cors import CORSMiddleware
from fastapi.middleware.gzip import GZipMiddleware
import time

app = FastAPI()

# CORS（跨域資源分享）— 前後端分離必設
app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://myapp.com", "http://localhost:3000"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 自動 gzip 壓縮
app.add_middleware(GZipMiddleware, minimum_size=1000)

# 自訂中介層（每個 request 前後都會跑）
@app.middleware("http")
async def log_time(request: Request, call_next):
    t0 = time.perf_counter()
    response = await call_next(request)
    response.headers["X-Process-Time"] = str(time.perf_counter() - t0)
    return response

§24.7錯誤處理

python 
from fastapi import HTTPException
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError

# 直接拋 HTTPException
@app.get("/users/{id}")
async def get_user(id: int):
    user = await find_user(id)
    if not user:
        raise HTTPException(
            status_code=404,
            detail="使用者不存在",
            headers={"X-Error-Code": "USER_NOT_FOUND"}
        )
    return user

# 自訂例外 + handler
class QuotaExceeded(Exception):
    def __init__(self, limit: int):
        self.limit = limit

@app.exception_handler(QuotaExceeded)
async def quota_handler(request, exc):
    return JSONResponse(
        status_code=429,
        content={"error": "quota_exceeded", "limit": exc.limit}
    )

# 覆寫預設驗證錯誤訊息
@app.exception_handler(RequestValidationError)
async def validation_handler(request, exc):
    return JSONResponse(
        status_code=400,
        content={"errors": exc.errors()}
    )

§24.8認證與授權（JWT + OAuth2）

python 
# uv add python-jose[cryptography] passlib[bcrypt]
from fastapi import Depends, HTTPException
from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
from jose import jwt, JWTError
from passlib.context import CryptContext
from datetime import datetime, timedelta, UTC

SECRET = "your-secret-key"      # 放環境變數！
ALGO = "HS256"

pwd_ctx = CryptContext(schemes=["bcrypt"])
oauth2 = OAuth2PasswordBearer(tokenUrl="token")

def hash_pwd(p): return pwd_ctx.hash(p)
def verify_pwd(p, h): return pwd_ctx.verify(p, h)

def create_token(data: dict, expires: timedelta = timedelta(hours=1)):
    payload = {**data, "exp": datetime.now(UTC) + expires}
    return jwt.encode(payload, SECRET, algorithm=ALGO)

@app.post("/token")
async def login(form: Annotated[OAuth2PasswordRequestForm, Depends()]):
    user = await find_user(form.username)
    if not user or not verify_pwd(form.password, user.hashed):
        raise HTTPException(401, "帳號或密碼錯誤")
    token = create_token({"sub": user.username})
    return {"access_token": token, "token_type": "bearer"}

async def get_current_user(token: Annotated[str, Depends(oauth2)]):
    try:
        payload = jwt.decode(token, SECRET, algorithms=[ALGO])
        username = payload["sub"]
    except JWTError:
        raise HTTPException(401, "token 無效")
    return await find_user(username)

@app.get("/me")
async def me(user: Annotated[User, Depends(get_current_user)]):
    return user
⚠️ 安全要點
  • SECRET 絕不寫死在程式碼 —— 用環境變數或 secret manager
  • HTTPS 是必要的 —— token 在明文 HTTP 上會被攔截
  • 密碼永遠用 bcrypt/argon2，絕不用 md5/sha256 直接雜湊
  • Token 建議短效（如 15 分鐘 access + 長效 refresh token）
  • 對敏感操作要 CSRF 防護與 rate limit

§24.9異步資料庫（SQLAlchemy 2.0）

python 
# uv add "sqlalchemy[asyncio]" asyncpg
from sqlalchemy.ext.asyncio import AsyncSession, create_async_engine, async_sessionmaker
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column
from sqlalchemy import select

DATABASE_URL = "postgresql+asyncpg://user:pwd@localhost/db"

engine = create_async_engine(DATABASE_URL, echo=False)
AsyncSessionLocal = async_sessionmaker(engine, expire_on_commit=False)

class Base(DeclarativeBase): pass

class User(Base):
    __tablename__ = "users"
    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str]
    email: Mapped[str] = mapped_column(unique=True)

async def get_db():
    async with AsyncSessionLocal() as session:
        yield session

@app.get("/users", response_model=list[UserOut])
async def list_users(db: Annotated[AsyncSession, Depends(get_db)]):
    result = await db.execute(select(User))
    return result.scalars().all()

@app.post("/users", response_model=UserOut)
async def create_user(
    payload: UserCreate,
    db: Annotated[AsyncSession, Depends(get_db)],
):
    user = User(**payload.model_dump())
    db.add(user)
    await db.commit()
    await db.refresh(user)
    return user
💡 SQLModel — 更簡潔的選擇

FastAPI 作者也寫了 SQLModel（結合 SQLAlchemy + Pydantic），可以一個 class 同時當 ORM 模型和 API schema，減少重複。適合中小型專案：uv add sqlmodel

§24.10Flask 與 Django 概覽

Flask（WSGI，輕量）

python 
# uv add flask
from flask import Flask, request, jsonify

app = Flask(__name__)

@app.route("/")
def index():
    return jsonify(hello="world")

@app.route("/items", methods=["POST"])
def create():
    data = request.get_json()
    # 沒有內建型別驗證 — 要自己做
    return jsonify(data), 201

# 啟動：flask --app main run

Flask 優點：簡單、生態大、學習曲線平緩。缺點：非 async-first，缺少自動文件、自動驗證。

Django（全套框架）

bash 
uv add django djangorestframework

django-admin startproject mysite
python manage.py startapp blog
python manage.py migrate
python manage.py createsuperuser
python manage.py runserver

Django 優點：內建 ORM + admin 後台 + 認證系統 + template 引擎，適合內容型網站。缺點：入門曲線陡、async 支援較弱（正在改進）。
