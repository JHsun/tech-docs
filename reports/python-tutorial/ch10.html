CHAPTER 10 · PART 3

特殊方法與Dunder

為什麼 Python 的 len(lst)、a + b、for x in obj、with ... 都能運作？因為這些語法背後都連結到類別的 Dunder 方法（雙底線方法，Double UNDERscore）。掌握 Dunder，你的類別就能像 Python 內建型別一樣自然使用。

Python 3.14 Data Model 資料模型 預估閱讀 25 分鐘

§0章節導論

Python 的 Data Model（資料模型）規定了各種 dunder 方法與語言運算子的對應關係。例如：

  • obj + other → obj.__add__(other)
  • len(obj) → obj.__len__()
  • str(obj) → obj.__str__()
  • for x in obj: → obj.__iter__()
  • with obj: → obj.__enter__() / obj.__exit__()

實作這些方法，讓自訂類別融入 Python 語法，這叫「Pythonic 的類別設計」。

§10.1__repr__ 與 __str__（字串表示）

python 
class Point:
    def __init__(self, x, y):
        self.x, self.y = x, y

    def __repr__(self):
        """開發者導向：精確、最好能 eval 重建"""
        return f"Point(x={self.x!r}, y={self.y!r})"

    def __str__(self):
        """使用者導向：易讀、友善"""
        return f"({self.x}, {self.y})"

p = Point(3, 4)
print(p)              # (3, 4)         ← 用 __str__
print(repr(p))        # Point(x=3, y=4) ← __repr__
print([p])            # [Point(x=3, y=4)]（容器內用 repr）
f"{p!s}"               # 強制 __str__
f"{p!r}"               # 強制 __repr__
💡 最佳實踐

至少實作 __repr__ —— 即使沒有 __str__，print() 會 fallback 用 __repr__，這比預設的 <Point object at 0x...> 有用得多。理想的 __repr__ 應讓 eval(repr(obj)) == obj。

§10.2__eq__ 與 __hash__（相等與雜湊）

python 
class Point:
    def __init__(self, x, y):
        self.x, self.y = x, y

    def __eq__(self, other):
        if not isinstance(other, Point):
            return NotImplemented
        return (self.x, self.y) == (other.x, other.y)

    def __hash__(self):
        return hash((self.x, self.y))

p1 = Point(1, 2)
p2 = Point(1, 2)
print(p1 == p2)   # True（值相等）
print(p1 is p2)   # False（不同物件）

# 有 __hash__ 才能當 set 元素或 dict key
points = {p1, p2, Point(3, 4)}
print(points)         # 只有 2 個元素（p1, p2 被視為相同）
⚠️ __eq__ 與 __hash__ 必須一致

若兩物件 a == b，則必須 hash(a) == hash(b)，否則 set/dict 會出錯。
定義 __eq__ 會讓 __hash__ 自動變成 None（物件變成 unhashable），這通常不是你想要的 —— 所以記得同時定義 __hash__。可變物件通常不應該有 __hash__。

§10.3排序相關的 Dunder

python 
from functools import total_ordering

@total_ordering       # 只要實作 __eq__ 和 __lt__，自動補完 > <= >=
class Priority:
    def __init__(self, level):
        self.level = level

    def __eq__(self, other):
        return self.level == other.level

    def __lt__(self, other):
        return self.level < other.level

Priority(1) < Priority(2)       # True
sorted([Priority(3), Priority(1)])  # 自動排序

# 完整六個比較運算子
# __lt__ <    __le__ <=
# __gt__ >    __ge__ >=
# __eq__ ==   __ne__ !=

§10.4算術運算子重載

python 
class Vector:
    def __init__(self, x, y):
        self.x, self.y = x, y

    def __add__(self, other):           # v1 + v2
        return Vector(self.x + other.x, self.y + other.y)

    def __sub__(self, other):           # v1 - v2
        return Vector(self.x - other.x, self.y - other.y)

    def __mul__(self, scalar):          # v * 3
        return Vector(self.x * scalar, self.y * scalar)

    def __rmul__(self, scalar):         # 3 * v（反向）
        return self * scalar

    def __neg__(self):                  # -v
        return Vector(-self.x, -self.y)

    def __abs__(self):                  # abs(v) — 長度
        return (self.x**2 + self.y**2) ** 0.5

    def __repr__(self):
        return f"Vector({self.x}, {self.y})"

v1 = Vector(1, 2)
v2 = Vector(3, 4)
print(v1 + v2)      # Vector(4, 6)
print(v1 * 3)       # Vector(3, 6)
print(3 * v1)       # Vector(3, 6)  ← 靠 __rmul__
print(-v1)          # Vector(-1, -2)
print(abs(v2))       # 5.0

-__sub____rsub____isub__
*__mul____rmul____imul__
/__truediv____rtruediv____itruediv__
//__floordiv____rfloordiv____ifloordiv__

§10.5容器協議

python 
class ShoppingCart:
    def __init__(self):
        self._items = {}

    def __len__(self):            # len(cart)
        return sum(self._items.values())

    def __getitem__(self, key):     # cart["apple"]
        return self._items[key]

    def __setitem__(self, key, v):  # cart["apple"] = 3
        self._items[key] = v

    def __delitem__(self, key):     # del cart["apple"]
        del self._items[key]

    def __contains__(self, key):    # "apple" in cart
        return key in self._items

    def __bool__(self):            # bool(cart) / if cart:
        return bool(self._items)

cart = ShoppingCart()
cart["apple"] = 3
cart["banana"] = 2
print(len(cart))         # 5
print(cart["apple"])     # 3
print("banana" in cart) # True
del cart["apple"]
if cart:
    print("還有東西")

§10.6迭代協議

python 
class Range:
    """自製 range"""
    def __init__(self, start, stop):
        self.start = start
        self.stop = stop

    def __iter__(self):             # for x in obj
        n = self.start
        while n < self.stop:
            yield n
            n += 1

for i in Range(0, 5):     # 0 1 2 3 4
    print(i)

# 也會支援 list()、sum()、in 等
list(Range(0, 3))     # [0, 1, 2]
sum(Range(0, 10))    # 45

§10.7__call__ 可呼叫物件

實作 __call__ 後，實例可以像函式一樣被呼叫。適合「帶狀態的函式」。

python 
class Multiplier:
    def __init__(self, factor):
        self.factor = factor

    def __call__(self, x):
        return x * self.factor

double = Multiplier(2)
double(5)          # 10，像函式！
double(10)         # 20

# 檢查是否 callable
callable(double)    # True
# 也可應用於裝飾器類別
class CountCalls:
    def __init__(self, fn):
        self.fn = fn
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        return self.fn(*args, **kwargs)

@CountCalls
def hello(): print("hi")

hello(); hello(); hello()
print(hello.count)   # 3

§10.8Context Manager：with 語法

實作 __enter__ 與 __exit__，讓物件能配合 with 使用 —— 保證資源正確釋放。

python 
class Timer:
    def __enter__(self):
        import time
        self.t0 = time.perf_counter()
        return self           # as 後面綁定的值

    def __exit__(self, exc_type, exc_val, tb):
        import time
        self.elapsed = time.perf_counter() - self.t0
        print(f"耗時 {self.elapsed:.3f}s")
        # return True 可「吞掉」例外；回傳 None/False 讓例外傳出
        return False

with Timer() as t:
    total = sum(range(10**6))
print(t.elapsed)

用 contextlib 簡化寫法

python 
from contextlib import contextmanager

@contextmanager
def timer():
    import time
    t0 = time.perf_counter()
    try:
        yield                  # 上半段 = __enter__，下半段 = __exit__
    finally:
        print(f"{time.perf_counter()-t0:.3f}s")

with timer():
    sum(range(10**6))

§10.9屬性存取的 Dunder

python 
class Config:
    def __init__(self, data):
        self._data = data

    def __getattr__(self, name):
        """當一般屬性查不到時才呼叫"""
        if name in self._data:
            return self._data[name]
        raise AttributeError(name)

    def __setattr__(self, name, value):
        """每次設值都呼叫"""
        if name.startswith("_"):
            super().__setattr__(name, value)   # 內部屬性正常處理
        else:
            self._data[name] = value

    def __delattr__(self, name):
        del self._data[name]

c = Config({"theme": "dark"})
print(c.theme)          # 'dark'（用屬性語法存取 dict）
c.lang = "zh"           # 透過 __setattr__ 寫入 dict
print(c._data)          # {'theme': 'dark', 'lang': 'zh'}
⚠️ __setattr__ 遞迴陷阱

在 __setattr__ 內如果寫 self.attr = value 會無限遞迴！必須用 super().__setattr__(name, value) 或直接操作 self.__dict__[name]。
