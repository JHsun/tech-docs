📊 B. Metric Aggregations（16 種）

從 doc 集合計算單一數值：sum、avg、cardinality、percentiles 等。

Aggregation
範例
說明
avg
{ "avg": { "field": "price" } }
平均值
sum
{ "sum": { "field": "amount" } }
加總
min / max
{ "max": { "field": "price" } }
最小 / 最大
value_count
{ "value_count": { "field": "user_id" } }
欄位有值的 doc 數（含重複）
cardinality
{
  "cardinality": {
    "field": "user_id",
    "precision_threshold": 100
  }
}
不重複計數（HyperLogLog++ 近似）。precision 1-40000，越高越準越耗 RAM
stats
{ "stats": { "field": "price" } }
一次回傳 count / min / max / sum / avg
extended_stats
{ "extended_stats": { "field": "price" } }
stats + 變異數、標準差、平方和
percentiles
{
  "percentiles": {
    "field":    "latency_ms",
    "percents": [50, 95, 99, 99.9]
  }
}
p50 / p95 / p99 / p99.9（近似 TDigest）
percentile_ranks
{
  "percentile_ranks": {
    "field":  "latency_ms",
    "values": [100, 500, 1000]
  }
}
反查：100ms 以下的比例是多少？
median_absolute_deviation
{ "median_absolute_deviation": { "field": "price" } }
MAD：對異常值不敏感的離散度
weighted_avg
{
  "weighted_avg": {
    "value":  { "field": "score" },
    "weight": { "field": "weight" }
  }
}
加權平均
top_hits
{
  "top_hits": {
    "size": 3,
    "sort": [ { "@timestamp": "desc" } ],
    "_source": [ "title", "ts" ]
  }
}
每 bucket 取 N 筆原始 doc（含內容）
top_metrics
{
  "top_metrics": {
    "metrics": [ { "field": "price" }, { "field": "stock" } ],
    "sort":    { "@timestamp": "desc" }
  }
}
取每 bucket 最新的指標值（比 top_hits 快）
matrix_stats
{
  "matrix_stats": {
    "fields": [ "price", "stock" ]
  }
}
多欄位協方差、相關係數
scripted_metric
{
  "scripted_metric": {
    "init_script":    "state.transactions = []",
    "map_script":     "state.transactions.add(doc.amount.value)",
    "combine_script": "double total = 0; for (t in state.transactions) total += t; return total",
    "reduce_script":  "double total = 0; for (s in states) total += s; return total"
  }
}
完全自訂的聚合（4 階段腳本）
geo_bounds / geo_centroid
{ "geo_centroid": { "field": "location" } }
地理範圍 / 中心點
💡 percentiles vs percentile_ranks
  • percentiles：給百分位 (95)，回傳對應值（"p95 = 250ms"）
  • percentile_ranks：給值 (200ms)，回傳百分比（"< 200ms 佔 87%"）

🪣 C. Bucket Aggregations（16 種）

把 doc 分組到 bucket（類似 SQL GROUP BY）。每 bucket 可再嵌 sub-agg。

Aggregation
範例
說明
terms
{
  "terms": {
    "field":  "category",
    "size":   10,
    "order":  { "_count": "desc" }
  }
}
依欄位值分組（最常用）。預設 size=10；高 cardinality 改 composite
terms with sub-agg ordering
{
  "terms": {
    "field": "category",
    "order": { "total_sales": "desc" }
  },
  "aggs": {
    "total_sales": { "sum": { "field": "amount" } }
  }
}
依 sub-agg 排序（找最賺錢分類）
multi_terms
{
  "multi_terms": {
    "terms": [
      { "field": "country" },
      { "field": "category" }
    ],
    "size": 10
  }
}
多欄位組合分組（取代 sub-terms 在某些場景）
significant_terms
{
  "significant_terms": {
    "field": "tag",
    "size":  5
  }
}
找「異常頻繁」的 term（statistically significant）
rare_terms
{ "rare_terms": { "field": "user_id", "max_doc_count": 1 } }
找罕見值（出現次數 ≤ N）
date_histogram
{
  "date_histogram": {
    "field":             "@timestamp",
    "calendar_interval": "1d",
    "format":            "yyyy-MM-dd",
    "time_zone":         "Asia/Taipei",
    "min_doc_count":     0,
    "extended_bounds": {
      "min": "2026-01-01",
      "max": "2026-12-31"
    }
  }
}
時間直方圖。calendar_interval (1m/1h/1d/1w/1M/1y) 或 fixed_interval (90m/2h)
histogram
{
  "histogram": {
    "field":    "price",
    "interval": 100
  }
}
數值直方圖（價格段）
auto_date_histogram
{
  "auto_date_histogram": {
    "field":   "@timestamp",
    "buckets": 24
  }
}
自動選 interval（給 24 個 bucket）
range
{
  "range": {
    "field": "price",
    "ranges": [
      { "key": "cheap",   "to":   100 },
      { "key": "medium",  "from": 100, "to": 500 },
      { "key": "premium", "from": 500 }
    ]
  }
}
自訂範圍分組
date_range
{
  "date_range": {
    "field": "@timestamp",
    "ranges": [
      { "key": "this_week", "from": "now-7d/d" },
      { "key": "last_week", "from": "now-14d/d", "to": "now-7d/d" }
    ]
  }
}
日期範圍
filters
{
  "filters": {
    "filters": {
      "errors":   { "term":  { "level": "ERROR" } },
      "warnings": { "term":  { "level": "WARN" } },
      "info":     { "term":  { "level": "INFO" } }
    }
  }
}
依多個 filter 分組（每 filter 一 bucket）
composite ⭐
{
  "composite": {
    "size":  1000,
    "sources": [
      { "day":     { "date_histogram": {
          "field": "@timestamp", "calendar_interval": "1d" } } },
      { "service": { "terms": { "field": "service.name" } } }
    ],
    "after":   { "day": 1735689600000, "service": "api" }
  }
}
高 cardinality 必用！可分頁（用 after_key）。不能依 metric 排序
nested
{
  "nested": {
    "path": "comments"
  },
  "aggs": {
    "by_author": { "terms": { "field": "comments.author" } }
  }
}
對 nested 欄位 agg
geo_distance
{
  "geo_distance": {
    "field":  "location",
    "origin": { "lat": 25.03, "lon": 121.5 },
    "unit":   "km",
    "ranges": [
      { "to": 1 },
      { "from": 1, "to": 5 },
      { "from": 5 }
    ]
  }
}
依距離分組（如 1km / 1-5km / 5km+）
geohash_grid / geotile_grid
{
  "geotile_grid": {
    "field":     "location",
    "precision": 8,
    "size":      10000
  }
}
把地圖切格 → 每格 doc 數（heatmap 用）
sampler / diversified_sampler
{
  "sampler": {
    "shard_size": 200
  }
}
取樣 N 筆做後續 agg（significant_terms 常用）
random_sampler
{
  "random_sampler": {
    "probability": 0.01
  }
}
隨機取樣 1%，海量資料分析超快
⚠️ terms vs composite 何時用
  • terms：cardinality < 10000，希望依 metric 排序
  • composite：cardinality 高（百萬+），需要分頁，不能依 metric 排序（只能依 source key 排）
  • 常見錯誤：用 terms + size: 100000 → 記憶體爆炸 → 改用 composite

🔁 D. Pipeline Aggregations（11 種）

對其他 agg 的輸出再運算（不直接讀 doc）。用 buckets_path 指向來源。

Aggregation
範例
說明
moving_avg / moving_fn
"smoothed": {
  "moving_fn": {
    "buckets_path": "the_sum",
    "window":       7,
    "script":       "MovingFunctions.unweightedAvg(values)"
  }
}
移動平均（7 bucket window）
cumulative_sum
"cumulative_sales": {
  "cumulative_sum": {
    "buckets_path": "the_sum"
  }
}
累積總和（每 bucket = 自己 + 之前所有）
derivative
"diff": {
  "derivative": {
    "buckets_path": "the_sum"
  }
}
對每 bucket 算與前 bucket 的差異
bucket_script
"profit_pct": {
  "bucket_script": {
    "buckets_path": {
      "rev":  "revenue",
      "cost": "cost"
    },
    "script": "(params.rev - params.cost) / params.rev * 100"
  }
}
在 bucket 內用 script 計算新 metric
bucket_selector
"only_high": {
  "bucket_selector": {
    "buckets_path": { "total": "total_sales" },
    "script":       "params.total > 10000"
  }
}
過濾 bucket（HAVING）。低於閾值的 bucket 不回傳
bucket_sort
"top_5_buckets": {
  "bucket_sort": {
    "sort": [ { "total_sales": "desc" } ],
    "size": 5
  }
}
對 bucket 排序 + 分頁（用於 composite 後的 metric 排序）
min_bucket / max_bucket
"best_day": {
  "max_bucket": {
    "buckets_path": "daily>total_sales"
  }
}
Sibling agg：找出全期最高的那個 bucket
avg_bucket / sum_bucket
"weekly_avg": {
  "avg_bucket": {
    "buckets_path": "daily>total_sales"
  }
}
對所有 bucket 的 metric 取平均 / 加總
stats_bucket / extended_stats_bucket
"sales_stats": {
  "stats_bucket": {
    "buckets_path": "daily>total_sales"
  }
}
對 bucket metric 算 stats
percentiles_bucket
"daily_percentiles": {
  "percentiles_bucket": {
    "buckets_path": "daily>total_sales",
    "percents":     [25, 50, 75, 95]
  }
}
對 bucket metric 算百分位
serial_diff
"yoy": {
  "serial_diff": {
    "buckets_path": "monthly>total",
    "lag":          12
  }
}
與 N bucket 前比較（月對月、年對年）
💡 buckets_path 語法
  • my_metric：同層的 metric agg
  • my_bucket>my_metric：bucket 內的 metric（> 進入子層）
  • my_bucket>my_metric.avg：multi-value metric 取特定 value（.）
  • my_bucket._count：bucket 文件數
  • my_bucket._bucket_count：sub-bucket 數量

🧪 E. 經典 Recipes

📈 Recipe 1：Top N users by spending（30 天）

GET /orders/_search
{
  "size": 0,
  "query": {
    "range": { "@timestamp": { "gte": "now-30d" } }
  },
  "aggs": {
    "top_users": {
      "terms": {
        "field": "user_id",
        "size":  10,
        "order": { "total_spent": "desc" }
      },
      "aggs": {
        "total_spent": { "sum":      { "field": "amount" } },
        "order_count": { "value_count": { "field": "_id" } },
        "avg_order":   { "avg":      { "field": "amount" } }
      }
    }
  }
}

⏱️ Recipe 2：每 service 每天 p95 latency（含 alert filter）

GET /metrics-*/_search
{
  "size": 0,
  "query": {
    "range": { "@timestamp": { "gte": "now-7d" } }
  },
  "aggs": {
    "by_service": {
      "terms": { "field": "service.name", "size": 20 },
      "aggs": {
        "daily": {
          "date_histogram": {
            "field":             "@timestamp",
            "calendar_interval": "1d",
            "min_doc_count":     0
          },
          "aggs": {
            "p95_latency": {
              "percentiles": {
                "field":    "duration_ms",
                "percents": [95]
              }
            },
            "error_rate": {
              "filter": { "term": { "status": "error" } }
            }
          }
        },
        "alert_high_p95": {
          "bucket_selector": {
            "buckets_path": { "p95": "daily>p95_latency[95]" },
            "script":       "params.p95 > 1000"
          }
        }
      }
    }
  }
}

結合 terms + date_histogram + percentiles + bucket_selector 過濾出高 latency 的服務。

🎯 Recipe 3：轉換漏斗（Funnel）

GET /events/_search
{
  "size": 0,
  "aggs": {
    "funnel": {
      "filters": {
        "filters": {
          "1_visit":    { "term": { "event": "page_view" } },
          "2_signup":   { "term": { "event": "signup" } },
          "3_activate": { "term": { "event": "activated" } },
          "4_purchase": { "term": { "event": "purchase" } }
        }
      }
    }
  }
}

filters agg 一次回多個 filter 的計數，計算每階段的轉換率。

📊 Recipe 4：Log rate 變化偵測（per minute）

GET /logs-*/_search
{
  "size": 0,
  "query": {
    "range": { "@timestamp": { "gte": "now-1h" } }
  },
  "aggs": {
    "errors_per_minute": {
      "date_histogram": {
        "field":         "@timestamp",
        "fixed_interval": "1m"
      },
      "aggs": {
        "error_count": {
          "filter": { "term": { "level": "ERROR" } }
        },
        "rate_change": {
          "derivative": { "buckets_path": "error_count._count" }
        },
        "moving": {
          "moving_fn": {
            "buckets_path": "error_count._count",
            "window":       5,
            "script":       "MovingFunctions.unweightedAvg(values)"
          }
        }
      }
    }
  }
}

每分鐘 error 數 + 變化率（derivative）+ 5 分鐘移動平均。可用在 anomaly detection 前的預處理。

⚡ F. 效能調校 Cheat List

技巧
做法
效果
set size: 0
不要 hits，只要 agg
省去 fetch + score；啟用 query cache
filter 在 query 層
先縮小 doc 集合再 agg
百萬 → 十萬：差 10× 時間
composite for high cardinality
取代 terms + 高 size
常數記憶體用量；可分頁
eager_global_ordinals
在 mapping 設 "eager_global_ordinals": true
常 agg 的 keyword 預先建 ordinal；refresh 慢一點換 agg 快
execution_hint
"execution_hint": "map" 對小 cardinality
global_ordinals (default) / map / global_ordinals_low_cardinality
調 cardinality precision
precision_threshold: 100 已足夠多場景
預設 3000；越高越耗 RAM
random_sampler🆕 9+
"probability": 0.01
海量資料近似分析快 100×
shard_size
terms agg "shard_size": size * 1.5
提升精度（避免邊緣 doc 漏算）
min_doc_count
"min_doc_count": 0 包含空 bucket（時序圖必用）
預設 1；圖表斷線即 0
避免 scripted_metric in hot path
scripted 比內建 agg 慢數倍
能用內建就不用 script

(function () {
  function init() {
    document.querySelectorAll('main pre').forEach(function (pre) {
      if (pre.parentElement.classList.contains('code-wrap')) return;
      const wrap = document.createElement('div');
      wrap.className = 'code-wrap';
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(pre);
      const lang = pre.querySelector('code')?.className.match(/language-(\w+)/)?.[1];
      if (lang) {
        const tag = document.createElement('span');
        tag.className = 'code-lang';
        tag.textContent = lang;
        wrap.appendChild(tag);
      }
      const btn = document.createElement('button');
      btn.className = 'copy-btn';
      btn.type = 'button';
      btn.textContent = '📋 複製';
      btn.addEventListener('click', function () {
        const text = pre.innerText;
        if (navigator.clipboard?.writeText) {
          navigator.clipboard.writeText(text).then(success).catch(fallback);
        } else {
          fallback();
        }
        function success() {
          btn.textContent = '✓ 已複製';
          btn.classList.add('copied');
          setTimeout(function () { btn.textContent = '📋 複製'; btn.classList.remove('copied'); }, 1800);
        }
        function fallback() {
          const ta = document.createElement('textarea');
          ta.value = text;
          document.body.appendChild(ta);
          ta.select();
          try { document.execCommand('copy'); success(); } catch (_) {}
          document.body.removeChild(ta);
        }
      });
      wrap.appendChild(btn);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

// Mermaid initialization with theme sync
(function () {
  function getMermaidTheme() {
    return document.documentElement.dataset.theme === 'dark' ? 'dark' : 'default';
  }
  function getMermaidConfig() {
    const isDark = document.documentElement.dataset.theme === 'dark';
    const cs = getComputedStyle(document.documentElement);
    return {
      startOnLoad: false,
      theme: isDark ? 'dark' : 'default',
      themeVariables: {
        fontFamily: cs.getPropertyValue('--font-sans').trim() || 'DM Sans, sans-serif',
        fontSize: '14px',
        primaryColor: isDark ? '#1e3a4a' : '#e0f2fe',
        primaryTextColor: isDark ? '#e6f2f8' : '#0c4a5e',
        primaryBorderColor: isDark ? '#5fb3d4' : '#0891b2',
        lineColor: isDark ? '#7aa3b8' : '#475569',
        secondaryColor: isDark ? '#2a3f4f' : '#f1f5f9',
        tertiaryColor: isDark ? '#1a2530' : '#fafafa',
        background: isDark ? '#1e2a35' : '#ffffff',
        mainBkg: isDark ? '#243441' : '#ecfeff',
        secondBkg: isDark ? '#1a2530' : '#f8fafc',
        nodeBorder: isDark ? '#5fb3d4' : '#0891b2',
        clusterBkg: isDark ? '#1a2530' : '#f1f5f9',
        clusterBorder: isDark ? '#3a5260' : '#cbd5e1',
        defaultLinkColor: isDark ? '#7aa3b8' : '#475569',
        titleColor: isDark ? '#e6f2f8' : '#0f172a',
        edgeLabelBackground: isDark ? '#243441' : '#ffffff',
        actorBkg: isDark ? '#243441' : '#ecfeff',
        actorBorder: isDark ? '#5fb3d4' : '#0891b2',
        actorTextColor: isDark ? '#e6f2f8' : '#0c4a5e',
        actorLineColor: isDark ? '#5fb3d4' : '#0891b2',
        signalColor: isDark ? '#7aa3b8' : '#334155',
        signalTextColor: isDark ? '#e6f2f8' : '#0f172a',
        labelBoxBkgColor: isDark ? '#1a2530' : '#fef3c7',
        labelBoxBorderColor: isDark ? '#a89060' : '#d97706',
        labelTextColor: isDark ? '#e6f2f8' : '#0f172a',
        loopTextColor: isDark ? '#e6f2f8' : '#0f172a',
        noteBkgColor: isDark ? '#3a4f3a' : '#fef3c7',
        noteBorderColor: isDark ? '#86c598' : '#d97706',
        noteTextColor: isDark ? '#e6f2f8' : '#0f172a',
      },
      flowchart: { useMaxWidth: true, htmlLabels: true, curve: 'basis' },
      sequence: { useMaxWidth: true, showSequenceNumbers: false, mirrorActors: true, bottomMarginAdj: 10 },
    };
  }

  let renderCounter = 0;
  async function renderAll() {
    if (typeof mermaid === 'undefined') return;
    renderCounter++;
    const myCounter = renderCounter;
    mermaid.initialize(getMermaidConfig());
    const blocks = document.querySelectorAll('.mermaid');
    for (const el of blocks) {
      // restore source if already rendered
      if (el.dataset.source) el.innerHTML = el.dataset.source;
      else el.dataset.source = el.innerHTML;
      el.removeAttribute('data-processed');
    }
    if (myCounter !== renderCounter) return;
    try {
      await mermaid.run({ querySelector: '.mermaid' });
    } catch (e) { console.warn('Mermaid render error', e); }
  }

  function init() {
    renderAll();
    window.addEventListener('jhsun:theme-changed', renderAll);
  }

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();
