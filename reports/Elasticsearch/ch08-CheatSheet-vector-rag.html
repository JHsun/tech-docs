🧬 A. dense_vector Mapping（量化選擇）

情境
命令 / Code
說明
基本 dense_vector
PUT my-vectors
{
  "mappings": {
    "properties": {
      "embedding": {
        "type":       "dense_vector",
        "dims":       384,
        "similarity": "cosine"
      }
    }
  }
}
dims 必須與 model 一致；similarity: cosine / dot_product / l2_norm
BBQ HNSW（9.x 預設）🆕 9+
"embedding": {
  "type":       "dense_vector",
  "dims":       768,
  "index_options": {
    "type":           "bbq_hnsw",
    "m":              16,
    "ef_construction": 100
  }
}
9.x 預設。32× 壓縮 + 自動 oversampling re-rank
DiskBBQ（極省 RAM）🆕 9.2+
"embedding": {
  "type":       "dense_vector",
  "dims":       1024,
  "index_options": {
    "type": "bbq_disk"
  }
}
不需把所有向量裝 RAM。可省 95% 記憶體。recall ≤ 95%
BBQ Flat（小資料集）
"index_options": { "type": "bbq_flat" }
Brute force 搜尋；不建 HNSW graph；適合 < 100K 向量
int8 / int4 量化
"index_options": { "type": "int8_hnsw" }
// 或
"index_options": { "type": "int4_hnsw" }
4× / 8× 壓縮；準度比 BBQ 高一點點；BBQ 通常更划算
HNSW（不量化，傳統）
"index_options": {
  "type":            "hnsw",
  "m":               16,
  "ef_construction": 100
}
原始精度 float32；耗 RAM。9.x 不建議，改用 BBQ
byte vector（int8）
"embedding": {
  "type":         "dense_vector",
  "element_type": "byte",
  "dims":         128
}
寫入時就是 int8；用於低成本本地 model
bfloat16🆕 9.3+
"embedding": {
  "type":         "dense_vector",
  "element_type": "bfloat16",
  "dims":         1024
}
9.3+ 新型別。比 float32 省一半，比 int8 準
不索引（純儲存）
"embedding": {
  "type":  "dense_vector",
  "dims":  768,
  "index": false
}
不建 ANN graph；只能用 script_score 做精確 KNN（適合小資料）
💡 量化策略 cheat sheet（9.x 推薦）
  • 多數場景：bbq_hnsw（9.x 預設）— 32× 壓縮、高準度
  • RAM 緊張：bbq_disk（9.2+）— 不需 RAM 裝向量
  • 小資料 < 100K：bbq_flat（brute force）
  • 極限 RAM：bbq_disk + on_disk_rescore（9.3+）
  • 需要絕對最高準度：hnsw（不量化，但耗 RAM）

🔄 C. Retriever（9.0+）— RRF / Linear / Rerank

9.x 推出 retriever 抽象，把 query / KNN / RRF / rerank 統一在一個結構下。

情境
命令 / Code
說明
Standard retriever
"retriever": {
  "standard": {
    "query": { "match": { "content": "elastic" } }
  }
}
與 query 等價的新語法
KNN retriever
"retriever": {
  "knn": {
    "field":          "embedding",
    "query_vector":   [...],
    "k":              50,
    "num_candidates": 200
  }
}
KNN 搜尋包成 retriever
RRF（Hybrid 推薦）🆕 9+
"retriever": {
  "rrf": {
    "retrievers": [
      {
        "standard": {
          "query": { "multi_match": {
              "query": "running shoes",
              "fields": [ "name", "description" ]
          }}
        }
      },
      {
        "knn": {
          "field":          "embedding",
          "query_vector":   [...],
          "k":              50,
          "num_candidates": 200
        }
      }
    ],
    "rank_window_size": 50,
    "rank_constant":     60
  }
}
BM25 + 向量。rank_constant=60 是業界標準
Linear retriever
"retriever": {
  "linear": {
    "retrievers": [
      { "retriever": { "standard": {...} }, "weight": 0.7 },
      { "retriever": { "knn":      {...} }, "weight": 0.3 }
    ],
    "rank_window_size": 100,
    "normalizer":        "minmax"
  }
}
加權混合（需自行調整權重）
text_similarity_reranker
"retriever": {
  "text_similarity_reranker": {
    "retriever": { "rrf": { ... } },
    "field":           "content",
    "inference_id":    "cohere-rerank",
    "inference_text":  "running shoes",
    "rank_window_size": 50
  }
}
對 retriever 結果做 rerank（cross-encoder model）
Rule retriever（pinned）
"retriever": {
  "rule": {
    "ruleset_ids":  ["promotion"],
    "match_criteria": { "query_string": "winter coat" },
    "retriever":   { "standard": {...} }
  }
}
套用 query rules（釘特定結果在前）
💡 RRF 為何用 60：rank_constant=60 是 RRF 論文與業界（Elasticsearch / OpenSearch / Qdrant）的標準。 Hybrid (BM25 + 向量) recall@10 從 78% 提升到 91%（無需調權重）。

🤖 I. 完整 RAG Pipeline（Python）

結合 hybrid retrieval + rerank + LLM via Inference API（secret 集中管理）：

from elasticsearch import Elasticsearch

es = Elasticsearch("https://es:9200", api_key="...", ca_certs="ca.crt")

def rag_pipeline(question: str) -> str:
    # 1. Hybrid retrieval + rerank
    res = es.search(index="knowledge-base", body={
        "retriever": {
            "text_similarity_reranker": {
                "retriever": {
                    "rrf": {
                        "retrievers": [
                            {
                                "standard": {
                                    "query": { "match": { "content": question } }
                                }
                            },
                            {
                                "standard": {
                                    "query": {
                                        "semantic": {
                                            "field": "content",
                                            "query": question
                                        }
                                    }
                                }
                            }
                        ],
                        "rank_window_size": 50,
                        "rank_constant":    60
                    }
                },
                "field":           "content",
                "inference_id":    "cohere-rerank",
                "inference_text":  question,
                "rank_window_size": 5
            }
        },
        "size": 5,
        "_source": ["title", "content", "metadata.url"]
    })

    # 2. 組 context（含 source citation）
    context_parts = []
    for hit in res["hits"]["hits"]:
        src = hit["_source"]
        context_parts.append(
            f"[{src['title']}] ({src['metadata']['url']})\n"
            f"{src['content']}"
        )
    context = "\n\n---\n\n".join(context_parts)

    # 3. 透過 ES Inference API 呼叫 LLM（secret 集中）
    res = es.options(request_timeout=120).perform_request(
        "POST", "/_inference/completion/claude",
        body={
            "input": (
                f"Use ONLY the following context to answer the question. "
                f"Cite sources by [title]. If answer not in context, say so.\n\n"
                f"Context:\n{context}\n\n"
                f"Question: {question}\n\n"
                f"Answer:"
            )
        }
    )
    return res.body["completion"][0]["result"]

📊 J. BBQ vs DiskBBQ vs HNSW 比較

方案
RAM 需求
適用 / 限制
HNSW (float32)
最高（全部向量在 RAM）
最高準度；9.x 不再推薦（除非有特殊需求）
BBQ HNSW 🆕 9+
1/32 of HNSW（量化 + 向量在 RAM）
9.x 預設；準度高（oversample + rerank）；常規場景
DiskBBQ 🆕 9.2+
~ 100MB 即可
向量在 disk；RAM 緊張時首選；recall ≤ 95%
int8_hnsw
1/4 of HNSW
中等壓縮；準度比 BBQ 略高
BBQ Flat
最低（無 graph）
小資料 < 100K 用；brute force 搜

⚡ K. RAG / Vector 最佳實踐

類別
建議
理由 / 數值
Chunking
200-500 token / chunk，10-20% overlap
太小：context 不足；太大：注意力稀釋
Strategy
FAQ → sentence；技術文件 → recursive；code → function
配合內容結構
Retrieval
永遠 hybrid（BM25 + 向量），不要純向量
純向量對 keyword（型號 / SKU）弱
RRF window
rank_window_size = top_k × 5-10
給融合足夠的候選
Rerank
大量文件加 cross-encoder rerank（Cohere / Elastic Rerank）
準度提升 10-20%；成本可接受
top_k 給 LLM
3-5 chunks；不要超過 10
太多反而稀釋注意力
Citation
prompt 要求 LLM 標註用了哪個 chunk
減少 hallucination；可追溯
Permission filter
retrieval 時加 ACL filter；用 DLS API key
避免使用者看到不該看的資料
Cost
用 ELSER 取代外部 embedding；BBQ 量化
省 95% embedding API 費用 + 95% 儲存
Cache
同 query 的 LLM 回應快取
省 token 費用
Eval
RAGAS / TruLens；recall@K + faithfulness
避免 hallucination 上線

     blocks
(function () {
  function init() {
    document.querySelectorAll('main pre').forEach(function (pre) {
      if (pre.parentElement.classList.contains('code-wrap')) return;
      const wrap = document.createElement('div');
      wrap.className = 'code-wrap';
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(pre);
      const lang = pre.querySelector('code')?.className.match(/language-(\w+)/)?.[1];
      if (lang) {
        const tag = document.createElement('span');
        tag.className = 'code-lang';
        tag.textContent = lang;
        wrap.appendChild(tag);
      }
      const btn = document.createElement('button');
      btn.className = 'copy-btn';
      btn.type = 'button';
      btn.textContent = '📋 複製';
      btn.addEventListener('click', function () {
        const text = pre.innerText;
        if (navigator.clipboard?.writeText) {
          navigator.clipboard.writeText(text).then(success).catch(fallback);
        } else {
          fallback();
        }
        function success() {
          btn.textContent = '✓ 已複製';
          btn.classList.add('copied');
          setTimeout(function () { btn.textContent = '📋 複製'; btn.classList.remove('copied'); }, 1800);
        }
        function fallback() {
          const ta = document.createElement('textarea');
          ta.value = text;
          document.body.appendChild(ta);
          ta.select();
          try { document.execCommand('copy'); success(); } catch (_) {}
          document.body.removeChild(ta);
        }
      });
      wrap.appendChild(btn);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

// Auto-build TOC from h2/h3 in 
    
(function () {
  function init() {
    const toc = document.querySelector('.toc ul');
    if (!toc) return;
    const headings = document.querySelectorAll('main h2, main h3');
    headings.forEach(function (h) {
      if (!h.id) {
        h.id = h.textContent.trim().toLowerCase()
          .replace(/[^\w\u4e00-\u9fff\s-]/g, '')
          .replace(/\s+/g, '-');
      }
      const li = document.createElement('li');
      const a = document.createElement('a');
      a.href = '#' + h.id;
      a.textContent = h.textContent;
      a.className = 'toc-' + h.tagName.toLowerCase();
      li.appendChild(a);
      toc.appendChild(li);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

  let renderCounter = 0;
  async function renderAll() {
    if (typeof mermaid === 'undefined') return;
    renderCounter++;
    const myCounter = renderCounter;
    mermaid.initialize(getMermaidConfig());
    const blocks = document.querySelectorAll('.mermaid');
    for (const el of blocks) {
      // restore source if already rendered
      if (el.dataset.source) el.innerHTML = el.dataset.source;
      else el.dataset.source = el.innerHTML;
      el.removeAttribute('data-processed');
    }
    if (myCounter !== renderCounter) return;
    try {
      await mermaid.run({ querySelector: '.mermaid' });
    } catch (e) { console.warn('Mermaid render error', e); }
  }

  function init() {
    renderAll();
    window.addEventListener('jhsun:theme-changed', renderAll);
  }

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();
