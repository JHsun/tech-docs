⚙️ A. Ingest Pipeline 操作

情境
命令 / Code
說明
建立 pipeline
PUT /_ingest/pipeline/my-pipeline
{
  "description": "Parse and normalize logs",
  "processors": [
    { "lowercase": { "field": "level" } },
    { "set":       { "field": "ingest_time", "value": "{{_ingest.timestamp}}" } }
  ],
  "on_failure": [
    { "set": { "field": "ingest.failure", "value": "{{ _ingest.on_failure_message }}" } }

🔨 B. Processors 速查（30+ 個）

Processor
範例
說明
set
{ "set": { "field": "env", "value": "production" } }
設欄位值（可用 mustache template）
set with template
{ "set": {
  "field": "full_path",
  "value": "{{host.name}}:{{file.path}}"
} }
從其他欄位組合
remove
{ "remove": { "field": [ "tmp_*", "agent.id" ] } }
刪除欄位（支援陣列）
rename
{ "rename": { "field": "src_ip", "target_field": "source.ip" } }
改名
convert
{ "convert": {
  "field":      "status",
  "type":       "integer",
  "ignore_missing": true
} }
型別轉換（integer / long / float / double / string / boolean / ip）
lowercase / uppercase
{ "lowercase": { "field": "user.email" } }
大小寫轉換
trim
{ "trim": { "field": "message" } }
去前後空白
split
{ "split": {
  "field":     "tags_str",
  "separator": ",",
  "target_field": "tags"
} }
字串切成陣列
join
{ "join": {
  "field":     "tags",
  "separator": ",",
  "target_field": "tags_str"
} }
陣列合成字串
gsub
{ "gsub": {
  "field":       "message",
  "pattern":     "password=\\w+",
  "replacement": "password=***"
} }
regex 取代（遮蔽 PII）
grok
{ "grok": {
  "field":    "message",
  "patterns": [ "%{IP:client.ip} %{WORD:method} %{URIPATHPARAM:url} %{NUMBER:status:int}" ]
} }
regex 解析（同 ES|QL GROK）
dissect
{ "dissect": {
  "field":   "message",
  "pattern": "%{ts} %{+ts} %{level} %{service} %{msg}"
} }
delimiter 解析（比 grok 快）
kv
{ "kv": {
  "field":         "params",
  "field_split":   "&",
  "value_split":   "=",
  "target_field":  "query"
} }
解析 k1=v1&k2=v2 格式
json
{ "json": {
  "field":         "raw_json",
  "target_field":  "data"
} }
解析 JSON 字串到欄位結構
date
{ "date": {
  "field":     "timestamp_str",
  "target_field": "@timestamp",
  "formats":   [ "yyyy-MM-dd HH:mm:ss", "ISO8601" ],
  "timezone":  "Asia/Taipei"
} }
字串轉 ISO datetime
date_index_name
{ "date_index_name": {
  "field":             "@timestamp",
  "index_name_prefix": "logs-",
  "date_rounding":     "d"
} }
依時間戳改寫 _index（按天分 index）
geoip
{ "geoip": {
  "field":         "client.ip",
  "target_field":  "geo",
  "properties":    [ "city_name", "country_iso_code", "location" ]
} }
IP → 地理位置（內建 MaxMind GeoLite2）
user_agent
{ "user_agent": {
  "field":         "ua_string",
  "target_field":  "user_agent"
} }
解析 UA 字串
uri_parts
{ "uri_parts": {
  "field":         "request_url",
  "target_field":  "url"
} }
解析 URL 各部分
fingerprint
{ "fingerprint": {
  "fields":      [ "user.email", "device.id" ],
  "target_field": "_id",
  "method":      "SHA-256"
} }
產生確定性 ID（用於 idempotent insert）
hash
{ "hash": {
  "fields":      [ "ssn" ],
  "target_field": "ssn_hash",
  "salt":        "my-salt"
} }
產生 hash（PII 去識別化）
script
{ "script": {
  "source": """
    if (ctx.duration_ms != null) {
      ctx.duration_sec = ctx.duration_ms / 1000.0;
    }
  """
} }
Painless 腳本（最強但慢）
foreach
{ "foreach": {
  "field":       "items",
  "processor":   {
    "lowercase": { "field": "_ingest._value.name" }
  }
} }
對陣列每元素套 processor
pipeline
{ "pipeline": { "name": "shared-pipeline" } }
呼叫另一 pipeline（重用）
enrich
{ "enrich": {
  "policy_name":   "lang_policy",
  "field":         "lang_id",
  "target_field":  "language"
} }
從 enrich index 加欄位（同 ES|QL ENRICH）
inference🆕 9+
{ "inference": {
  "model_id":      "elser-default",
  "input_output": [
    {
      "input_field":  "content",
      "output_field": "content_embedding"
    }
  ]
} }
在 ingest 時產 embedding（半 RAG）
community_id
{ "community_id": {} }
Network flow community ID（SIEM 用）
registered_domain
{ "registered_domain": {
  "field":         "host_name",
  "target_field":  "domain"
} }
取 eTLD+1（如 sub.example.co.uk → example.co.uk）
network_direction
{ "network_direction": {
  "internal_networks": [ "private" ]
} }
判斷 inbound / outbound / internal
append
{ "append": {
  "field": "tags",
  "value": [ "production" ]
} }
陣列追加值
drop
{ "drop": {
  "if": "ctx.level == 'DEBUG'"
} }
丟掉整 doc（如過濾 DEBUG log）
fail
{ "fail": {
  "if":      "ctx.user == null",
  "message": "user field is required"
} }
主動失敗（驗證）

🛡️ C. 錯誤處理（條件 / fallback）

機制
範例
說明
processor 條件
{ "lowercase": {
  "field": "email",
  "if":    "ctx.email != null"
} }
if 用 Painless；只有 true 才執行
processor 級 fallback
{ "convert": {
  "field": "count",
  "type":  "integer",
  "on_failure": [
    { "set": { "field": "count_error", "value": true } }
  ]
} }
單一 processor 失敗的 fallback
pipeline 級 fallback
"on_failure": [
  { "set": { "field": "ingest.failure_msg", "value": "{{ _ingest.on_failure_message }}" } },
  { "set": { "field": "ingest.failed_processor", "value": "{{ _ingest.on_failure_processor_type }}" } }
]
整 pipeline 任何 processor 失敗時的處理
ignore_failure
{ "convert": {
  "field": "count",
  "type":  "integer",
  "ignore_failure": true
} }
失敗時靜默跳過
ignore_missing
{ "lowercase": {
  "field":          "email",
  "ignore_missing": true
} }
欄位不存在時不報錯

C2. 配合 Failure Store（9.1+）

對 data stream 啟用 failure store（見 CS 03 §H）後，pipeline 失敗的 doc 自動進入 ::failures，可後續修復。

📦 D. Filebeat（log shipper）

情境
命令 / Code
說明
filestream input（9.x 推薦）
# filebeat.yml
filebeat.inputs:
  - type: filestream
    id:   nginx-access
    paths:
      - /var/log/nginx/access.log
    parsers:
      - container: ~
    processors:
      - add_host_metadata: ~
      - add_kubernetes_metadata: ~
9.x 棄用 log input，新 deployment 用 filestream
輸出到 ES
output.elasticsearch:
  hosts:    [ "https://es:9200" ]
  api_key:  "${ES_API_KEY}"
  pipeline: "filebeat-9.3-default"
  ssl.certificate_authorities: [ "/etc/ssl/ca.crt" ]
建議用 API key（見 CS 09）；指定 pipeline
輸出到 Logstash
output.logstash:
  hosts: [ "logstash:5044" ]
  ssl.certificate_authorities: [ "/etc/ssl/ca.crt" ]
走 Logstash 統一處理層
啟用 module（如 nginx）
# 啟用
filebeat modules enable nginx

# 設定
# /etc/filebeat/modules.d/nginx.yml
- module: nginx
  access:
    enabled: true
    var.paths: [ "/var/log/nginx/access.log*" ]
Module 自帶 ingest pipeline + dashboard
setup（一次）
# 載 template + dashboard + pipelines
filebeat setup -e \
  -E "output.elasticsearch.hosts=[\"https://es:9200\"]" \
  -E "output.elasticsearch.api_key=ID:KEY"
第一次跑要做
啟動
sudo systemctl enable --now filebeat
sudo systemctl status filebeat
sudo journalctl -u filebeat -f
系統服務 + 看日誌
除錯
# 前景跑 + 詳細 log
filebeat -e -d "*" -c filebeat.yml
-d "*" 啟用所有 debug
⚠️ 9.x 重要變更
  • old log input 已移除：必須改用 filestream
  • module 直接連 ES → 自動載 ingest pipeline
  • module 連 Logstash → 需手動 filebeat setup --pipelines

🔄 G. Logstash（複雜資料管道）

情境
命令 / Code
說明
基本 pipeline.conf
# /etc/logstash/conf.d/main.conf
input {
  beats { port => 5044 }
  syslog { port => 514 }
}

filter {
  if "nginx" in [tags] {
    grok {
      match => { "message" => "%{COMBINEDAPACHELOG}" }
    }
    date {
      match => [ "timestamp", "dd/MMM/yyyy:HH:mm:ss Z" ]
    }
    useragent {
      source => "agent"
      target => "user_agent"
    }
    geoip {
      source => "clientip"
      target => "geo"
    }
  }
  mutate {
    remove_field => [ "host", "agent" ]
  }
}

output {
  elasticsearch {
    hosts    => [ "https://es:9200" ]
    api_key  => "${ES_API_KEY}"
    index    => "logs-%{[@metadata][beat]}-%{+YYYY.MM.dd}"
    pipeline => "%{[@metadata][pipeline]}"
    cacert   => "/etc/ssl/ca.crt"
  }
}
input → filter → output 三段
常用 filter
# grok 解析
grok { match => { "message" => "%{IP:ip} %{WORD:method}" } }

# dissect 比 grok 快
dissect { mapping => { "message" => "%{ts} %{level} %{msg}" } }

# 修改欄位
mutate {
  rename => { "old_field" => "new_field" }
  remove_field => [ "tmp_*" ]
  convert => { "status" => "integer" }
  uppercase => [ "level" ]
  gsub => [ "message", "password=\\w+", "password=***" ]
}

# 條件
if [status] >= 500 {
  mutate { add_tag => [ "server_error" ] }
}

# 時間
date {
  match => [ "ts", "ISO8601", "yyyy-MM-dd HH:mm:ss" ]
  target => "@timestamp"
  timezone => "Asia/Taipei"
}

# 自訂 ruby
ruby {
  code => "event.set(\"name_lower\", event.get(\"name\").downcase)"
}
grok / dissect / mutate / date / ruby 為核心
多 pipeline
# /etc/logstash/pipelines.yml
- pipeline.id:   nginx-pipeline
  path.config:   "/etc/logstash/conf.d/nginx.conf"
  pipeline.workers: 4

- pipeline.id:   syslog-pipeline
  path.config:   "/etc/logstash/conf.d/syslog.conf"
  pipeline.workers: 2
一個 process 跑多 pipeline；資源獨立
persistent queue（防丟資料）
# /etc/logstash/logstash.yml
queue.type:        persisted
queue.max_bytes:   8gb
queue.checkpoint.writes: 1024
本地 disk queue；ES 短暫不可用時不丟訊息
Dead Letter Queue
# logstash.yml
dead_letter_queue.enable: true

# 之後可從 DLQ 讀
input {
  dead_letter_queue {
    path        => "/var/lib/logstash/dead_letter_queue"
    pipeline_id => "nginx-pipeline"
  }
}
失敗的 event 進 DLQ 不丟失；可後續修正後重送
除錯
# 設定檢查（不啟動）
bin/logstash -f config/main.conf --config.test_and_exit

# Auto-reload
bin/logstash -f config/main.conf --config.reload.automatic

# stdout debug output
output { stdout { codec => rubydebug } }
改 conf 後測試再上 prod
💡 何時用 Logstash？
  • 多輸入 / 多輸出：Beats + Kafka + Syslog + DB → ES + S3 + Kafka
  • 複雜 transform：> 30 個 step，需要 ruby 邏輯
  • buffering / DLQ：persistent queue 防 ES 不可用時丟資料
  • 非 Elastic 目的地：寫到外部系統
否則優先用 Ingest Pipeline（在 ES 內，零外部依賴）。

🤔 H. 工具選擇對照

工具
特性
適用
Filebeat (9.x)
單機 log shipper；極輕量；module 自帶 pipeline
🟢 簡單 log → ES；K8s DaemonSet；< 100MB RAM
Metricbeat
單機 metric 採集；module 化
🟢 系統 / 服務 metrics；Stack monitoring
Packetbeat / Auditbeat / Heartbeat
網路封包 / Linux audit / 主機監測
🟢 SIEM / Observability 場景
Elastic Agent
統一 agent，整合 Beats + Endpoint + APM；Fleet 集中管理
🟢 規模化部署；GUI 集中管理；推薦新 deployment
Logstash
中央資料管道；多 input / 多 output / 重 transform
🟡 複雜 transform / 多目的地路由 / 600+ plugins
Ingest pipeline
在 ES 內處理；無外部依賴；快
🟢 取代 Logstash 簡單場景；< 30 processors

(function () {
  function init() {
    document.querySelectorAll('main pre').forEach(function (pre) {
      if (pre.parentElement.classList.contains('code-wrap')) return;
      const wrap = document.createElement('div');
      wrap.className = 'code-wrap';
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(pre);
      const lang = pre.querySelector('code')?.className.match(/language-(\w+)/)?.[1];
      if (lang) {
        const tag = document.createElement('span');
        tag.className = 'code-lang';
        tag.textContent = lang;
        wrap.appendChild(tag);
      }
      const btn = document.createElement('button');
      btn.className = 'copy-btn';
      btn.type = 'button';
      btn.textContent = '📋 複製';
      btn.addEventListener('click', function () {
        const text = pre.innerText;
        if (navigator.clipboard?.writeText) {
          navigator.clipboard.writeText(text).then(success).catch(fallback);
        } else {
          fallback();
        }
        function success() {
          btn.textContent = '✓ 已複製';
          btn.classList.add('copied');
          setTimeout(function () { btn.textContent = '📋 複製'; btn.classList.remove('copied'); }, 1800);
        }
        function fallback() {
          const ta = document.createElement('textarea');
          ta.value = text;
          document.body.appendChild(ta);
          ta.select();
          try { document.execCommand('copy'); success(); } catch (_) {}
          document.body.removeChild(ta);
        }
      });
      wrap.appendChild(btn);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

// Auto-build TOC from h2/h3 in 
    
(function () {
  function init() {
    const toc = document.querySelector('.toc ul');
    if (!toc) return;
    const headings = document.querySelectorAll('main h2, main h3');
    headings.forEach(function (h) {
      if (!h.id) {
        h.id = h.textContent.trim().toLowerCase()
          .replace(/[^\w\u4e00-\u9fff\s-]/g, '')
          .replace(/\s+/g, '-');
      }
      const li = document.createElement('li');
      const a = document.createElement('a');
      a.href = '#' + h.id;
      a.textContent = h.textContent;
      a.className = 'toc-' + h.tagName.toLowerCase();
      li.appendChild(a);
      toc.appendChild(li);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

  let renderCounter = 0;
  async function renderAll() {
    if (typeof mermaid === 'undefined') return;
    renderCounter++;
    const myCounter = renderCounter;
    mermaid.initialize(getMermaidConfig());
    const blocks = document.querySelectorAll('.mermaid');
    for (const el of blocks) {
      // restore source if already rendered
      if (el.dataset.source) el.innerHTML = el.dataset.source;
      else el.dataset.source = el.innerHTML;
      el.removeAttribute('data-processed');

    if (myCounter !== renderCounter) return;
    try {
      await mermaid.run({ querySelector: '.mermaid' });

  }
