💡 Query Context vs Filter Context
  • Query context：計算 score（match 多好）。耗 CPU、不 cache。在 query、must、should
  • Filter context：是 / 否（match 與否）。會 cache，超快。在 filter、must_not、aggs
規則：能用 filter 就用 filter。只有需要相關度排序時才用 must/should。

📝 B. Match Family（全文 / analyzed）

對 text 欄位用；查詢時會被 analyzer 處理（小寫、stem、停用詞...）。

情境
命令 / Code
說明
match 單欄位全文
{ "match": { "title": "elasticsearch tutorial" } }
會 analyze；OR 連接（任一字 match 就算）
match 含 operator
{
  "match": {
    "title": {
      "query":    "elasticsearch tutorial",
      "operator": "and",
      "fuzziness": "AUTO"
    }
  }
}
and：所有字必須 match；fuzziness AUTO：自動容錯
multi_match 多欄位
{
  "multi_match": {
    "query":  "elasticsearch",
    "fields": [ "title^3", "content", "tags^2" ],
    "type":   "best_fields"
  }
}
^3 加權；type: best_fields / most_fields / cross_fields / phrase / phrase_prefix / bool_prefix
match_phrase 完整片語
{
  "match_phrase": {
    "title": {
      "query": "elasticsearch tutorial",
      "slop":  2
    }
  }
}
字必須相鄰且順序相同；slop 容忍距離
match_phrase_prefix 自動完成
{ "match_phrase_prefix": { "title": "elas" } }
前綴 + 片語（自動完成用）
simple_query_string
{
  "simple_query_string": {
    "query":  "+elastic -outdated",
    "fields": [ "title", "content" ]
  }
}
使用者輸入安全的查詢字串：+ 必須、- 排除、"..." 片語
query_string（進階 Lucene）
{
  "query_string": {
    "query":           "title:elastic AND status:published",
    "default_operator": "AND"
  }
}
不要直接接受 user input（語法錯會報錯）
match_all / match_none
{ "match_all":  {} }
{ "match_none": {} }
全部 / 全不

🎯 C. Term Family（精確 / 不 analyzed）

對 keyword、數字、boolean、date 欄位用；不會 analyze。

情境
命令 / Code
說明
term 精確比對
{ "term": { "status": "published" } }
不會 analyze！對 keyword、數字、boolean 用
terms 多值 OR
{ "terms": { "tags": [ "search", "nosql", "java" ] } }
任一 match 即命中
terms with lookup
{
  "terms": {
    "user_id": {
      "index": "users",
      "id":    "alice",
      "path":  "groups"
    }
  }
}
從另一 doc 取陣列當 terms（避免 hard-coded）
range 範圍
{
  "range": {
    "price": { "gte": 100, "lt": 500 }
  }
}
gte / gt / lte / lt；可用於數字、日期
range 日期 + date math
{
  "range": {
    "@timestamp": {
      "gte": "now-1h/h",
      "lte": "now/h"
    }
  }
}
now-1h/h：1 小時前並按 hour 對齊
exists 欄位存在
{ "exists": { "field": "user.email" } }
配合 must_not 找不存在的：{ "bool": { "must_not": [...] } }
prefix
{ "prefix": { "user_id": "alice" } }
前綴匹配；對 keyword 欄位有效
wildcard
{ "wildcard": { "user_id": "ali*" } }
* 多字、? 單字。慎用，效能差
regexp
{
  "regexp": {
    "user_id": {
      "value":     "a[lr]ice.*",
      "flags":     "ALL",
      "case_insensitive": true
    }
  }
}
正規表達式；非常昂貴
fuzzy 容錯
{
  "fuzzy": {
    "user.first": {
      "value":     "alis",
      "fuzziness": "AUTO"
    }
  }
}
Damerau-Levenshtein 距離；AUTO 依字長自動
ids
{ "ids": { "values": [ "1", "2", "3" ] } }
依 _id 找
⚠️ 常見錯誤：對 text 欄位用 term 找不到資料！ 原因：text 寫入時被切詞（如 "Hello World" → ["hello", "world"]），term 找 "Hello World" 對不上。 解法：對 text 用 match；對 keyword 用 term，或用 title.keyword multi-field（CS 02）。

📋 E. Bool 完整範例（產品搜尋）

典型電商搜尋場景：用戶輸入「laptop」，過濾在庫且 ≤ 2000 元，蘋果或 ≥4.5 星優先，排除配件。

GET /products/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "multi_match": {
            "query":  "laptop",
            "fields": [ "name^2", "description" ]
          }
        }
      ],
      "filter": [
        { "term":  { "in_stock": true } },
        { "range": { "price": { "lte": 2000 } } }
      ],
      "should": [
        { "term":  { "brand": "Apple" } },
        { "range": { "rating": { "gte": 4.5 } } }
      ],
      "minimum_should_match": 1,
      "must_not": [
        { "term": { "category": "accessories" } }
      ]
    }
  }
}

依距離近到遠

📃 G. 分頁（淺 / 深 / PIT）

情境
命令 / Code
說明
淺分頁（< 10000）
"from": 100,
"size": 20
from + size 預設上限 10000
調 max_result_window
PUT /my-index/_settings
{
  "index.max_result_window": 50000
}
不建議大幅調高（吃 heap）
search_after（深分頁）
GET /my-index/_search
{
  "size": 100,
  "sort": [
    { "@timestamp": "asc" },
    { "_id":         "asc" }
  ],
  "search_after": [ "2026-05-01T00:00:00Z", "abc-123" ]
}
上次最後一筆的 sort 值；無上限；需穩定 sort tie-breaker (_id)
PIT（穩定快照）🆕 9+
# 1. 建 PIT
POST /my-index/_pit?keep_alive=5m

# 2. 用 PIT search
GET /_search
{
  "size": 100,
  "pit": {
    "id":         "<pit_id>",
    "keep_alive": "5m"
  },
  "sort": [ { "@timestamp": "asc" }, { "_id": "asc" } ],
  "search_after": [ ... ]
}

# 3. 完成後關 PIT
DELETE /_pit
{ "id": "<pit_id>" }
Point-In-Time：穩定資料快照；分頁期間資料不變動。取代 scroll API
⚠️ 何時用什麼
  • ≤ 10000 筆：from + size
  • > 10000 筆但即時資料 OK：search_after
  • 大量資料且需要穩定快照：PIT + search_after（推薦）
  • scroll API 已棄用，新 code 用 PIT

🔮 I. 進階查詢（nested / function_score / script）

情境
命令 / Code
說明
nested query
{
  "nested": {
    "path": "comments",
    "query": {
      "bool": {
        "must": [
          { "match": { "comments.author": "alice" } },
          { "match": { "comments.text":   "elasticsearch" } }
        ]
      }
    },
    "score_mode":  "avg",
    "inner_hits":  {}
  }
}
對 nested 欄位查詢；inner_hits 回傳命中的子文件
has_parent / has_child
{
  "has_child": {
    "type": "answer",
    "query": { "match": { "body": "elasticsearch" } }
  }
}
Join 欄位的關聯查詢
function_score
{
  "function_score": {
    "query": { "match": { "title": "elastic" } },
    "functions": [
      {
        "filter": { "term": { "featured": true } },
        "weight": 10
      },
      {
        "gauss": {
          "@timestamp": {
            "origin": "now",
            "scale":  "7d",
            "decay":  0.5
          }
        }
      }
    ],
    "score_mode":   "sum",
    "boost_mode":   "multiply"
  }
}
客製化 score 公式：filter+weight、gauss/linear/exp decay、script_score
script_score
{
  "script_score": {
    "query": { "match_all": {} },
    "script": {
      "source": "Math.log(2 + doc['likes'].value) * 0.5"
    }
  }
}
用 Painless 腳本計分
boosting
{
  "boosting": {
    "positive":      { "match": { "title": "elastic" } },
    "negative":      { "match": { "tag":   "deprecated" } },
    "negative_boost": 0.5
  }
}
降低（不排除）特定條件的 score
constant_score
{
  "constant_score": {
    "filter": { "term": { "tag": "tech" } },
    "boost": 1.2
  }
}
把 filter 變成固定 score 的 query
dis_max
{
  "dis_max": {
    "queries": [
      { "match": { "title":   "search" } },
      { "match": { "content": "search" } }
    ],
    "tie_breaker": 0.3
  }
}
取多 sub-query 中最高 score（不累加）
more_like_this
{
  "more_like_this": {
    "fields":  [ "title", "content" ],
    "like":    [ { "_index": "blog", "_id": "1" } ],
    "min_term_freq":  1,
    "max_query_terms": 12
  }
}
找相似文件
geo_bounding_box
{
  "geo_bounding_box": {
    "location": {
      "top_left":     { "lat": 25.5, "lon": 121.0 },
      "bottom_right": { "lat": 24.5, "lon": 122.0 }
    }
  }
}
矩形範圍內的點
geo_distance
{
  "geo_distance": {
    "distance": "10km",
    "location": { "lat": 25.03, "lon": 121.5 }
  }
}

📅 J. Date Math 速查

運算式
意義
技巧
now
當前時間（毫秒）
基準錨點
now-1h
1 小時前
單位：y / M / w / d / h / H / m / s
now-1d/d
昨天 00:00:00（按天對齊）
/d rounding 提升 cache hit rate
now-1y+1M/d
一年前一個月後，按天對齊
可組合運算
2026-05-01||+1d
指定日期 +1 天
|| 後是 anchor 日期
now/M
本月 1 日 00:00
常用月份對齊
💡 Cache 提升技巧： 用 now/d 取代 now 讓查詢可被 cache（同一天內值不變）。 例："gte": "now-1d/d" 在當天內始終是同個值，filter cache 命中率高。

(function () {
  function init() {
    document.querySelectorAll('main pre').forEach(function (pre) {
      if (pre.parentElement.classList.contains('code-wrap')) return;
      const wrap = document.createElement('div');
      wrap.className = 'code-wrap';
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(pre);
      const lang = pre.querySelector('code')?.className.match(/language-(\w+)/)?.[1];
      if (lang) {
        const tag = document.createElement('span');
        tag.className = 'code-lang';
        tag.textContent = lang;
        wrap.appendChild(tag);
      }
      const btn = document.createElement('button');
      btn.className = 'copy-btn';
      btn.type = 'button';
      btn.textContent = '📋 複製';
      btn.addEventListener('click', function () {
        const text = pre.innerText;
        if (navigator.clipboard?.writeText) {
          navigator.clipboard.writeText(text).then(success).catch(fallback);
        } else {
          fallback();
        }
        function success() {
          btn.textContent = '✓ 已複製';
          btn.classList.add('copied');
          setTimeout(function () { btn.textContent = '📋 複製'; btn.classList.remove('copied'); }, 1800);
        }
        function fallback() {
          const ta = document.createElement('textarea');
          ta.value = text;
          document.body.appendChild(ta);
          ta.select();
          try { document.execCommand('copy'); success(); } catch (_) {}
          document.body.removeChild(ta);
        }
      });
      wrap.appendChild(btn);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

// Auto-build TOC from h2/h3 in 
    
(function () {
  function init() {
    const toc = document.querySelector('.toc ul');
    if (!toc) return;
    const headings = document.querySelectorAll('main h2, main h3');
    headings.forEach(function (h) {
      if (!h.id) {
        h.id = h.textContent.trim().toLowerCase()
          .replace(/[^\w\u4e00-\u9fff\s-]/g, '')
          .replace(/\s+/g, '-');
      }
      const li = document.createElement('li');
      const a = document.createElement('a');
      a.href = '#' + h.id;
      a.textContent = h.textContent;
      a.className = 'toc-' + h.tagName.toLowerCase();
      li.appendChild(a);
      toc.appendChild(li);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

  let renderCounter = 0;
  async function renderAll() {
    if (typeof mermaid === 'undefined') return;
    renderCounter++;
    const myCounter = renderCounter;
    mermaid.initialize(getMermaidConfig());
    const blocks = document.querySelectorAll('.mermaid');
    for (const el of blocks) {
      // restore source if already rendered
      if (el.dataset.source) el.innerHTML = el.dataset.source;
      else el.dataset.source = el.innerHTML;
      el.removeAttribute('data-processed');
    }
    if (myCounter !== renderCounter) return;
    try {
      await mermaid.run({ querySelector: '.mermaid' });
    } catch (e) { console.warn('Mermaid render error', e); }
  }

  function init() {
    renderAll();
    window.addEventListener('jhsun:theme-changed', renderAll);
  }

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
