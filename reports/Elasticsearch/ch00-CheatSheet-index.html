📖 使用建議

💡 建議用法
  • Bookmark 索引頁（本頁）：每次從這裡的「我想做什麼？」進入
  • 列印 A4 貼牆：每張速查卡都優化過列印 layout，可單張列印
  • 桌面用三欄高密度：寬螢幕 1100px+ 自動三欄；窄螢幕自動單欄
  • 看 deprecation 警告：升級前必跑 GET /_migration/deprecations
  • 不確定指令安全度：看每行旁的 🟢🟡🔴 圓點
  • 需要完整解釋：每張卡末尾都有「回正書 CH XX」連結

⚠️ 速查表使用規則

  1. 所有指令以 Kibana Dev Tools Console 格式呈現。實際用 curl 時：
    • GET /_cluster/health → curl -X GET 'https://es:9200/_cluster/health'
    • 需加認證：-u user:pass 或 -H "Authorization: ApiKey ..."
    • 啟用 TLS：--cacert ca.crt 或 -k（不驗證）
  2. 標記 危險的指令：執行前先 snapshot（CS 10）
  3. 修改 cluster setting 用 persistent，重啟才會保留；transient 重啟即消失
  4. 9.x 已棄用 transient setting，盡量用 persistent
  5. Production 操作前先在 staging 試——尤其涉及 reroute、reindex 大量資料
  6. 本速查表不取代正書：第一次學新主題時請看正書章節

(function () {
  function init() {
    document.querySelectorAll('main pre').forEach(function (pre) {
      if (pre.parentElement.classList.contains('code-wrap')) return;
      const wrap = document.createElement('div');
      wrap.className = 'code-wrap';
      pre.parentNode.insertBefore(wrap, pre);
      wrap.appendChild(pre);
      const lang = pre.querySelector('code')?.className.match(/language-(\w+)/)?.[1];
      if (lang) {
        const tag = document.createElement('span');
        tag.className = 'code-lang';
        tag.textContent = lang;
        wrap.appendChild(tag);
      }
      const btn = document.createElement('button');
      btn.className = 'copy-btn';
      btn.type = 'button';
      btn.textContent = '📋 複製';
      btn.addEventListener('click', function () {
        const text = pre.innerText;
        if (navigator.clipboard?.writeText) {
          navigator.clipboard.writeText(text).then(success).catch(fallback);
        } else {
          fallback();
        }
        function success() {
          btn.textContent = '✓ 已複製';
          btn.classList.add('copied');
          setTimeout(function () { btn.textContent = '📋 複製'; btn.classList.remove('copied'); }, 1800);
        }
        function fallback() {
          const ta = document.createElement('textarea');
          ta.value = text;
          document.body.appendChild(ta);
          ta.select();
          try { document.execCommand('copy'); success(); } catch (_) {}
          document.body.removeChild(ta);
        }
      });
      wrap.appendChild(btn);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

// Auto-build TOC from h2/h3 in 
    
(function () {
  function init() {
    const toc = document.querySelector('.toc ul');
    if (!toc) return;
    const headings = document.querySelectorAll('main h2, main h3');
    headings.forEach(function (h) {
      if (!h.id) {
        h.id = h.textContent.trim().toLowerCase()
          .replace(/[^\w\u4e00-\u9fff\s-]/g, '')
          .replace(/\s+/g, '-');
      }
      const li = document.createElement('li');
      const a = document.createElement('a');
      a.href = '#' + h.id;
      a.textContent = h.textContent;
      a.className = 'toc-' + h.tagName.toLowerCase();
      li.appendChild(a);
      toc.appendChild(li);
    });
  }
  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', init);
  else init();
})();

  let renderCounter = 0;
  async function renderAll() {
    if (typeof mermaid === 'undefined') return;
    renderCounter++;
    const myCounter = renderCounter;
    mermaid.initialize(getMermaidConfig());
    const blocks = document.querySelectorAll('.mermaid');
    for (const el of blocks) {
      // restore source if already rendered
      if (el.dataset.source) el.innerHTML = el.dataset.source;
      else el.dataset.source = el.innerHTML;
      el.removeAttribute('data-processed');

    if (myCounter !== renderCounter) return;
    try {
      await mermaid.run({ querySelector: '.mermaid' });
    } catch (e) { console.warn('Mermaid render error', e); }
  }
