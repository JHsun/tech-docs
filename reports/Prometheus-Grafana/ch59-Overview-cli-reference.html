§ 2. promtool

Prometheus 內建 CLI，跟 prometheus binary 一起發行。

2.1 設定檔驗證

# 驗證 prometheus.yml
promtool check config prometheus.yml

# 驗證 rule files
promtool check rules rules/*.yaml

# Output（成功）
Checking prometheus.yml
  SUCCESS: 0 rule files found
  SUCCESS: prometheus.yml is valid prometheus config file syntax

# Output（失敗）
Checking rules/api.yaml
  FAILED:
  unknown function "rate2" in expression: rate2(http_requests_total[5m])

2.2 PromQL Query CLI

# Instant query
promtool query instant http://localhost:9090 'up'

# Output（columnar）
{instance="localhost:9100", job="node"} => 1 @[1714579200.123]

# Range query
promtool query range \
  --start=2026-05-02T10:00:00Z \
  --end=2026-05-02T11:00:00Z \
  --step=30s \
  http://localhost:9090 \
  'rate(http_requests_total[5m])'

# 找 series
promtool query series http://localhost:9090 \
  --match='{job="api",status=~"5.."}'

# 找 label
promtool query labels http://localhost:9090
promtool query labels http://localhost:9090 service

2.3 Unit Test for rules

把 alerting / recording rule 寫測試（很重要！）：

# tests/api-rules.yaml
rule_files:
  - ../rules/api.yaml

evaluation_interval: 1m

tests:
  - interval: 1m
    input_series:
      - series: 'http_requests_total{service="api",status="500"}'
        values: '0+10x10'   # 從 0 開始，每分鐘增 10，共 10 點
      - series: 'http_requests_total{service="api",status="200"}'
        values: '0+90x10'

    alert_rule_test:
      - eval_time: 5m
        alertname: HighErrorRate
        exp_alerts:
          - exp_labels:
              service: api
              severity: warning
            exp_annotations:
              summary: "High error rate on api"

    promql_expr_test:
      - expr: 'sum by (service) (rate(http_requests_total{status=~"5.."}[5m]))'
        eval_time: 5m
        exp_samples:
          - labels: '{service="api"}'
            value: 0.1666666

# 跑測試
promtool test rules tests/api-rules.yaml

# Output
Unit Testing:  tests/api-rules.yaml
  SUCCESS

2.4 TSDB 工具

# 分析 TSDB
promtool tsdb analyze /var/lib/prometheus/data

# Output
Block ID: 01ETT28D6B8948J87NZXY8VYD9
  Duration: 2h
  Series: 1234567
  Label names: 234
  ...
  Highest cardinality labels:
    job: 1234567 series in 50 values
    instance: 1234567 series in 100 values
    ...

# Dump TSDB block
promtool tsdb dump /var/lib/prometheus/data
promtool tsdb list /var/lib/prometheus/data

2.5 設定 lint

# 檢查 metric / label 命名是否符合慣例
promtool check metrics < metrics.txt

# Output
http_requests_total non-counter metrics should not have "_total" suffix

§ 3. amtool

3.1 連線設定

# 設環境變數（避免每次 -alertmanager.url）
export ALERTMANAGER_URL=http://localhost:9093

# 或 config 檔（~/.config/amtool/config.yml）
alertmanager.url: "http://localhost:9093"
output: extended    # extended | simple | json

3.2 列出 alerts

# 列出當前 firing alert
amtool alert query

# 用 matcher filter
amtool alert query alertname=HighErrorRate severity=critical

# JSON 輸出
amtool alert query --output=json | jq '.[].labels'

3.3 推送 alert

amtool alert add \
  alertname=DiskFull \
  severity=critical \
  instance=server1 \
  --start=2026-05-02T10:00:00Z \
  --end=2026-05-02T11:00:00Z \
  --annotation summary='Disk full on server1'

3.4 Silence 管理

# 列出 active silence
amtool silence query

# 加 silence（matcher 用 = 或 =~）
amtool silence add \
  alertname=HighLatency \
  service=~"api|catalog" \
  --duration=4h \
  --comment="Maintenance window" \
  --author="ops@example.com"

# 取得 silence ID
amtool silence query --within=1h

# 取消 silence
amtool silence expire abc-123-def

# 過期 ID 列表
amtool silence expire $(amtool silence query -q | grep "alertname=Test")

3.5 Config 驗證 / 顯示

# 驗證 alertmanager.yml
amtool check-config alertmanager.yml

# Output
Checking 'alertmanager.yml'  SUCCESS
Found:
  - global config
  - route
  - 5 inhibit rules
  - 8 receivers
  - 2 templates

# 顯示當前 server config（從 API）
amtool config show

# 顯示當前 routing tree（樹狀）
amtool config routes show

# 測試一條 alert 會走哪 receiver
amtool config routes test \
  alertname=HighErrorRate severity=critical service=api

§ 4. grafanactl

Grafana Labs 的新世代 CLI（v11+），用於 Grafana resource as code。Ch 43 介紹過。 替代舊 grafana-cli 的 dashboard 管理功能。

4.1 設定 context

# 設定 default context
grafanactl config set contexts.default.grafana.server https://grafana.example.com
grafanactl config set contexts.default.grafana.org-id 1
grafanactl config set contexts.default.grafana.token glsa_xxxx

# 切換 context
grafanactl config use-context production
grafanactl config use-context staging

# 顯示當前
grafanactl config view

4.2 Resources

# 列 resource type
grafanactl resources list

# Output: dashboards, folders, datasources, alertrules, etc.

# 列 dashboard
grafanactl get dashboards
grafanactl get dashboards --folder=production

# Get specific dashboard
grafanactl get dashboard  --output=json > my-dashboard.json
grafanactl get dashboard  --output=yaml > my-dashboard.yaml

4.3 Apply（GitOps 風格）

# 從檔案 apply（類 kubectl apply）
grafanactl apply -f my-dashboard.yaml
grafanactl apply -f dashboards/    # 整 folder

# Pull all to local
grafanactl pull --output-dir=./grafana-resources/

# Push all from local
grafanactl push --input-dir=./grafana-resources/

4.4 Diff

# 看本地 vs server 差異
grafanactl diff -f my-dashboard.yaml

# CI 中常用：apply 前先 diff
grafanactl diff -f dashboards/ --output=summary

§ 5. grafana-cli

跟 Grafana server 一起裝的 CLI（在 Grafana host 上），用於 server 端管理。不是給 client 用。

5.1 Plugin 管理

# 安裝 plugin
grafana-cli plugins install grafana-piechart-panel
grafana-cli plugins install grafana-clock-panel 1.3.0   # 指定版本

# 列出已裝 plugin
grafana-cli plugins ls

# 更新所有 plugin
grafana-cli plugins update-all

# 移除
grafana-cli plugins remove grafana-piechart-panel

5.2 Admin 操作

# 重設 admin 密碼
grafana-cli admin reset-admin-password newpassword

# Data migration（升級時用）
grafana-cli admin data-migration encrypt-datasource-passwords

# 顯示版本
grafana-cli --version

5.3 從哪裝 plugin

# 預設從 Grafana plugin catalog
# 自訂 source
grafana-cli --pluginUrl https://my-plugin.tar.gz plugins install my-plugin

# 不允許 sign 的 plugin（unsigned）
GF_PLUGINS_ALLOW_LOADING_UNSIGNED_PLUGINS=my-plugin grafana-cli plugins install my-plugin

§ 6. mimirtool

6.1 認證

export MIMIR_ADDRESS=https://mimir.example.com
export MIMIR_TENANT_ID=tenant-prod
export MIMIR_API_KEY=xxxxx   # 或用 X-API-Key header

# 也可命令列
mimirtool rules list \
  --address=https://mimir.example.com \
  --id=tenant-prod \
  --key=xxxxx

6.2 Rules 管理

# 列出所有 rule group
mimirtool rules list

# 顯示完整 rule yaml
mimirtool rules print

# 上傳 rule
mimirtool rules load my-rules.yaml

# 上傳 + sync（刪除 server 上有但檔案沒的）
mimirtool rules sync my-rules.yaml

# 刪除特定 rule group
mimirtool rules delete --namespace=default --rule-group=api-alerts

# Lint rule（檢查語法）
mimirtool rules lint my-rules.yaml

# 測試 rule（同 promtool test）
mimirtool rules check my-rules.yaml --tests-file=tests.yaml

# 把 Prometheus rule 轉 Mimir 格式
mimirtool rules prepare prom-rules.yaml > mimir-rules.yaml

6.3 Cardinality 分析

# 找出 cardinality 最高的 metric
mimirtool analyze rule-groups my-rules.yaml

# 找出 dashboard 中用到的 metric
mimirtool analyze grafana --address=https://grafana.example.com --key=glsa_xxx

# 找 Prometheus 中沒被使用的 metric（unused metrics）
mimirtool analyze prometheus --address=http://prometheus:9090

# 完整工作流：找出 80% unused metric！
# Step 1: dump Grafana 用的 metric
mimirtool analyze grafana \
  --address=$GRAFANA_URL \
  --key=$GRAFANA_TOKEN \
  -o metrics-in-grafana.json

# Step 2: dump Prometheus rule 用的 metric
mimirtool analyze ruler \
  --address=$MIMIR_URL \
  --id=$TENANT_ID \
  --key=$MIMIR_KEY \
  -o metrics-in-ruler.json

# Step 3: 比對找 unused
mimirtool analyze prometheus \
  --address=$PROM_URL \
  --grafana-metrics-file=metrics-in-grafana.json \
  --ruler-metrics-file=metrics-in-ruler.json

# 結果常見：80% 的 metric 從未被 query！

6.4 Alertmanager 設定

# 取得當前 AM config
mimirtool alertmanager get

# 載入 AM config
mimirtool alertmanager load alertmanager.yaml

# UTF-8 migration（v2.12+）
mimirtool alertmanager migrate-utf8 \
  --input alertmanager.yaml \
  --output alertmanager-utf8.yaml

6.5 Remote Read（資料 export）

# Dump 一段時間資料到 stdout
mimirtool remote-read dump \
  --selector='up{job="node"}' \
  --address=https://mimir.example.com \
  --remote-read-path=/api/v1/read \
  --start=-1h

# Export 到 local TSDB（給 Prometheus 分析）
mimirtool remote-read export \
  --selector='{job="node"}' \
  --address=https://mimir.example.com \
  --tsdb-path=./local-tsdb

# 用 promtool 分析 export 出的 TSDB
promtool tsdb analyze ./local-tsdb

6.6 Config 驗證

# 驗證 Mimir config
mimirtool config validate mimir.yaml

# 顯示所有 config 預設值
mimirtool config defaults

# 比對版本之間 config 變化
mimirtool config diff old-mimir.yaml new-mimir.yaml

§ 7. logcli (Loki)

7.1 設定

export LOKI_ADDR=http://loki:3100
export LOKI_USERNAME=admin
export LOKI_PASSWORD=xxx

# 或 X-Scope-OrgID（multi-tenant）
export LOKI_ORG_ID=tenant-prod

7.2 LogQL Query

# Instant query
logcli query '{job="api"} |= "ERROR"'

# Range
logcli query \
  --from='2026-05-02T10:00:00Z' \
  --to='2026-05-02T11:00:00Z' \
  '{job="api"} |= "ERROR"'

# 限制行數
logcli query --limit=100 '{job="api"}'

# JSON 輸出（給 jq）
logcli query --output=jsonl '{job="api"}' | jq '.line'

# Tail（streaming）
logcli query --tail '{job="api"}'

7.3 Labels / Series

logcli labels                # 所有 label
logcli labels job            # 特定 label 的所有值
logcli series '{job="api"}'  # 找 series

7.4 統計

# 用 LogQL 算 rate
logcli query 'rate({job="api"} |= "ERROR" [5m])'

# Top 10 noisy services
logcli query 'topk(10, sum by (service) (rate({} [5m])))'

§ 8. tempo-cli

8.1 用途

Tempo 自家 CLI，主要給 admin / debugging 用：

  • 查特定 trace 內容
  • 檢查 backend block
  • migrating tenant data

8.2 常用指令

# 取得 trace
tempo-cli view trace  --backend=s3 --bucket=...

# 列出 backend block
tempo-cli list blocks --tenant-id=single-tenant

# 檢查 block 完整性
tempo-cli check-blocks single-tenant

# Migrate tenant
tempo-cli migrate-tenant --from=old --to=new

# 取得 block info
tempo-cli view index  --tenant-id=...

§ 9. profilecli (Pyroscope)

9.1 用途

Pyroscope 的 CLI，可以從 server 抓 profile 給離線分析。

9.2 常用

export PROFILECLI_URL=http://pyroscope:4040

# 列出 application
profilecli query series

# 查 profile（dump 為 pprof）
profilecli query merge \
  --from=now-1h \
  --to=now \
  --query='process_cpu:cpu:nanoseconds:cpu:nanoseconds{service_name="api"}' \
  --output=cpu.pprof

# 用 pprof 工具看
go tool pprof cpu.pprof
# 或互動式
go tool pprof -http=:8080 cpu.pprof

§ 10. 其他常用工具

10.1 grizzly (grr)

Grafana Labs 統一的 GitOps CLI，覆蓋多家：

# 安裝
brew install grafana/grafana/grizzly

# Init project
grr init my-project

# Pull 整 Grafana
grr pull -d ./resources/

# Apply（dry-run）
grr apply -d ./resources/ --dry-run

# Apply（實際）
grr apply -d ./resources/

# Diff
grr diff -d ./resources/

# Watch（持續同步）
grr watch -d ./resources/

# 支援 resource：
# - DashboardFolder / Dashboard
# - DataSource
# - AlertRuleGroup / NotificationPolicy / ContactPoint
# - SyntheticMonitoring（Grafana Cloud）
# - PrometheusRuleGroup（Mimir）
# - LogAlertRuleGroup（Loki）

10.2 prom2json

把 metric endpoint 的 plain text 轉 JSON：

prom2json http://localhost:9100/metrics | jq '.[] | select(.name=="up")'

10.3 prometheus_unit_test

用 Docker 跑 unit test（不用裝 promtool）：

docker run --rm \
  -v $(pwd):/work \
  -w /work \
  prom/prometheus:latest \
  promtool test rules tests/api-rules.yaml

10.4 sloth

SLO YAML → Prometheus rule 自動產生（Ch 56）：

# 產生
sloth generate -i slo-defs.yaml -o prometheus-rules.yaml

# 驗證
sloth validate -i slo-defs.yaml

# K8s mode（CRD）
sloth k8s-controller

10.5 cortextool（deprecated）

📌 Deprecated

cortextool 已 deprecated，請改用 mimirtool。 cortextool 的功能都搬到 mimirtool。

§ 11. CI/CD 整合範例

11.1 GitHub Actions：rule lint + test

name: Validate Prometheus Rules
on:
  pull_request:
    paths:
      - 'rules/**.yaml'
      - 'tests/**.yaml'

jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install promtool
        run: |
          wget https://github.com/prometheus/prometheus/releases/.../prometheus-linux-amd64.tar.gz
          tar xzf prometheus-*.tar.gz
          sudo mv prometheus-*/promtool /usr/local/bin/

      - name: Lint rules
        run: promtool check rules rules/*.yaml

      - name: Run unit tests
        run: promtool test rules tests/*.yaml

      - name: Validate Mimir rules
        run: |
          docker run --rm -v $PWD:/work -w /work \
            grafana/mimirtool rules lint rules/*.yaml

11.2 GitHub Actions：dashboard sync

name: Deploy Grafana Dashboards
on:
  push:
    branches: [main]
    paths:
      - 'dashboards/**'

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install grizzly
        run: |
          curl -L https://github.com/grafana/grizzly/releases/.../grr-linux-amd64 \
            -o /usr/local/bin/grr
          chmod +x /usr/local/bin/grr

      - name: Diff
        env:
          GRAFANA_URL: https://grafana.example.com
          GRAFANA_TOKEN: ${{ secrets.GRAFANA_TOKEN }}
        run: grr diff -d dashboards/

      - name: Apply
        if: github.event_name == 'push'
        env:
          GRAFANA_URL: https://grafana.example.com
          GRAFANA_TOKEN: ${{ secrets.GRAFANA_TOKEN }}
        run: grr apply -d dashboards/

11.3 GitLab CI：完整 observability pipeline

stages:
  - validate
  - test
  - deploy

validate-rules:
  stage: validate
  image: prom/prometheus:latest
  script:
    - promtool check rules rules/*.yaml

test-rules:
  stage: test
  image: prom/prometheus:latest
  script:
    - promtool test rules tests/*.yaml

deploy-rules:
  stage: deploy
  image: grafana/mimirtool:latest
  only:
    - main
  script:
    - mimirtool rules sync rules/*.yaml

deploy-dashboards:
  stage: deploy
  image: grafana/grizzly:latest
  only:
    - main
  script:
    - grr apply -d dashboards/

11.4 unused metric audit（每月跑）

#!/bin/bash
# monthly-metric-audit.sh

# 1. 從 Grafana 抓 metric in dashboards
mimirtool analyze grafana \
  --address=$GRAFANA_URL \
  --key=$GRAFANA_TOKEN \
  -o /tmp/metrics-in-grafana.json

# 2. 從 ruler 抓 metric in rules
mimirtool analyze ruler \
  --address=$MIMIR_URL \
  --id=$TENANT_ID \
  --key=$MIMIR_KEY \
  -o /tmp/metrics-in-ruler.json

# 3. 找 unused metric
mimirtool analyze prometheus \
  --address=$PROM_URL \
  --grafana-metrics-file=/tmp/metrics-in-grafana.json \
  --ruler-metrics-file=/tmp/metrics-in-ruler.json \
  -o /tmp/unused-report.json

# 4. 報告 top 100 unused metric（給 team review）
jq '.unused_metrics[:100]' /tmp/unused-report.json > unused-top100.json

# 5. 發 Slack 通知
echo "Monthly audit: $(jq '.in_use_count' /tmp/unused-report.json) used, $(jq '.not_in_use_count' /tmp/unused-report.json) unused" \
  | curl -X POST -d @- $SLACK_WEBHOOK

✅ 本章重點摘要

30 秒回顧
  • promtool：check config / rules、query CLI、test rules（unit test！）、TSDB 分析
  • Rule unit test 用 input_series + alert_rule_test + promql_expr_test
  • amtool：silence 管理（add / query / expire）、alert query、config 驗證、route test
  • Silence 必填 matcher + duration + comment + author
  • grafanactl：新世代 Grafana CLI（v11+），context-based，類 kubectl，給 GitOps 用
  • grafana-cli：跟 server 一起裝，主要管 plugin，admin reset password
  • mimirtool：rule sync / lint / test、cardinality 分析、找 unused metric（80% 通常從未 query！）
  • mimirtool unused metric 工作流：grafana → ruler → prometheus 三步
  • mimirtool 也支援 alertmanager config / UTF-8 migration / remote-read export
  • logcli：Loki LogQL CLI，支援 query / labels / tail / 統計
  • tempo-cli：admin / debug 用，view trace / list blocks / migrate tenant
  • profilecli：Pyroscope CLI，dump 為 pprof 給 go tool pprof
  • grizzly (grr)：統一 GitOps CLI，覆蓋 Grafana / Mimir / Loki / Tempo
  • cortextool deprecated，改用 mimirtool
  • CI/CD 必做：rule lint + unit test + dashboard sync + 每月 unused metric audit
