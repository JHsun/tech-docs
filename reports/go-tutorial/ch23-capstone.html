綜合專案 URL Shortener

最後一章 — 把前 22 章的知識全部縫起來。我們將實作一個可直接執行的 URL shortener 服務：支援建立短網址、重導、熱門統計、限流、優雅關機、完整測試。

技術組合 · HTTP + JSON + context + sync + slog 實作目標 · 可 build、可執行、有測試

專案規格

我們要做的是單一二進位的 URL shortener — 可以擴充到多服務，但先從可運作的最小版本開始。

API 設計

端點行為
POST /api/shorten建立短網址，回 JSON
GET /{code}重導到原網址並計數
GET /api/stats/{code}查單個短網址的點擊數
GET /api/top?n=10熱門短網址排行
GET /healthz健康檢查

FILE · internal/shortener/shortener.go 
package shortener

import (
    "context"
    "crypto/rand"
    "encoding/base64"
    "errors"
    "net/url"
    "time"
)

// 公開錯誤 — 外層可用 errors.Is 分類
var (
    ErrNotFound      = errors.New("shortener: code not found")
    ErrInvalidURL    = errors.New("shortener: invalid URL")
    ErrCodeCollision = errors.New("shortener: code collision")
)

// Link 是一筆短網址紀錄
type Link struct {
    Code      string    `json:"code"`
    URL       string    `json:"url"`
    CreatedAt time.Time `json:"created_at"`
    Hits      int64     `json:"hits"`
}

// Store 是領域層定義的「我需要什麼能力」— 實作在哪不關我事
type Store interface {
    Save(ctx context.Context, link *Link) error
    Get(ctx context.Context, code string) (*Link, error)
    IncHit(ctx context.Context, code string) error
    Top(ctx context.Context, n int) ([]*Link, error)
}

// Service 是 shortener 的業務邏輯
type Service struct {
    store Store
    clock func() time.Time    // 注入以便測試
}

func NewService(store Store) *Service {
    return &Service{
        store: store,
        clock: time.Now,
    }
}

// Shorten 驗證 URL 並產生短 code；失敗時回傳具名 error
func (s *Service) Shorten(ctx context.Context, rawURL string) (*Link, error) {
    if _, err := url.ParseRequestURI(rawURL); err != nil {
        return nil, ErrInvalidURL
    }

    // 試最多 5 次 — 碰撞機率本就極低，真碰到就再來一次
    for i := 0; i < 5; i++ {
        code, err := randomCode(7)
        if err != nil {
            return nil, err
        }
        link := &Link{
            Code:      code,
            URL:       rawURL,
            CreatedAt: s.clock(),
        }
        if err := s.store.Save(ctx, link); err != nil {
            if errors.Is(err, ErrCodeCollision) {
                continue
            }
            return nil, err
        }
        return link, nil
    }
    return nil, ErrCodeCollision
}

// Resolve 找出 code 對應的 URL，並非同步累加點擊數
func (s *Service) Resolve(ctx context.Context, code string) (*Link, error) {
    link, err := s.store.Get(ctx, code)
    if err != nil {
        return nil, err
    }

    // 非同步更新點擊 — 不阻塞重導；斷開 ctx 避免跟著原 request 被取消
    detached := context.WithoutCancel(ctx)
    go s.store.IncHit(detached, code)

    return link, nil
}

// Top 回傳熱門前 n 個
func (s *Service) Top(ctx context.Context, n int) ([]*Link, error) {
    if n <= 0 || n > 100 {
        n = 10
    }
    return s.store.Top(ctx, n)
}

// randomCode 產生長度 n 的 URL-safe 隨機 code
func randomCode(n int) (string, error) {
    // base64 URL encoding 每 3 bytes 產 4 字元
    bytes := make([]byte, (n*3+3)/4)
    if _, err := rand.Read(bytes); err != nil {
        return "", err
    }
    return base64.RawURLEncoding.EncodeToString(bytes)[:n], nil
}

記憶體實作：memstore

用 sync.RWMutex + map 的經典組合。真實服務會替換成 Postgres / Redis，但介面契約一樣。

FILE · internal/shortener/memstore.go 
package shortener

import (
    "context"
    "sort"
    "sync"
    "sync/atomic"
)

// MemStore 是 in-memory 的 Store 實作
type MemStore struct {
    mu    sync.RWMutex
    links map[string]*memEntry
}

// memEntry 把 hits 獨立成 atomic 欄位 — 可無鎖遞增
type memEntry struct {
    link *Link
    hits atomic.Int64
}

func NewMemStore() *MemStore {
    return &MemStore{
        links: make(map[string]*memEntry),
    }
}

func (m *MemStore) Save(_ context.Context, link *Link) error {
    m.mu.Lock()
    defer m.mu.Unlock()

    if _, exists := m.links[link.Code]; exists {
        return ErrCodeCollision
    }
    m.links[link.Code] = &memEntry{link: link}
    return nil
}

func (m *MemStore) Get(_ context.Context, code string) (*Link, error) {
    m.mu.RLock()
    entry, ok := m.links[code]
    m.mu.RUnlock()

    if !ok {
        return nil, ErrNotFound
    }

    // 複製出一份（含最新 hits）避免呼叫端修改內部狀態
    out := *entry.link
    out.Hits = entry.hits.Load()
    return &out, nil
}

func (m *MemStore) IncHit(_ context.Context, code string) error {
    m.mu.RLock()
    entry, ok := m.links[code]
    m.mu.RUnlock()

    if !ok {
        return ErrNotFound
    }
    entry.hits.Add(1)    // atomic — 不需要寫鎖
    return nil
}

func (m *MemStore) Top(_ context.Context, n int) ([]*Link, error) {
    m.mu.RLock()
    out := make([]*Link, 0, len(m.links))
    for _, e := range m.links {
        copy := *e.link
        copy.Hits = e.hits.Load()
        out = append(out, &copy)
    }
    m.mu.RUnlock()

    sort.Slice(out, func(i, j int) bool {
        return out[i].Hits > out[j].Hits
    })
    if len(out) > n {
        out = out[:n]
    }
    return out, nil
}

package shortener

import (
    "context"
    "errors"
    "testing"
    "time"

func newTestService() *Service {
    svc := NewService(NewMemStore())
    // 固定時鐘 — 讓測試可重現
    svc.clock = func() time.Time {
        return time.Date(2026, 1, 1, 0, 0, 0, 0, time.UTC)
    }
    return svc
}

func TestShorten(t *testing.T) {
    tests := []struct {
        name    string
        url     string
        wantErr error
    }{
        {"valid https", "https://go.dev", nil},
        {"valid http", "http://example.com/path?q=1", nil},
        {"empty", "", ErrInvalidURL},
        {"not a url", "hello world", ErrInvalidURL},
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            t.Parallel()
            svc := newTestService()
            ctx := context.Background()

            link, err := svc.Shorten(ctx, tc.url)
            if !errors.Is(err, tc.wantErr) {
                t.Fatalf("err = %v, want %v", err, tc.wantErr)
            }
            if tc.wantErr != nil {
                return
            }
            if link.Code == "" {
                t.Error("empty code")
            }
            if link.URL != tc.url {
                t.Errorf("URL = %q, want %q", link.URL, tc.url)
            }
        })
    }
}

func TestResolveAndHits(t *testing.T) {
    svc := newTestService()
    ctx := context.Background()

    link, err := svc.Shorten(ctx, "https://go.dev")
    if err != nil {
        t.Fatal(err)
    }

    // 連續 3 次 resolve
    for i := 0; i < 3; i++ {
        if _, err := svc.Resolve(ctx, link.Code); err != nil {
            t.Fatal(err)
        }
    }

    // IncHit 是非同步的，給一點時間
    time.Sleep(50 * time.Millisecond)

    got, _ := svc.store.Get(ctx, link.Code)
    if got.Hits != 3 {
        t.Errorf("Hits = %d, want 3", got.Hits)
    }
}

func TestResolveNotFound(t *testing.T) {
    svc := newTestService()
    _, err := svc.Resolve(context.Background(), "missing")
    if !errors.Is(err, ErrNotFound) {
        t.Errorf("err = %v, want ErrNotFound", err)
    }
}

func TestTop(t *testing.T) {
    svc := newTestService()
    ctx := context.Background()

    // 建三個並製造點擊差異
    urls := []string{"https://a.com", "https://b.com", "https://c.com"}
    codes := make([]string, len(urls))
    for i, u := range urls {
        link, _ := svc.Shorten(ctx, u)
        codes[i] = link.Code
    }

    // a 點 1 次、b 點 3 次、c 點 2 次
    clicks := map[int]int{0: 1, 1: 3, 2: 2}
    for i, n := range clicks {
        for j := 0; j < n; j++ {
            svc.Resolve(ctx, codes[i])
        }
    }
    time.Sleep(100 * time.Millisecond)

    top, _ := svc.Top(ctx, 3)
    if len(top) != 3 {
        t.Fatalf("len(top) = %d", len(top))
    }
    // 應該按 hits 降冪：b > c > a
    if top[0].URL != "https://b.com" {
        t.Errorf("top[0] = %q, want b.com", top[0].URL)
    }
}

package httpapi

import (
    "encoding/json"
    "errors"
    "net/http"
    "strconv"

    "github.com/myorg/urlshort/internal/shortener"
)

type API struct {
    svc     *shortener.Service
    baseURL string    // e.g. "http://localhost:8080"
}

func NewAPI(svc *shortener.Service, baseURL string) *API {
    return &API{svc: svc, baseURL: baseURL}
}

type shortenReq struct {
    URL string `json:"url"`
}

type shortenResp struct {
    Code     string `json:"code"`
    ShortURL string `json:"short_url"`
    URL      string `json:"url"`
}

func (a *API) shorten(w http.ResponseWriter, r *http.Request) {
    var req shortenReq
    dec := json.NewDecoder(r.Body)
    dec.DisallowUnknownFields()
    if err := dec.Decode(&req); err != nil {
        writeError(w, http.StatusBadRequest, "invalid body")
        return
    }

    link, err := a.svc.Shorten(r.Context(), req.URL)
    switch {
    case errors.Is(err, shortener.ErrInvalidURL):
        writeError(w, http.StatusBadRequest, "invalid url")
        return
    case err != nil:
        writeError(w, http.StatusInternalServerError, "internal error")
        return
    }

    writeJSON(w, http.StatusCreated, shortenResp{
        Code:     link.Code,
        ShortURL: a.baseURL + "/" + link.Code,
        URL:      link.URL,
    })
}

func (a *API) redirect(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    link, err := a.svc.Resolve(r.Context(), code)
    switch {
    case errors.Is(err, shortener.ErrNotFound):
        writeError(w, http.StatusNotFound, "not found")
        return
    case err != nil:
        writeError(w, http.StatusInternalServerError, "internal error")
        return
    }
    http.Redirect(w, r, link.URL, http.StatusFound)
}

func (a *API) stats(w http.ResponseWriter, r *http.Request) {
    code := r.PathValue("code")
    // 直接 Get 而不 Resolve — 查詢不算一次點擊
    link, err := a.svc.ResolveWithoutCounting(r.Context(), code)
    if errors.Is(err, shortener.ErrNotFound) {
        writeError(w, http.StatusNotFound, "not found")
        return
    }
    if err != nil {
        writeError(w, http.StatusInternalServerError, "internal error")
        return
    }
    writeJSON(w, http.StatusOK, link)
}

func (a *API) top(w http.ResponseWriter, r *http.Request) {
    n, _ := strconv.Atoi(r.URL.Query().Get("n"))
    links, err := a.svc.Top(r.Context(), n)
    if err != nil {
        writeError(w, http.StatusInternalServerError, "internal error")
        return
    }
    writeJSON(w, http.StatusOK, links)
}

func (a *API) healthz(w http.ResponseWriter, _ *http.Request) {
    w.Write([]byte("ok"))
}

// --- helpers ---

func writeJSON(w http.ResponseWriter, code int, v any) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(code)
    json.NewEncoder(w).Encode(v)
}

func writeError(w http.ResponseWriter, code int, msg string) {
    writeJSON(w, code, map[string]string{"error": msg})
}

Middleware：logging / recover / rate limit

FILE · internal/httpapi/middleware.go 
package httpapi

import (
    "log/slog"
    "net/http"
    "runtime/debug"
    "sync"
    "time"

    "golang.org/x/time/rate"
)

// Middleware 是「接收 handler、回傳 handler」的函式
type Middleware func(http.Handler) http.Handler

// Chain 把多個 middleware 串起來；前面的先執行
func Chain(h http.Handler, mws ...Middleware) http.Handler {
    for i := len(mws) - 1; i >= 0; i-- {
        h = mws[i](h)
    }
    return h
}

// statusRecorder 攔截 WriteHeader 以便 log 狀態碼
type statusRecorder struct {
    http.ResponseWriter
    code int
}

func (s *statusRecorder) WriteHeader(c int) {
    s.code = c
    s.ResponseWriter.WriteHeader(c)
}

// Logging 記錄每個請求
func Logging(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        rec := &statusRecorder{ResponseWriter: w, code: 200}
        next.ServeHTTP(rec, r)
        slog.Info("request",
            "method", r.Method,
            "path", r.URL.Path,
            "status", rec.code,
            "duration_ms", time.Since(start).Milliseconds(),
            "remote", r.RemoteAddr,
        )
    })
}

// Recover 攔截 panic 轉成 500
func Recover(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if rec := recover(); rec != nil {
                slog.Error("panic", "err", rec, "stack", string(debug.Stack()))
                http.Error(w, "internal server error", 500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

// RateLimit 每個 IP 各自的 limiter — 防濫用的最小實作
type RateLimit struct {
    mu      sync.Mutex
    clients map[string]*rate.Limiter
    r       rate.Limit
    burst   int
}

func NewRateLimit(rps rate.Limit, burst int) *RateLimit {
    return &RateLimit{
        clients: make(map[string]*rate.Limiter),
        r:       rps,
        burst:   burst,
    }
}

func (rl *RateLimit) get(ip string) *rate.Limiter {
    rl.mu.Lock()
    defer rl.mu.Unlock()
    lim, ok := rl.clients[ip]
    if !ok {
        lim = rate.NewLimiter(rl.r, rl.burst)
        rl.clients[ip] = lim
    }
    return lim
}

func (rl *RateLimit) Middleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        if !rl.get(r.RemoteAddr).Allow() {
            http.Error(w, "too many requests", 429)
            return
        }
        next.ServeHTTP(w, r)
    })
}

package httpapi

import "net/http"

func (a *API) Routes(mws ...Middleware) http.Handler {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /healthz", a.healthz)
    mux.HandleFunc("POST /api/shorten", a.shorten)
    mux.HandleFunc("GET /api/stats/{code}", a.stats)
    mux.HandleFunc("GET /api/top", a.top)
    mux.HandleFunc("GET /{code}", a.redirect)    // 必須放最後 — wildcard
    return Chain(mux, mws...)
}
FILE · cmd/server/main.go 
package main

import (
    "context"
    "log/slog"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"

    "github.com/myorg/urlshort/internal/httpapi"
    "github.com/myorg/urlshort/internal/shortener"
    "golang.org/x/time/rate"
)

func main() {
    // 結構化 JSON 日誌
    logger := slog.New(slog.NewJSONHandler(os.Stdout, nil))
    slog.SetDefault(logger)

    // 設定（env 優先，否則預設）
    addr := envOr("ADDR", ":8080")
    baseURL := envOr("BASE_URL", "http://localhost"+addr)

    // 組裝 — wiring root
    store := shortener.NewMemStore()
    svc := shortener.NewService(store)
    api := httpapi.NewAPI(svc, baseURL)

    rl := httpapi.NewRateLimit(rate.Limit(10), 20)    // 10 rps, burst 20

    handler := api.Routes(
        httpapi.Recover,
        httpapi.Logging,
        rl.Middleware,
    )

    srv := &http.Server{
        Addr:              addr,
        Handler:           handler,
        ReadHeaderTimeout: 5 * time.Second,
        ReadTimeout:       10 * time.Second,
        WriteTimeout:      15 * time.Second,
        IdleTimeout:       60 * time.Second,
    }

    // 非阻塞啟動
    go func() {
        slog.Info("server listening", "addr", addr)
        if err := srv.ListenAndServe(); err != http.ErrServerClosed {
            slog.Error("server error", "err", err)
            os.Exit(1)
        }
    }()

    // 等中斷信號
    ctx, stop := signal.NotifyContext(context.Background(),
        syscall.SIGINT, syscall.SIGTERM)
    defer stop()
    <-ctx.Done()

    slog.Info("shutting down")
    sctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()
    if err := srv.Shutdown(sctx); err != nil {
        slog.Error("shutdown error", "err", err)
    }
    slog.Info("bye")
}

func envOr(k, def string) string {
    if v := os.Getenv(k); v != "" {
        return v
    }
    return def
}

func TestShortenHandler(t *testing.T) {

# CGO 關閉、靜態連結、去除符號
RUN CGO_ENABLED=0 GOOS=linux go build \
    -ldflags="-s -w" \
    -o /out/urlshort ./cmd/server

# --- runtime stage ---
FROM gcr.io/distroless/static-debian12:nonroot
COPY --from=build /out/urlshort /urlshort
EXPOSE 8080
USER nonroot:nonroot
ENTRYPOINT ["/urlshort"]
為什麼用 distroless？ distroless/static 只有執行 Go 靜態二進位需要的 CA certs — 沒有 shell、沒有 package manager。最終 image 小於 10MB、攻擊面極窄。用 nonroot user 進一步降低風險。

把它變成生產級服務的下一步

這個 MVP 可以運作，但距離真正「上線服務百萬使用者」還有事情要做。以下是練習清單。

① 替換持久化

實作 PostgresStore 或 RedisStore — 只需要滿足 shortener.Store 介面，Service 一行都不用改。

② 加上 prometheus metrics

暴露 /metrics endpoint；記錄 request count、latency histogram、active goroutines。

③ 加 TLS 與 HTTP/2

srv.ListenAndServeTLS，或用反向代理（Nginx / Caddy / Cloud Load Balancer）處理。

④ 整合 OpenTelemetry

distributed tracing — 對微服務架構尤其有用。go.opentelemetry.io/otel。

⑤ 加 auth middleware

JWT / API key，可重用 Chain 模式插進去。

⑥ 加 cache layer

熱點 code 放 LRU in-memory；用 singleflight 合併同時的 DB 查詢。

⑦ 加 load test

用 vegeta / k6 壓測，配 pprof 找瓶頸。

⑧ fuzzing

對 shorten 寫 fuzz 測試 — 各種奇怪 URL 輸入都不該 panic。

結語：從 Python 到 Go 的心路

走完 23 章，你應該體會到 Go 的設計哲學：少即是多。

Go 的核心價值
① 可讀性勝過巧思。Go 的程式碼像 Python 一樣簡潔，但執行像 C 一樣快。
② 靜態型別提早抓錯。來自 Python 的你應該特別有感：refactor 不再是賭博。
③ 錯誤是值。沒有隱藏的例外 — 每個出錯的地方都在 code 裡明寫。
④ Goroutine 與 Channel 是招牌。並發從「困難的特殊技能」變成「日常工具」。
⑤ 工具鏈內建。格式化、測試、profile、fuzzing — 不用挑框架、不用吵架。
⑥ 生態務實。標準庫已經夠強，第三方只是補充。
