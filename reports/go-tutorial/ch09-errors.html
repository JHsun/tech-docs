CHAPTER 09 · 核心語法

錯誤處理 Error Handling

Go 沒有 try/catch — 錯誤是值，用 if err != nil 處理。這個設計看似囉唆，實際上讓每個錯誤的處理位置明確可見，是 Go 可靠性哲學的核心。

核心介面 · error { Error() string } 標準套件 · errors · fmt

Go 的錯誤處理哲學

Go 把錯誤當作「一種回傳值」— 不是例外（exception）也不是狀態碼。函式慣例：最後一個回傳值是 error，成功時為 nil。

// 標準簽名模式
func readFile(path string) ([]byte, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return nil, err      // 把錯誤往上丟
    }
    return data, nil         // 成功回傳 nil error
}

// 呼叫端的標準慣用法
data, err := readFile("config.json")
if err != nil {
    return fmt.Errorf("load config: %w", err)
}
// 成功路徑繼續往下走
process(data)
對照 Python Python 的 try: ... except FileNotFoundError: ... 是「隱藏」的控制流 — 呼叫者不看 code 無法知道哪些錯誤可能發生。Go 把這些寫進回傳簽名，強迫呼叫端正視錯誤。
絕對不要忽略 err data, _ := readFile(...) 是極度危險的寫法（linter 會警告）。真的確定不會錯（例如讀取常數）才用 _，否則一律處理。

error 只是一個介面

沒有神奇的 built-in error 型別 — 它就是個只有一個方法的介面，任何型別只要實作 Error() string 就能當作 error。

// 內建的 error 定義（實際上在 builtin 套件）
type error interface {
    Error() string
}

// 最簡單的產生方式：errors.New
import "errors"
err := errors.New("something went wrong")

// 帶格式化：fmt.Errorf
err := fmt.Errorf("invalid id %d", id)

自訂錯誤型別

// 當你需要攜帶額外欄位時
type ValidationError struct {
    Field   string
    Message string
}

func (e *ValidationError) Error() string {
    return fmt.Sprintf("%s: %s", e.Field, e.Message)
}

// 使用
func validate(u User) error {
    if u.Name == "" {
        return &ValidationError{Field: "Name", Message: "required"}
    }
    return nil
}
為什麼指標接收器？ 自訂 error 幾乎都用 *T（指標接收器 + 傳指標）。原因是比較：兩個 errors.New("x") 用 == 比較是 false，但同一個 error 物件的指標 == 自己才是 true。這是 sentinel error 的基礎。

errors 套件三寶 · Is / As / Unwrap

Go 1.13 加入的錯誤包裝（error wrapping）機制，讓你既能保留完整錯誤鏈、又能精準檢查錯誤類型。

fmt.Errorf + %w：包裝錯誤

func loadConfig(path string) error {
    f, err := os.Open(path)
    if err != nil {
        // %w 包裝原錯誤，保留完整鏈
        return fmt.Errorf("loadConfig %q: %w", path, err)
    }
    defer f.Close()
    // ...
    return nil
}

// 錯誤訊息會變成：
// loadConfig "x.json": open x.json: no such file or directory

errors.Is：檢查是否為某個特定錯誤

import "errors"

err := loadConfig("missing.json")
if errors.Is(err, os.ErrNotExist) {
    fmt.Println("檔案不存在")
    // errors.Is 會沿著 %w 包裝鏈一路找下去
}

// 定義自己的 sentinel error（套件層級的哨兵常數）
var ErrUserNotFound = errors.New("user not found")

if errors.Is(err, ErrUserNotFound) {
    // 處理...
}

errors.As：取出特定型別的錯誤

var vErr *ValidationError
if errors.As(err, &vErr) {
    fmt.Println("欄位:", vErr.Field)
    fmt.Println("訊息:", vErr.Message)
}

// errors.As 也會沿著包裝鏈找，並把找到的錯誤賦值給傳入的指標
函式問題回傳
errors.Is(err, target)err 是不是 target（或包裝了 target）？bool
errors.As(err, &v)err 鏈中有沒有 v 型別的錯誤？bool + 填入 v
errors.Unwrap(err)取出被 %w 包裝的原錯誤error（可能 nil）

三種錯誤策略：什麼時候用哪種？

Sentinel Error

var ErrNotFound = errors.New("...")
呼叫端用 errors.Is 比對。適合有限、穩定的錯誤類別（io.EOF、sql.ErrNoRows）。

Typed Error

自訂 struct 實作 error。呼叫端用 errors.As 取出欄位。適合需要攜帶上下文（ValidationError 的 Field 欄位、HTTPError 的 StatusCode）。

Opaque Error

只回傳 fmt.Errorf("...")，呼叫端只讀訊息、不分類。適合多數一般情境 — 愈不透明的 API 愈有彈性。

決策樹 ① 呼叫端需要分類判斷嗎？不用 → 用 Opaque（最常見）。
② 需要判斷但沒有附加資料？用 Sentinel。
③ 需要判斷 + 附加資料？用 Typed。

panic & recover · 程式的緊急出口

Go 有 panic，但哲學是「panic 只用於不可恢復的程式錯誤」 — 陣列越界、nil 解引用、程式設計錯誤。業務邏輯錯誤絕對用 error。

// panic 會立刻中止當前函式，往上 unwind 執行所有 defer
func mustHave(x int) {
    if x < 0 {
        panic("negative number not allowed")
    }
}

// recover 只能在 defer 中有效；抓到 panic 值，讓 goroutine 存活
func safeRun(fn func()) {
    defer func() {
        if r := recover(); r != nil {
            log.Printf("recovered: %v", r)
        }
    }()
    fn()
}

// 典型應用：HTTP middleware 阻止單一 request 的 panic 拖垮整個 server
func recoverMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if rec := recover(); rec != nil {
                log.Printf("panic: %v\n%s", rec, debug.Stack())
                http.Error(w, "internal error", 500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}
panic 是 last resort 初學者常把 panic 當 exception 用 — 錯。Go 的慣用法：
✓ 套件初始化失敗（init 函式）
✓ 明顯的程式邏輯錯誤（不變量違反）
✓ 生產環境的終極保護網（HTTP middleware）
✗ 檔案不存在、網路失敗、使用者輸入錯誤 — 這些都是 error！

實戰模板

TEMPLATE · 完整的錯誤分層處理 
package user

import (
    "database/sql"
    "errors"
    "fmt"
)

// 1) Sentinel errors — 套件公開的錯誤類別
var (
    ErrNotFound    = errors.New("user not found")
    ErrUnauthorized = errors.New("unauthorized")
)

// 2) Typed error — 攜帶欄位資訊
type ValidationError struct {
    Field string
    Rule  string
}

func (e *ValidationError) Error() string {
    return fmt.Sprintf("validation failed: %s (%s)", e.Field, e.Rule)
}

// 3) 函式中的包裝
func (s *Service) Get(id string) (*User, error) {
    if id == "" {
        return nil, &ValidationError{Field: "id", Rule: "required"}
    }

    u, err := s.db.QueryUser(id)
    if err != nil {
        if errors.Is(err, sql.ErrNoRows) {
            return nil, fmt.Errorf("user.Get %q: %w", id, ErrNotFound)
        }
        return nil, fmt.Errorf("user.Get %q: %w", id, err)
    }
    return u, nil
}

// 4) 呼叫端的分層判斷
u, err := svc.Get("x")
switch {
case errors.Is(err, ErrNotFound):
    // 404
case errors.Is(err, ErrUnauthorized):
    // 401
default:
    var vErr *ValidationError
    if errors.As(err, &vErr) {
        // 400 with field info
    }
    // 500
}
TEMPLATE · errors.Join · 聚合多個錯誤（Go 1.20+） 
import "errors"

func validate(u User) error {
    var errs []error
    if u.Name == "" {
        errs = append(errs, errors.New("name required"))
    }
    if u.Age < 0 {
        errs = append(errs, errors.New("age must be >= 0"))
    }
    if u.Email == "" {
        errs = append(errs, errors.New("email required"))
    }
    // errors.Join 會把多個錯誤合成一個；nil 會被忽略；全 nil 回傳 nil
    return errors.Join(errs...)
}

// errors.Is 也能在 Join 的錯誤鏈中運作
if err := validate(u); err != nil {
    fmt.Println(err)  // 每行一個錯誤
}

反模式大全

// ❌ 反模式 1：吞掉錯誤
data, _ := someFunc()

// ❌ 反模式 2：把 error 轉字串比對
if strings.Contains(err.Error(), "not found") { ... }
// 改用 errors.Is(err, ErrNotFound)

// ❌ 反模式 3：記錄又再往上丟
if err != nil {
    log.Println(err)
    return err   // 導致 log 被重複列印
}
// 原則：要嘛記錄、要嘛往上丟，別兩個都做

// ❌ 反模式 4：用 panic 代替 error
func getUser(id string) *User {
    if id == "" {
        panic("empty id")
    }
    // ...
}
// 業務錯誤一律用 error 回傳

// ❌ 反模式 5：回傳 *MyError 當 error（前章已提）
func do() error {
    var e *MyError
    // ... e 可能維持 nil
    return e    // ❌ 即使 e 是 nil，包成 error 就不是 nil 了
}
// ✅ 改成明確回傳 nil 或真的 error

本章速記

  1. error 是最後一個回傳值：(T, error) 是 Go 的招牌簽名。
  2. if err != nil 是一級語法：寫幾百次後會變成肌肉記憶。
  3. 永遠不要 _ = err：真的不關心才用 _，且幾乎沒有該情境。
  4. fmt.Errorf + %w 包裝錯誤：保留完整鏈，加上上下文。
  5. errors.Is 比對 sentinel，errors.As 取出 typed。
  6. errors.Join（Go 1.20+）聚合多個錯誤。
  7. sentinel 用 var ErrXxx = errors.New(...)：套件層級公開常數。
  8. typed error 用指標接收器：避免介面值比較的 nil 陷阱。
  9. panic 只用於不可恢復的錯誤：業務邏輯一律 error。
  10. recover 只在 defer 中有效：是 goroutine 的保險絲。
  11. 記錄 + 丟出只能擇一：避免重複日誌。
  12. 別用字串比對錯誤：永遠用 errors.Is / errors.As。
