Net / HTTP

Go 的 net/http 是生產級的 HTTP 實作 — 不是玩具、不是必須配 nginx。Go 1.22 全新的 routing 語法讓「不要 framework」從口號變成現實。

核心套件 · net/http · net/url · encoding/json Go 1.22+ · Method-aware routing

HTTP Client · 發送請求

import "net/http"

// 最簡單（但別在生產用 http.Get／Post）
resp, err := http.Get("https://api.example.com/data")
if err != nil {
    return err
}
defer resp.Body.Close()     // ⚠️ 一定要 Close！

data, err := io.ReadAll(resp.Body)
fmt.Println(resp.StatusCode, string(data))

// 檢查狀態碼
if resp.StatusCode != http.StatusOK {
    return fmt.Errorf("bad status: %s", resp.Status)
}
三個極常見 Bug ① 忘了 Close Body：連線洩漏，最終耗盡 file descriptor。
② 沒讀完 Body：keep-alive 連線無法重用，下次請求建新連線。
③ http.Get 沒超時：DefaultClient 沒預設 timeout，惡意 server 可讓你 goroutine 永遠掛住。

正確的 Client 設定

// 生產級 client：自訂 timeout + transport
client := &http.Client{
    Timeout: 10 * time.Second,      // 涵蓋整個請求（連線+請求+回應）
    Transport: &http.Transport{
        MaxIdleConns:        100,
        MaxIdleConnsPerHost: 10,
        IdleConnTimeout:     90 * time.Second,
        TLSHandshakeTimeout: 10 * time.Second,
    },
}

// 帶 ctx 與 header 的完整請求
req, err := http.NewRequestWithContext(ctx, "POST", url, body)
if err != nil {
    return err
}
req.Header.Set("Content-Type", "application/json")
req.Header.Set("Authorization", "Bearer " + token)
req.Header.Set("User-Agent", "myapp/1.0")

resp, err := client.Do(req)
if err != nil {
    return err
}
defer resp.Body.Close()

// 即使不用 body 也要讀完，否則 keep-alive 無法重用連線
if resp.StatusCode >= 400 {
    io.Copy(io.Discard, resp.Body)
    return fmt.Errorf("status %d", resp.StatusCode)
}

HTTP Server · 最小範例

package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "Hello, World!")
    })

    http.HandleFunc("/hello", func(w http.ResponseWriter, r *http.Request) {
        name := r.URL.Query().Get("name")
        fmt.Fprintf(w, "Hello, %s!", name)
    })

    // ⚠️ 生產環境不要用 http.ListenAndServe，它沒超時
    // 用下面 Server struct 的寫法
    http.ListenAndServe(":8080", nil)
}

Go 1.22+ 的新路由語法

過去 Go 的路由「只能比 prefix」— 不支援 method、不支援 path variable。Go 1.22 起 http.ServeMux 原生支援這些，常見應用不再需要第三方 router（gorilla/mux、chi 等）。

// 新語法：method + path + variable
mux := http.NewServeMux()

// 方法限定
mux.HandleFunc("GET /users", listUsers)
mux.HandleFunc("POST /users", createUser)

// 路徑變數用 {name}
mux.HandleFunc("GET /users/{id}", getUser)
mux.HandleFunc("DELETE /users/{id}", deleteUser)

// 在 handler 內取出變數
func getUser(w http.ResponseWriter, r *http.Request) {
    id := r.PathValue("id")       // "42"
    // ...
}

// 萬用字元 {path...} 匹配剩餘路徑
mux.HandleFunc("GET /files/{path...}", serveFile)
// GET /files/a/b/c → path = "a/b/c"

// Host + path 組合
mux.HandleFunc("api.example.com/v1/users", apiHandler)

// 精確匹配用 trailing {$}
mux.HandleFunc("GET /users/{$}", listUsers)
// 只匹配 /users/，不匹配 /users/123
優先級規則 新 mux 採用「最具體優先」規則（不再是註冊順序）：GET /users/{id} 比 /users/{path...} 更具體。衝突時編譯/啟動會直接 panic，不會默默選錯。

Request & ResponseWriter

func handler(w http.ResponseWriter, r *http.Request) {
    // === Request 常用欄位 ===
    r.Method              // "GET" / "POST" / ...
    r.URL.Path            // "/users/42"
    r.URL.RawQuery        // "a=1&b=2"
    r.URL.Query()         // url.Values（map[string][]string）
    r.URL.Query().Get("a") // "1"（只取第一個）
    r.PathValue("id")      // Go 1.22+ 的路徑變數
    r.Header.Get("Authorization")
    r.Cookie("session")   // 取 cookie
    r.RemoteAddr          // client IP:port（注意 proxy）
    r.Body                // io.ReadCloser — 讀完記得 Close
    r.Context()           // ctx — client 斷線會自動取消

    // === ResponseWriter 使用 ===
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(http.StatusCreated)   // 201
    w.Write(data)                        // 寫 body

    // ⚠️ 正確順序：先 Header，再 WriteHeader，最後 Write
    // 一旦呼叫 Write，header 就發出去了，之後改 header 無效
}

中介軟體模式 · 不用 framework

// middleware 的型別：接收 handler、回傳 handler
type Middleware func(http.Handler) http.Handler

// 範例：記錄每個請求
func logging(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        next.ServeHTTP(w, r)
        log.Printf("%s %s %v", r.Method, r.URL.Path, time.Since(start))
    })
}

// 範例：recover panic
func recovery(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if rec := recover(); rec != nil {
                log.Printf("panic: %v\n%s", rec, debug.Stack())
                http.Error(w, "internal error", 500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

// 範例：驗證
func auth(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        token := r.Header.Get("Authorization")
        user, err := verify(token)
        if err != nil {
            http.Error(w, "unauthorized", 401)
            return
        }
        // 把 user 放進 ctx 傳下去
        ctx := context.WithValue(r.Context(), userKey, user)
        next.ServeHTTP(w, r.WithContext(ctx))
    })
}

// 串接
func chain(h http.Handler, mws ...Middleware) http.Handler {
    for i := len(mws) - 1; i >= 0; i-- {
        h = mws[i](h)
    }
    return h
}

// 使用
handler := chain(mux, recovery, logging, auth)
srv.Handler = handler

優雅關機（Graceful Shutdown）

生產部署必備 — 收到 SIGTERM 後讓進行中的請求完成，不是硬中斷。

func main() {
    srv := &http.Server{
        Addr:    ":8080",
        Handler: mux,
        // ... timeouts ...
    }

    // 非阻塞啟動
    go func() {
        if err := srv.ListenAndServe(); err != http.ErrServerClosed {
            log.Fatalf("server error: %v", err)
        }
    }()

    // 等信號（Go 1.16+ 簡化版）
    ctx, stop := signal.NotifyContext(context.Background(),
        syscall.SIGINT, syscall.SIGTERM)
    defer stop()
    <-ctx.Done()

    log.Println("shutting down...")

    // 給 10 秒讓進行中請求完成
    shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    if err := srv.Shutdown(shutdownCtx); err != nil {
        log.Printf("forced shutdown: %v", err)
    }
}

實戰模板

TEMPLATE · 完整 REST API（無 framework） 
package main

import (
    "context"
    "encoding/json"
    "log"
    "net/http"
    "os/signal"
    "syscall"
    "time"
)

type Server struct {
    users map[string]User
    mu    sync.RWMutex
}

type User struct {
    ID   string `json:"id"`
    Name string `json:"name"`
}

func (s *Server) getUser(w http.ResponseWriter, r *http.Request) {
    id := r.PathValue("id")
    s.mu.RLock()
    u, ok := s.users[id]
    s.mu.RUnlock()
    if !ok {
        http.Error(w, "not found", 404)
        return
    }
    writeJSON(w, 200, u)
}

func (s *Server) createUser(w http.ResponseWriter, r *http.Request) {
    var u User
    if err := json.NewDecoder(r.Body).Decode(&u); err != nil {
        http.Error(w, err.Error(), 400)
        return
    }
    s.mu.Lock()
    s.users[u.ID] = u
    s.mu.Unlock()
    writeJSON(w, 201, u)
}

func writeJSON(w http.ResponseWriter, code int, v any) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(code)
    json.NewEncoder(w).Encode(v)
}

func main() {
    s := &Server{users: make(map[string]User)}

    mux := http.NewServeMux()
    mux.HandleFunc("GET /users/{id}", s.getUser)
    mux.HandleFunc("POST /users", s.createUser)

    srv := &http.Server{
        Addr:         ":8080",
        Handler:      chain(mux, recovery, logging),
        ReadTimeout:  5 * time.Second,
        WriteTimeout: 10 * time.Second,
        IdleTimeout:  60 * time.Second,
    }

    go srv.ListenAndServe()

    ctx, stop := signal.NotifyContext(context.Background(),
        syscall.SIGINT, syscall.SIGTERM)
    defer stop()
    <-ctx.Done()

    sctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()
    srv.Shutdown(sctx)
}

本章速記

  1. 永遠 defer Close(resp.Body)：否則連線洩漏。
  2. 讀完 Body 再關：讓 keep-alive 能重用連線。
  3. http.DefaultClient 沒超時：生產環境自建 Client。
  4. 用 NewRequestWithContext：支援取消與超時。
  5. 生產 Server 必設 timeouts：Read/Write/Idle/ReadHeader。
  6. Go 1.22+ mux 支援 method 與 path variable：大多數情況不用 framework。
  7. r.PathValue("name")：取路徑變數。
  8. WriteHeader 前設 Header：一旦 Write 就鎖定 header。
  9. Middleware 模式：func(Handler) Handler。
  10. recover middleware 是必備：一個 panic 別拖垮整個 server。
  11. signal.NotifyContext：信號轉 context，graceful shutdown 的現代寫法。
  12. srv.Shutdown(ctx)：優雅關機，停止接新請求等舊請求完成。
